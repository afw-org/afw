33 static void
34 impl_over_list(
35  afw_function_execute_t *x,
36  impl_call_over_list_cb_t callback,
37  void *data)
38 {
39 
40  afw_size_t functor_argc;
41  const afw_value_t * (*functor_argv);
42  const afw_iterator_t *iterator;
43  impl_call_over_list_cb_e_t e;
44 
45  /* Initialize param. */
46  afw_memory_clear(&e);
47  e.function = x->function;
48  e.data = data;
49  e.p = x->p;
50  e.scope = x->scope;
51 
52  /* Create skeleton call. */
53  functor_argc = x->argc - 1;
54  functor_argv = afw_pool_calloc(e.p, sizeof(afw_value_t *) * x->argc,
55  e.scope);
56  functor_argv[0] = afw_function_evaluate_function_parameter(
57  x->argv[1], e.p, e.scope);
58  e.functor = afw_value_call_create(NULL, functor_argc, functor_argv,
59  e.p, e.scope);
60 
61  /*
62  * Evaluate argv[n+1] to the new argv[n]. Replace first typed list with
63  * pointer to an allocated value of the same data type.
64  */
65  for (e.n = 1; e.n <= functor_argc; e.n++) {
66  functor_argv[e.n] = afw_value_evaluate(x->argv[e.n + 1], e.p, e.scope);
67  if (!e.entry_arg_ptr && afw_value_is_list(functor_argv[e.n])) {
68  e.entry_arg_ptr = &functor_argv[e.n];
69  e.list = ((const afw_value_list_t *)*e.entry_arg_ptr)->internal;
70  *e.entry_arg_ptr = NULL;
71  e.data_type = afw_list_get_data_type(e.list, e.scope);
72  if (e.data_type) {
73  *e.entry_arg_ptr = (const afw_value_t *)
74  afw_value_evaluated_allocate(e.data_type, e.p, e.scope);
75  }
76  }
77  }
78 
79  /* There must be a typed list in parameter list. */
80  if (!e.entry_arg_ptr) {
81  AFW_THROW_ERROR_Z(arg_error, "Missing typed list arg", e.scope);
82  }
83 
84  /* Call function with each entry in typed list as a single value. */
85  if (e.data_type) {
86  for (iterator = NULL;;) {
87  afw_list_get_next_internal(e.list, &iterator, NULL,
88  &e.entry_internal, e.scope);
89  if (!e.entry_internal) {
90  break;
91  }
92  memcpy(AFW_VALUE_INTERNAL(*e.entry_arg_ptr), e.entry_internal,
93  e.data_type->c_type_size);
94  e.entry_result = afw_value_evaluate(e.functor, e.p, e.scope);
95 
96  if (!callback(&e))
97  {
98  break;
99  }
100  }
101  }
102 
103  /* Call function with each entry in untyped list as a single value. */
104  else {
105  for (iterator = NULL;;) {
106  *e.entry_arg_ptr = afw_list_get_next_value(
107  e.list, &iterator, NULL, e.scope);
108  if (!*e.entry_arg_ptr) {
109  break;
110  }
111  e.entry_result = afw_value_evaluate(e.functor, e.p, e.scope);
112  if (!callback(&e))
113  {
114  break;
115  }
116  }
117  }
118 }

162 static const afw_value_t *
163 impl_bag_of_bag(
164  afw_function_execute_t *x,
165  afw_boolean_t all_1, afw_boolean_t all_2)
166 {
167  const afw_value_list_t *list1, *list2, *listx;
168  const afw_data_type_t *data_type_1, *data_type_2;
169  const afw_iterator_t *iterator1, *iterator2;
170  const void *internal1, *internal2;
171  void *e1, *e2;
172  const afw_value_t * f_argv[3];
173  const afw_value_t *v;
174  const afw_value_t *call;
175  afw_boolean_t is_true, any_1, any_2;
176 
177  /* The first arg is the function to call, and other 2 are typed lists. */
178  f_argv[0] = afw_function_evaluate_function_parameter(
179  x->argv[1], x->p, x->scope);
180  call = afw_value_call_create(NULL, 2, &f_argv[0], x->p, x->scope);
181 
182  AFW_FUNCTION_EVALUATE_REQUIRED_DATA_TYPE_PARAMETER(list1, 2, list);
183  AFW_FUNCTION_EVALUATE_REQUIRED_DATA_TYPE_PARAMETER(list2, 3, list);
184 
185  /* Work off of any* variables. */
186  any_1 = !all_1;
187  any_2 = !all_2;
188 
189  /* If any_of_all, reverse bags and any* values. */
190  if (!all_1 && all_2) {
191  listx = list1;
192  list1 = list2;
193  list2 = listx;
194  any_1 = !all_2;
195  any_2 = !all_1;
196  }
197 
198  /* Get required data type from each list. */
199  data_type_1 = afw_list_get_data_type(list1->internal, x->scope);
200  data_type_2 = afw_list_get_data_type(list2->internal, x->scope);
201  if (!data_type_1 || !data_type_2) {
202  AFW_THROW_ERROR_Z(general,
203  "list1 and list2 must both be typed",
204  x->scope);
205  }
206 
207  /* Allocate a single value to use for each bag1 entry. */
208  f_argv[1] = (afw_value_t *)afw_value_evaluated_allocate(
209  data_type_1, x->p, x->scope);
210  e1 = (void *)&((afw_value_evaluated_t *)f_argv[1])->internal;
211 
212  /* Allocate a single value to use for each bag2 entry. */
213  f_argv[2] = (afw_value_t *)afw_value_evaluated_allocate(
214  data_type_2, x->p, x->scope);
215  e2 = (void *)&((afw_value_evaluated_t *)f_argv[2])->internal;
216 
217  /* Call function for each combination of bag1 and bag2 entries. */
218  is_true = true;
219 
220  for (iterator1 = NULL;;) {
221  afw_list_get_next_internal(list1->internal,
222  &iterator1, NULL, &internal1, x->scope);
223  if (!internal1) {
224  break;
225  }
226  is_true = true;
227  for (iterator2 = NULL;;) {
228  afw_list_get_next_internal(list2->internal,
229  &iterator2, NULL, &internal2, x->scope);
230  if (!internal2) {
231  break;
232  }
233  memcpy(e1, internal1, data_type_1->c_type_size);
234  memcpy(e2, internal2, data_type_2->c_type_size);
235  v = afw_value_evaluate(call, x->p, x->scope);
236  if (!afw_value_is_boolean(v)) {
237  AFW_THROW_ERROR_Z(arg_error,
238  "First argument must be a boolean function", x->scope);
239  }
240 
241  /* If true, indicate and break if any_2 enough. */
242  if (((const afw_value_boolean_t *)v)->internal) {
243  is_true = true;
244  if (any_2) break;
245  }
246 
247  /* If false, indicate and break if all must be true. */
248  else {
249  is_true = false;
250  if (!any_2) break;
251  }
252  }
253 
254  /* If any_1 is ok, return true. */
255  if (any_1 && is_true) break;
256 
257  /* If all must be true, return false. */
258  if (!any_1 && !is_true) break;
259  }
260 
261  /* If made it though loop, result is whatever is_true is. */
262  return (is_true) ? afw_value_true : afw_value_false;
263 }

748  * Adaptive function: reduce
749  *
750  * afw_function_execute_reduce
751  *
752  * See afw_function_bindings.h for more information.
753  *
754  * Reduce calls functor for each value in list with two parameters, accumulator
755  * and value, and must return a value of any dataType. Parameter accumulator is
756  * the reduce() accumulator parameter value on first call and the return value
757  * of previous functor() call on subsequent calls. The dataType of the return
758  * value should normally be the same as accumulator, but this is not required.
759  *
760  * This function is pure, so it will always return the same result
761  * given exactly the same parameters and has no side effects.
762  *
763  * Declaration:
764  *
765  * function reduce(
766  * functor: (function (accumulator: any, value: any): any),
767  * accumulator: any,
768  * list: list
769  * ): any;
770  *
771  * Parameters:
772  *
773  * functor - (function (accumulator: any, value: any): any) This function is
774  * called for each value in a list. The returned value is passed as the
775  * accumulator parameter on the next call to functor().
776  *
777  * accumulator - (any dataType) This is an initial accumulator value passed
778  * to functor(). Normally, the dataType of accumulator will be the
779  * dataTape for the reduce() return value, but this is not required.
780  *
781  * list - (list) This is a list to be reduced.
782  *
783  * Returns:
784  *
785  * (any dataType) This is the final return value from functor() or the
786  * accumulator parameter value if list is empty.
787  */
788 const afw_value_t *
789 afw_function_execute_reduce(
790  afw_function_execute_t *x)

793  const afw_value_t *accumulator;
794  const afw_iterator_t *iterator;
795  const afw_value_t * f_argv[3];
796  const afw_value_t *call;
797 
798  f_argv[0] = afw_function_evaluate_function_parameter(
799  x->argv[1], x->p, x->scope);
800  call = afw_value_call_create(NULL, 2, &f_argv[0], x->p, x->scope);
801  AFW_FUNCTION_EVALUATE_REQUIRED_PARAMETER(accumulator, 2);
802  AFW_FUNCTION_EVALUATE_REQUIRED_DATA_TYPE_PARAMETER(list, 3, list);
803 
804  for (iterator = NULL;;) {
805  f_argv[2] = afw_list_get_next_value(list->internal, &iterator,
806  x->p, x->scope);
807  if (!f_argv[2]) {

809  }
810  f_argv[1] = accumulator;
811  accumulator = afw_value_evaluate(call, x->p, x->scope);
812  }

829 static afw_size_t
830 impl_partition(
831  impl_sort_ctx_t *ctx,
832  afw_size_t low,
833  afw_size_t high)
834 {
835  const void *pivot;
836  const void *value;
837  const afw_value_t *return_value;
838  afw_size_t i, j;
839 
840  /* Start pivot from high. */
841  pivot = ctx->values[high];
842 
843  /* Put everything to right and left of pivot based on compare. */
844  memcpy(&(ctx->args[2]->internal), pivot, ctx->c_type_size);
845  for (i = low, j = low; j <= high - 1; j++)
846  {
847  /* Call compareFunction with ctx->values[j] and pivot. */
848  memcpy(&(ctx->args[1]->internal), ctx->values[j], ctx->c_type_size);
849  return_value = afw_value_evaluate(ctx->compareFunction,
850  ctx->p, ctx->scope);
851  AFW_VALUE_ASSERT_IS_DATA_TYPE(return_value, boolean, ctx->scope);
852 
853  /*
854  * if compareFunction(values[j], pivot) is true, swap values[i] and
855  * value[j] then increment i
856  */
857  if (((const afw_value_boolean_t *)return_value)->internal)
858  {
859  value = ctx->values[i];
860  ctx->values[i] = ctx->values[j];
861  ctx->values[j] = value;
862  i++;
863  }
864  }
865 
866  /* Swap values[new pivot index] and values[high] */
867  value = ctx->values[i];
868  ctx->values[i] = ctx->values[high];
869  ctx->values[high] = value;
870 
871  /* Return new pivot index. */
872  return i;
873 }

877  impl_sort_ctx_t *ctx,

890  }

894  * Adaptive function: sort
895  *
896  * afw_function_execute_sort
897  *
898  * See afw_function_bindings.h for more information.
899  *
900  * This produces a list with values sorted based on result of compareFunction.
901  * The compareFunction is passed two values from the list and must return an
902  * integer less than 0 if the first value is less than the second value, 0 if
903  * they are equal, and a integer greater than 0 if the first value is greater
904  * than the second value.
905  *
906  * This function is pure, so it will always return the same result
907  * given exactly the same parameters and has no side effects.
908  *
909  * Declaration:
910  *
911  * function sort(
912  * compareFunction: (function (value1: any, value2: any): integer),
913  * list: list
914  * ): list;
915  *
916  * Parameters:
917  *
918  * compareFunction - (function (value1: any, value2: any): integer) This
919  * function is called with two value from list.
920  *
921  * list - (list) This is the list to sort.
922  *
923  * Returns:
924  *
925  * (list) This the the resulting sorted list.
926  */
927 const afw_value_t *
928 afw_function_execute_sort(
929  afw_function_execute_t *x)
930 {
931  const afw_value_list_t *list;
932  const afw_list_t *result_list;
933  const afw_data_type_t *data_type;
934  const afw_iterator_t *iterator;
935  const void **value;
936  impl_sort_ctx_t ctx;
937 
938  /* Initialize sort ctx. */
939  afw_memory_clear(&ctx);
940  ctx.p = x->p;
941  ctx.scope = x->scope;
942 
943  /* The first arg is the function to call, and other 2 are typed lists. */
944  ctx.args[0] = (afw_value_evaluated_t *)
945  afw_function_evaluate_function_parameter(x->argv[1], ctx.p, ctx.scope);
946  ctx.compareFunction = afw_value_call_create(NULL,
947  2, (const afw_value_t * const *)&ctx.args[0], ctx.p, ctx.scope);
948  AFW_FUNCTION_EVALUATE_REQUIRED_DATA_TYPE_PARAMETER(list, 2, list);
949 
950  /* Get the data type and count. If count is 0, return empty list. */
951  data_type = afw_list_get_data_type(list->internal, ctx.scope);
952  if (!data_type) {
953  AFW_THROW_ERROR_Z(general,
954  "sort() requires list to be typed", ctx.scope);
955  }
956  ctx.c_type_size = data_type->c_type_size;
957  ctx.count = afw_list_get_count(list->internal, ctx.scope);
958  if (ctx.count == 0) {
959  return data_type->empty_list_value;
960  }
961 
962  /* Allocate two values of the correct data type. */
963  ctx.args[1] = afw_value_evaluated_allocate(data_type, ctx.p, ctx.scope);
964  ctx.args[2] = afw_value_evaluated_allocate(data_type, ctx.p, ctx.scope);
965 
966  /* Make array of pointers to internal values. */
967  ctx.values = afw_pool_malloc(ctx.p, sizeof(void *) * ctx.count, ctx.scope);
968  for (iterator = NULL, value = ctx.values;
969  afw_list_get_next_internal(list->internal, &iterator,
970  NULL, value, ctx.scope);
971  value++);
972 
973  /* Sort. */
974  if (ctx.count > 0) {
975  impl_quick_sort(&ctx, 0, ctx.count - 1);
976  }
977 
978  /* Return sorted list. */
979  result_list = afw_list_create_wrapper_for_array(
980  ctx.values, true, data_type, ctx.count, ctx.p, ctx.scope);
981  return afw_value_create_list(result_list, ctx.p, ctx.scope);
982 }
