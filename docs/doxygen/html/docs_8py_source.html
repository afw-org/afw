253 # simple little function that takes paragraphs that likely came from XML and
254 # splits them on \n\n, into logical paragraphs. Then, on each paragraph, it
255 # splits on whitespace and rejoins with a single space. This is to ensure that
256 # the paragraphs are formatted nicely in the HTML output.
257 def get_paras(text):
258  if not text:
259  return None
260 
261  paras = re.split(r"\n +\n", text)
262  return "\n\n".join(map(lambda x: " ".join(x.split()), paras))
