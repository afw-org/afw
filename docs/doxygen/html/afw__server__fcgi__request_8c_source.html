60 /* Create an adaptive object in memory. */
61 afw_server_fcgi_internal_request_t *
62 afw_server_fcgi_internal_create_request(
63  afw_server_fcgi_internal_t *server,
64  FCGX_Request *fcgx_request,
65  afw_scope_t * scope)
66 {
67  afw_server_fcgi_internal_request_t *self = NULL;
68  const afw_value_t *value;
69  const afw_utf8_z_t *length_z;
70  const afw_utf8_octet_t *c;
71  afw_size_t len;
72 
73  /* Allocate memory for self. */
74  self = afw_scope_calloc_type(afw_server_fcgi_internal_request_t, scope);
75 
76  /* Initialize Self. */
77  self->pub.inf = &impl_afw_request_inf;
78  self->pub.scope = scope;
79  self->fcgx_request = fcgx_request;
80  self->server = server;
81  self->pub.read_content_cb = impl_read_content_cb;
82  self->pub.write_content_cb = impl_write_content_cb;
83 
84  /* Create properties object and request:: qualifier. */
85  self->pub.properties = afw_server_fcgi_internal_create_properties_object(
86  self, scope);
87  afw_runtime_scope_set_object(self->pub.properties, true, scope);
88  afw_scope_push_qualifier_object(&afw_s_request, self->pub.properties,
89  true, scope->p, scope);
90 
91  /* Get request method. */
92  value = afw_object_get_property(self->pub.properties,
93  &AFW_REQUEST_s_PN_REQUEST_METHOD, scope);
94  if (!value) {
95  AFW_THROW_ERROR_Z(general,
96  AFW_REQUEST_Q_PN_REQUEST_METHOD " property required.", scope);
97  }
98  self->pub.method = afw_value_as_utf8(value, scope->p, scope);
99 
100  /* Get request URI. */
101  self->pub.uri = afw_object_old_get_property_as_utf8(self->pub.properties,
102  &AFW_REQUEST_s_PN_PATH_INFO, scope->p, scope);
103  if (!self->pub.uri || self->pub.uri->len == 0) {
104  self->pub.uri = afw_object_old_get_property_as_utf8(self->pub.properties,
105  &AFW_REQUEST_s_PN_REQUEST_URI, scope->p, scope);
106  if (self->pub.uri && self->pub.uri->len != 0) {
107  for (c = self->pub.uri->s, len = self->pub.uri->len;
108  len > 0; c++, len--)
109  {
110  if (*c == '?') {
111  self->pub.uri = afw_utf8_create(
112  self->pub.uri->s, c - self->pub.uri->s,
113  scope->p, scope);
114  }
115  }
116  }
117  }
118  if (!self->pub.uri || self->pub.uri->len == 0) {
119  AFW_THROW_ERROR_Z(general,
120  AFW_REQUEST_Q_PN_PATH_INFO " or " AFW_REQUEST_Q_PN_REQUEST_URI
121  " property required", scope);
122  }
123 
124  /* Get request query string. */
125  value = afw_object_get_property(self->pub.properties,
126  &AFW_REQUEST_s_PN_QUERY_STRING, scope);
127  if (!value) {
128  AFW_THROW_ERROR_Z(general,
129  AFW_REQUEST_Q_PN_QUERY_STRING " property required.", scope);
130  }
131  self->pub.query_string = afw_value_as_utf8(value, scope->p, scope);
132 
133  /* Get request content type. */
134  self->pub.content_type = afw_object_old_get_property_as_utf8(
135  self->pub.properties,
136  &AFW_REQUEST_s_PN_CONTENT_TYPE, scope->p, scope);
137 
138  /* Get request content length. */
139  value = afw_object_get_property(self->pub.properties,
140  &AFW_REQUEST_s_PN_CONTENT_LENGTH, scope);
141  if (value) {
142  length_z = afw_value_as_utf8_z(value, scope->p, scope);
143  self->pub.content_length = atoi(length_z);
144  }
145 
146  /* Get request accept header. */
147  value = afw_object_get_property(self->pub.properties,
148  &AFW_REQUEST_s_PN_HTTP_ACCEPT, scope);
149  if (value) {
150  self->pub.accept = afw_utf8_parse_csv(
151  afw_value_as_utf8(value, scope->p, scope),
152  scope->p, scope);
153  }
154 
155  /* Return new adaptor. */
156  return self;
157 }

196 /*
197  * Implementation of method read_request_body of interface afw_request.
198  */
199 void
200 impl_afw_request_read_raw_request_body(
201  const afw_request_t * instance,
202  afw_size_t buffer_size,
203  void * buffer,
204  afw_size_t * size,
205  afw_boolean_t * more_to_read,
206  afw_scope_t * scope)
207 {
208  afw_server_fcgi_internal_request_t *self =
209  (afw_server_fcgi_internal_request_t *)instance;
210  int c;
211  unsigned char *b = buffer;
212 
213  /* Make sure in correct state then set it. */
214  if (self->state > afw_request_state_content_read) {
215  AFW_THROW_ERROR_Z(general,
216  "read_request_body() called out of order", scope);
217  }
218  self->state = afw_request_state_content_read;
219 
220 
221  /* Read into buffer until buffer full or EOF. */
222  *more_to_read = 1;
223  for (*size = 0; *size < buffer_size; (*size)++) {
224  c = FCGX_GetChar(self->fcgx_request->in);
225  if (c == EOF) {
226  *more_to_read = 0;
227  break;
228  }
229  *(b++) = c;
230  }
231 }

236  * Implementation of method set_response_status_code of interface afw_request.
237  */
238 void
239 impl_afw_request_set_response_status_code(
240  const afw_request_t * instance,
241  const afw_utf8_t * code,
242  const afw_utf8_t * reason,
243  afw_scope_t * scope)
244 {
245  afw_server_fcgi_internal_request_t *self =
246  (afw_server_fcgi_internal_request_t *)instance;
247  int rv = 0;
248 
249  /* Ignore status if already past state. */
250  if (self->state >= afw_request_state_status_set) {
251  return;
252  }
253  self->state = afw_request_state_status_set;
254 
255  /*
256  * Set response status. This is done in FastCGI by setting a "Status"
257  * header that is examined but not sent to the client client.
258  */
259  if (reason) {
260  rv = FCGX_FPrintF(self->fcgx_request->out,
261  "Status:%" AFW_UTF8_FMT " %" AFW_UTF8_FMT
262  AFW_CRLF,
263  AFW_UTF8_FMT_ARG(code), AFW_UTF8_FMT_ARG(reason));
264  }
265  else {
266  rv = FCGX_FPrintF(self->fcgx_request->out,
267  "Status:%" AFW_UTF8_FMT AFW_CRLF,
268  AFW_UTF8_FMT_ARG(code));
269  }
270  if (rv == -1) {
271  AFW_THROW_ERROR_Z(general,
272  "FCGX_FPrintF() failed", scope);
273  }
274 }

312  * Implementation of method write_response_body of interface afw_request.
313  */
314 void
315 impl_afw_request_write_raw_response_body(
316  const afw_request_t * instance,
317  apr_size_t size,
318  const void * buffer,
319  afw_scope_t * scope)
320 {
321  afw_server_fcgi_internal_request_t *self =
322  (afw_server_fcgi_internal_request_t *)instance;
323  apr_size_t remaining;
324  const char *next;
325  int rv;
326  int n;
327 
328  /* Make sure in correct state. */
329  if (self->state > afw_request_state_response_written) {
330  AFW_THROW_ERROR_Z(general,
331  "write_response_body() called out of order", scope);
332  }
333 
334  /* If this is the first write_response, write AFW_CRLF. */
335  if (self->state < afw_request_state_response_written) {
336  rv = FCGX_PutStr(AFW_CRLF, AFW_CRLF_STRLEN,
337  self->fcgx_request->out);
338  if (rv == -1) {
339  AFW_THROW_ERROR_RV_Z(client_closed, fcgi, rv,
340  "FCGX_PutStr() failed", scope);
341  }
342  }
343 
344  /* Set state*/
345  self->state = afw_request_state_response_written;
346 
347  /* Write response. */
348  for (next = (const char *)buffer, remaining = size;
349  remaining > 0;
350  next = next + n,
351  remaining = remaining - n)
352  {
353  n = (remaining > INT_MAX) ? INT_MAX : (int)remaining;
354  rv = FCGX_PutStr(next, n, self->fcgx_request->out);
355  if (rv == -1) {
356  AFW_THROW_ERROR_RV_Z(client_closed, fcgi, rv,
357  "FCGX_PutStr() failed", scope);
358  }
359  }
360 }
