208  * notification()
209  *
210  * This routine can be called by any component, and is used to alert the user that a notification is

216  let messageStr = message;
217  let notifications = this.state.notifications;
218 
219  if (type === "error") {
220  /* if the user passed in an Error object, look for a response, which would be passed back by the server */
221  if (message && (message instanceof Error)) {
222  if (message.response && message.response.data) {
223  if (message.response.data.message) {
224  messageStr = message.response.data.message;
225  notifications = [...notifications, {...message.response.data, timestamp: new Date()}];
226  } else {
227  messageStr = message.message;
228  notifications = [...notifications, {...message, timestamp: new Date()}];
229  }
230  } else {
231  notifications = [...notifications, {
232  timestamp: new Date(), ...message
233  }];
234  }
235  } else if (typeof(message) === "object") {
236  /* \fixme there should be an adaptive object of type error message here */
237  messageStr = message["message"];
238  notifications = [...notifications, {message: messageStr, error: message, timestamp: new Date()}];
