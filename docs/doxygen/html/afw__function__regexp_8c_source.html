35  * listing?: any

37  *
38  * Parameters:
39  *
40  * source - (regexp) regexp string to compile.
41  *
42  * listing - (optional any dataType) If specified, a compiler listing is
43  * produced instead of an unevaluated expression value.
44  *
45  * This parameter can be an integer between 0 and 10 of a string that is
46  * used for indentation. If 0 is specified, no whitespace is added to the
47  * resulting string. If 1 through 10 is specified, that number of spaces
48  * is used.
49  *
50  * Returns:
51  *
52  * (unevaluated)
53  */
