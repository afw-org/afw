18  *
19  * Then calling its requirejs function:
20  *
21  * window.requirejs(["mylib"], function(myLib) {
22  * myLib.hello();
23  * });
24  *
25  */

27  const script = document.createElement("script");
28  script.src = url; /* /path/file.js */
29 
30  return new Promise((resolve, reject) => {

32  /* now you can invoke: window.myFunc(); defined inside file */
33  resolve();
34  };
35 
36  script.onerror = () => {
37  reject();
38  };
39 
40  document.getElementsByTagName("head")[0].appendChild(script);

42 }
