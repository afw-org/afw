52 export const useBreadcrumbs = (root, reducer) => {
53 
54  const {pathname} = useLocation();
55 
56  let breadcrumbItems = [];
57 
58  if (pathname) {
59  if (root)
60  breadcrumbItems.push(root);
61 
62  if (reducer)
63  breadcrumbItems = reducer(breadcrumbItems);
64 
65  else {
66  const parts = pathname.split("/");
67  let link = root ? root.link : "";
68  for (let i = 2; i < parts.length; i++) {

70  breadcrumbItems.push({
71  text: parts[i],
72  link

74  }
75  }
76  }

78  return breadcrumbItems;

81 /**
82  * useLocalJSONStorage()
83  *
84  * A hook that provides the user with the value of Local Storage identifier
85  * (within "afw") and provides notifications/update routines when they are
86  * changed.
87  *
88  * All values in/out are in JSON formatted strings.
89  *
90  * @param {string} identifier
91  * @param {any} initialValue
92  */
93 export const useLocalJSONStorage = (identifier, initialValue) => {
94 
95  const [storage, setStorage] = useState();
96  const [error, setError] = useState();
97 
98  useEffect(() => {
99  if (identifier) {
100  let afwStorage = localStorage.getItem("afw");
101  if (!afwStorage) {
102  /* if we don't have "afw" in storage yet, go ahead and create it */
103  localStorage.setItem("afw", "{}");
104  afwStorage = "{}";
105  }
106 
107  try {
108  /* parse the JSON result from localStorage back into a value */
109  afwStorage = JSON.parse(afwStorage);
110  if (!afwStorage[identifier] && initialValue) {
111  afwStorage[identifier] = initialValue;
112  localStorage.setItem("afw", JSON.stringify(afwStorage));
113  }
114  setStorage(afwStorage[identifier]);
115  } catch (e) {
116  /* incognito mode? size limit? */
117  setError(e);
118  }
119  }
120  }, [identifier, initialValue]);
121 
122  const onChanged = useCallback((s) => {
123  try {
124  let afwStorage = localStorage.getItem("afw");
125  if (afwStorage) {
126  afwStorage = JSON.parse(afwStorage);
127 
128  afwStorage[identifier] = s;
129  localStorage.setItem("afw", JSON.stringify(afwStorage));
130 
131  setStorage(s);
132  }
133  } catch (e) {
134  /* incognito mode? size limit? */
135  setError(e);
136  }
137  }, [identifier]);
138 
139  return { storage, onChanged, error };
140 };
