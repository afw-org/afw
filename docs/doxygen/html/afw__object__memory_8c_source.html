32 afw_object_create_with_options(
33  int options, const afw_pool_t *p, afw_scope_t *scope)
34 {
35  afw_object_internal_memory_object_t *self;
36 
37  /* If neither unmanaged or cede_p, create subpool for object. */
38  if (options &
39  (AFW_OBJECT_MEMORY_OPTION_unmanaged | AFW_OBJECT_MEMORY_OPTION_cede_p))
40  {
41  p = afw_pool_create(p, scope);
42  }
43 
44  /* Allocate memory for self and initialize. */
45  self = afw_pool_calloc_type(p, afw_object_internal_memory_object_t, scope);
46  self->pub.inf = &impl_afw_object_inf;
47  self->pub.p = p;
48  self->unmanaged = AFW_OBJECT_MEMORY_OPTION_IS(options, unmanaged);
49  self->clone_on_set = AFW_OBJECT_MEMORY_OPTION_IS(options, clone_on_set);
50  self->reference_count = (self->unmanaged) ? 0 : 1;
51  self->setter.inf = &impl_afw_object_setter_inf;
52  self->setter.object = (const afw_object_t *)self;
53 
54  /* Return new object. */
55  return (const afw_object_t *)self;
56 }

81  self->pub.inf = &impl_afw_object_inf;
82  self->pub.p = p;
83  self->pub.meta.embedding_object = embedding_object;
84  self->pub.meta.id = property_name;
85  self->managed_by_entity = true;
86  self->reference_count = 1;
87  self->setter.inf = &impl_afw_object_setter_inf;
88  self->setter.object = (const afw_object_t *)self;
89  self->clone_on_set = embedder->clone_on_set;
90 
91  /* Set embedded object as property of embedding object. */

124  * Implementation of method release of interface afw_object.
125  */
126 void
127 impl_afw_object_release(
128  const afw_object_t * instance,
129  afw_scope_t * scope)
130 {
131  afw_object_internal_memory_object_t * self =
132  (afw_object_internal_memory_object_t *)instance;
133  const afw_object_t *entity;
134 
135  /* If unmanaged, just return. */
136  if (self->unmanaged) return;
137 
138  /*
139  * If embedded object managed by parent, call parent's release and
140  * return.
141  */
142  if (self->managed_by_entity) {
143  AFW_OBJECT_GET_ENTITY(entity, instance);
144  afw_object_release(entity, scope);
145  return;
146  }
147 
148  /*
149  * If managing self, decrement reference count. If count is zero, release
150  * pool.
151  */
152  if (afw_atomic_integer_decrement(&self->reference_count) == 0) {
153  afw_pool_release(instance->p, scope);
154  }
155 }

224 const afw_value_t *
225 impl_afw_object_get_next_property(
226  const afw_object_t * instance,

229  afw_scope_t * scope)

233  afw_object_internal_name_value_entry_t *e;
234 
235  /* If iterator is NULL, get first. */
236  if (!*iterator) {
237  e = self->first_property;
238  }
239 
240  /* If iterator is not NULL, get next. */
241  else {
242  e = (afw_object_internal_name_value_entry_t *)*iterator;
243  e = e->next;
244  }
245 
246  /* Skip any deleted properties. */
247  for (; e && !e->value; e = e->next);
248 
249  /* Set iterator to entry. */
250  *iterator = (afw_iterator_t *)e;
251 
252  /* If e NULL, return not found. */
253  if (!e)
254  {
255  if (property_name) {
256  *property_name = NULL;
257  }
258  return NULL;
259  }
260 
261  /* Return entries property name and value. */
262  if (property_name) {
263  *property_name = e->name;
264  }
265  return e->value;
266 }

328 impl_afw_object_setter_set_property(

331  const afw_value_t * value,

337  afw_object_internal_name_value_entry_t *e;

341 
342  for (e = self->first_property,final_e = NULL; e; e = e->next) {
343  final_e = e;
344  if (afw_utf8_equal(e->name, property_name)) {
345  e->value = value;
346  return;
347  }
348  }
349  if (!value) return; /* Delete property just return. */
350  e = afw_pool_calloc_type(instance->object->p,
351  afw_object_internal_name_value_entry_t, scope);
352  e->name = (property_name) ? property_name : &afw_s_a_empty_string;
353 
363  e->value = (self->clone_on_set)
364  ? afw_value_clone(value, self->pub.p, scope)
365  : value;
366  if (final_e) {
367  final_e->next = e;
368  }
369  else {
370  self->first_property = e;
371  }
