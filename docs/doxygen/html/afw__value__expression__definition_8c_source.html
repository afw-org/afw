68 impl_afw_value_optional_evaluate(
69  const afw_value_t * instance,
70  const afw_pool_t * p,
71  afw_scope_t * scope)
72 {
73  const afw_value_expression_definition_t *self =
74  (const afw_value_expression_definition_t *)instance;
75 
76  const afw_value_assignment_t * const *assignment;
77  int local_top;
78  const afw_value_t *v;
79  const afw_value_t *result;
80  const afw_compile_value_contextual_t *saved_contextual;
81 
82  /* Push value on evaluation stack. */
83  afw_scope_evaluation_stack_push_value(instance, scope);
84  saved_contextual = scope->error->contextual;
85  scope->error->contextual = self->contextual;
86 
87  /* Save stack top which will be restored on return. */
88  local_top = afw_scope_begin_stack_frame(scope);
89  result = NULL;
90  AFW_TRY {
91 
92  /* Push all assignments on to scope stack. */
93  if (self->assignments) {
94  for (assignment = self->assignments; *assignment; assignment++) {
95  v = afw_value_evaluate((*assignment)->value, p, scope);
96  afw_scope_set_local_variable((*assignment)->name, v, scope);
97  }
98  }
99 
100  /* Evaluate expression value. */
101  result = afw_value_evaluate(self->value, p, scope);
102  }
103 
104  AFW_FINALLY{
105 
106  /* Restore scope stack top to what it was on entry. */
107  afw_scope_end_stack_frame(local_top, scope);
108 
109  }
110 
111  AFW_ENDTRY;
112 
113  /* Pop value from evaluation stack and return result. */
114  afw_scope_evaluation_stack_pop_value(scope);
115  scope->error->contextual = saved_contextual;
116  return result;
117 }
