22 /**
23  * Handles dataType=object
24  */
25 export const _Object = (props) => {
26  const [object, setObject] = useState();
27 
28  const property = useAfwProperty(props);
29  const changed = useEventId({ object, eventId: "onChildChanged" });
30  const model = useModel();
31  const operational = useOperational(props);
32  const isMounted = useIsMounted();
33 
34  const {value, valueMeta, onChanged, onValidate} = props;
35 
36  /**
37  * Embedded objects that do not have values yet, sometimes still need to be
38  * rendered, if the configured view options requests it. We do this by creating
39  * an object on the fly here. Once it's contents change, we can attach it to the
40  * property that is embedding it.
41  */
42  useEffect(() => {
43 
44  const newObject = async () => {
45  const parent = property.getEmbeddingObject();
46 
47  const object = model.newObject({
48  adaptorId: parent ? parent.getAdaptorId() : undefined,
49  objectTypeId: property.getDataTypeParameter(),
50  object: {}
51  });
52  await object.initialize();
53 
54  if (isMounted())
55  setObject(object);
56  };
57 
58  if (model && (value === undefined) && property)
59  newObject();
60 
61  }, [model, value, property, isMounted]);
62 
63  /**
64  * This routine captures object changes made to new, embedded objects. When a new embedded
65  * object changes, then we go ahead and assign it to the property.
66  */
67  useEffect(() => {
68  if (object && property && changed) {
69  property.setValue(object);
70  }
71  }, [object, property, changed]);
72 
73  if (!object && !value)
74  return null;
75 
76  /* lookup the appropriate layout for this object */
77  const Layout = layoutByObject({ object: object || value, operational });
78 
79  return (
80  <AdaptiveLayoutMergeProviders data={{ object: object || value }}>
81  <Layout
82  {...props}
83  {...valueMeta}
84  aria-label={property?.getLabel() || property?.getName()}
85  onChanged={onChanged}
86  onValidate={onValidate}
87  />
88  </AdaptiveLayoutMergeProviders>
89  );
90 };
91 
92 _Object.displayName = "Object";
93 
94 export default _Object;
