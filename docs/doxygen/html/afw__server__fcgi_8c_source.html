165 /* Process a request. */
166 static void impl_process_request(afw_server_fcgi_internal_t *self,
167  const afw_request_handler_t *director,
168  afw_server_fcgi_internal_server_thread_t *server_thread,
169  afw_xctx_t *xctx)
170 {
171  const afw_request_t * request;
172 
173  /* Create request. */
174  request = (const afw_request_t *)afw_server_fcgi_internal_create_request(
175  self, server_thread->fcgx_request, xctx);
176  xctx->request = request;
177 
178  afw_trace_fz(1, self->flag_index_trace_process,
179  &self->pub, xctx,
180  "Request begin thread %d concurrent %d",
181  server_thread->thread->thread_number,
182  server_thread->server->pub.concurrent);
183 
184  /* Handle request errors in request session xctx. */
185  AFW_TRY{
186 
187  afw_request_impl_trace_begin(request, xctx);
188 
189  /* Process request. */
190  afw_request_handler_process(self->director, request, xctx);
191  }
192 
193  /* If client_closed, don't try to write response, just log it. */
194  AFW_CATCH(client_closed) {
195  afw_error_write_log(afw_log_priority_info, AFW_ERROR_THROWN, xctx);
196  }
197 
198  /* If memory error, try not to use memory. */
199  AFW_CATCH(memory) {
200  afw_error_write_log(afw_log_priority_info, AFW_ERROR_THROWN, xctx);
201  }
202 
203  /* For all other errors, log and write to response body. */
204  AFW_CATCH_UNHANDLED{
205  afw_error_write_log(afw_log_priority_info, AFW_ERROR_THROWN, xctx);
206  afw_request_write_error_to_response_body(request,
207  afw_error_http_status(AFW_ERROR_THROWN), AFW_ERROR_THROWN, xctx);
208  }
209 
210  /* Always cleanup. */
211  AFW_FINALLY{
212  afw_request_release(request, xctx);
213  }
214 
215  AFW_ENDTRY;
216 
217  afw_trace_fz(1, self->flag_index_trace_process,
218  &self->pub, xctx,
219  "Request end thread %d concurrent %d",
220  server_thread->thread->thread_number,
221  server_thread->server->pub.concurrent);
222 }

225 /* Request thread run loop. */
226 static void* AFW_THREAD_FUNCTION
227 impl_afw_server_request_thread_start(const afw_thread_t *thread,
228  void *arg)
229 {
230  afw_server_fcgi_internal_server_thread_t *server_thread = arg;
231  afw_server_fcgi_internal_t * server = server_thread->server;
232  afw_xctx_t *xctx = thread->xctx;
233  afw_xctx_t *request_session_xctx = NULL;
234  afw_integer_t request_count;
235  afw_integer_t concurrent;
236  afw_integer_t max_concurrent;
237 
238  /* Initialize a FCGX_Request for this thread. */
239  server_thread->fcgx_request = afw_xctx_calloc_type(FCGX_Request, xctx);
240  FCGX_InitRequest(server_thread->fcgx_request, server->sock,
241  FCGI_FAIL_ACCEPT_ON_INTR);
242 
243  /* Request loop. */
244  while (!afw_xctx_environment_is_terminating(xctx) &&
245  FCGX_Accept_r(server_thread->fcgx_request) >= 0)
246  {
247  /* Increment count. */
248  request_count = afw_atomic_integer_increment(&server->pub.request_count);
249 
250  /* Process request. */
251  AFW_TRY {
252 
253  /* Increment concurrent count. */
254  max_concurrent = server->pub.max_concurrent;
255  concurrent = afw_atomic_integer_increment(&server->pub.concurrent);
256  if (concurrent > max_concurrent) {
257  afw_atomic_integer_cas(&server->pub.max_concurrent,
258  max_concurrent, concurrent);
259  }
260 
261  /* Create Adaptive Framework xctx for request session. */
262  request_session_xctx = afw_xctx_create(
263  &AFW_XCTX_s_NAME_REQUEST_SESSION, request_count, xctx);
264 
265  /* Process request. */
266  impl_process_request(server, server->director,
267  server_thread, request_session_xctx);
268  }
269 
270  AFW_CATCH_UNHANDLED{
271  /* Increment unhandled error count. */
272  afw_atomic_integer_increment(&server->pub.unhandled_errors);
273  }
274 
275  AFW_FINALLY {
276 
277  /* Decrement concurrent count. */
278  afw_atomic_integer_decrement(&server->pub.concurrent);
279 
280  /* Release xctx resources. */
281  if (request_session_xctx) {
282  afw_xctx_release(request_session_xctx, xctx);
283  }
284 
285  }
286 
287  AFW_ENDTRY;
288  }
289 
290  /* Normal return. */
291  return NULL;
292 }

296  * Implementation of method run of interface afw_server.

298 void
299 impl_afw_server_run(
300  const afw_server_t * instance,
301  const afw_request_handler_t * handler,
302  afw_xctx_t *xctx)
303 {
304  afw_server_fcgi_internal_t * server =
305  (afw_server_fcgi_internal_t *)instance;
306 
307  afw_integer_t count;
308  afw_server_fcgi_internal_server_thread_t *server_thread;
309  afw_thread_attr_t *thread_attr;
310 
311  server->director = handler;
312  server->xctx = xctx;
313  server->threads = afw_xctx_calloc(
314  sizeof(afw_server_fcgi_internal_server_thread_t) *
315  (afw_size_t)server->pub.thread_count,
316  xctx);
317 
320  /* The default thread attribute: detachable */
321  thread_attr = afw_thread_attr_create(xctx->p, xctx);
322 
323 
324  /* Create the request threads. */
325  for (count = 1, server_thread = server->threads;
326  count <= server->pub.thread_count;
327  count++, server_thread++)
328  {
329  server_thread->server = server;
330  server_thread->thread = afw_thread_create(
331  thread_attr,
332  impl_afw_server_request_thread_start, server_thread,
333  &AFW_XCTX_s_NAME_REQUEST_THREAD,
334  count, xctx);
335  }
336 
337  /* Wait for request threads to die. */
338  for (count = 1, server_thread = server->threads;
339  count <= server->pub.thread_count;
340  count++, server_thread++)
341  {
342  afw_thread_join(server_thread->thread, xctx);
343  }
344 
345 
346  /* Run is finished. */
347 }
