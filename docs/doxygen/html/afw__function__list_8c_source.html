81  *
82  * Converts 1 or more values in a list. If A value is a list, its individual
83  * values are included.

91  *
92  * Declaration:
93  *
94  * function list(
95  * ...values: (list of any)
96  * ): list;
97  *
98  * Parameters:
99  *
100  * values - (0 or more any dataType) Value to convert.
101  *
102  * Returns:
103  *
104  * (list) Converted value.
105  *
106  * Errors thrown:
107  *
108  * cast_error - value could not be converted
109  */
110 const afw_value_t *
111 afw_function_execute_list(
112  afw_function_execute_t *x)
113 {
114  afw_size_t n;
115  const afw_list_t *list;
116  const afw_value_t *value;
117 
118  list = afw_list_create_generic(x->p, x->scope);
119  for (n = 1; n <= x->argc; n++) {

213  *
214  * Declaration:

216  * function join(
217  * value: list,
218  * separator?: string
219  * ): string;
220  *
221  * Parameters:
222  *
223  * value - (list) A list of values of any data type.
224  *
225  * separator - (optional string) The separator to use. If not specified, a
226  * comma (,) is used.
227  *
228  * Returns:
229  *
230  * (string) Joined list values.
231  */
232 const afw_value_t *
233 afw_function_execute_join(
234  afw_function_execute_t *x)
235 {
236  const afw_value_list_t *value;
237  const afw_value_string_t *separator;
238  const afw_utf8_t *sep;
239  const afw_iterator_t *iterator;
240  const afw_list_t *list_of_string;
241  const afw_utf8_t *s;
242  afw_size_t len;
243  afw_value_string_t *result;
244  afw_utf8_octet_t *c;
245 
246  AFW_FUNCTION_EVALUATE_REQUIRED_DATA_TYPE_PARAMETER(value, 1, list);
247 
248  sep = &afw_s_a_comma;
249  if (AFW_FUNCTION_PARAMETER_IS_PRESENT(2)) {
250  AFW_FUNCTION_EVALUATE_REQUIRED_DATA_TYPE_PARAMETER(separator, 2, string);
251  sep = &separator->internal;
252  }
253 
254  /* Create list of strings and get len of all stings and separators. */
255  list_of_string = afw_list_convert_to_list_of_strings(value->internal,
256  x->p, x->scope);
257  for (iterator = NULL, len = 0;;) {
258  s = afw_list_of_string_get_next(list_of_string, &iterator, x->scope);
259  if (!s) {
260  break;
261  }
262  len += s->len + sep->len;
263  }
264 
265  /* If len > 0 adjust len for extra separator, make result and return. */
266  if (len > 0) {
267  len -= sep->len;
268  c = afw_pool_malloc(x->p, len, x->scope);
269  result = afw_value_allocate_string(x->p, x->scope);
270  result->internal.len = len;
271  result->internal.s = c;
272  for (iterator = NULL, len = 0;;) {
273  s = afw_list_of_string_get_next(
274  list_of_string, &iterator, x->scope);
275  if (!s) {
276  break;
277  }
278 
279  /* If not first time, add separator. */
280  if (c != result->internal.s) {
281  memcpy(c, sep->s, sep->len);
282  c += sep->len;
283  }
284 
285  /* Add this string. */
286  memcpy(c, s->s, s->len);
287  c += s->len;
288  }
289  return (const afw_value_t *)result;
290  }
291 
292  /* If len is zero, return empty string. */
293  else {
294  return afw_value_empty_string;
295  }
296 }

364  * given exactly the same parameters and has no side effects.
365  *
366  * Declaration:
367  *
368  * function slice(
369  * list: list,
370  * startIndex?: integer,
371  * endIndex?: integer
372  * ): list;
373  *
374  * Parameters:
375  *
376  * list - (list) The list to slice.
377  *
378  * startIndex - (optional integer) This is the zero based starting index. Use
379  * negative number to index from the end of the list. If not specified,
380  * an index of 0 is assumed.
381  *
382  * endIndex - (optional integer) This is the zero based ending index. If
383  * positive, this is one more than the index of the last value to include
384  * in the list. If negative, the index is from the end of the list. If
385  * not specified, the slice is from startIndex up to and including the
386  * end of the list.
387  *
388  * Returns:
389  *
390  * (list) A list containing the selected values. If all of the values are the
391  * same data type, the list will be a list of that data type.
392  */
393 const afw_value_t *
394 afw_function_execute_slice(
395  afw_function_execute_t *x)
396 {
397  const afw_value_list_t *list;
398  const afw_list_t *result_list;
399  const afw_value_integer_t *integer;
400  const afw_iterator_t *iterator;
401  const afw_value_t *value;
402  afw_integer_t start;
403  afw_integer_t end;
404  afw_integer_t count;
405 
406  AFW_FUNCTION_EVALUATE_REQUIRED_DATA_TYPE_PARAMETER(list, 1, list);
407  count = (afw_integer_t)afw_list_get_count(list->internal, x->scope);
408 
409  /* Determine start index. */
410  start = 0;
411  if (AFW_FUNCTION_PARAMETER_IS_PRESENT(2)) {
412  AFW_FUNCTION_EVALUATE_REQUIRED_DATA_TYPE_PARAMETER(integer, 2, integer);
413  start = integer->internal;
414  if (start < 0) {
415  start = count + start;
416  if (start < 0) {
417  start = 0;
418  }
419  }
420  }
421 
422  /* Determine end index. */
423  end = count;
424  if (AFW_FUNCTION_PARAMETER_IS_PRESENT(3)) {
425  AFW_FUNCTION_EVALUATE_REQUIRED_DATA_TYPE_PARAMETER(integer, 3, integer);
426  end = integer->internal;
427  if (end < 0) {
428  end = count + end;
429  }
430  else if (end > count) {
431  end = count;
432  }
433  }
434 
435  /* Create and return a list with slice. */
436  result_list = afw_list_create_generic(x->p, x->scope);
437  for (iterator = NULL, count = 0; count < end; count++) {
438  value = afw_list_get_next_value(list->internal, &iterator,
439  x->p, x->scope);
440  if (!value) {
441  AFW_THROW_ERROR_Z(general, "Expecting a value", x->scope);
442  }
443  if (start <= count) {
444  afw_list_add_value(result_list, value, x->scope);
445  }
446  }
447  afw_list_determine_data_type_and_set_immutable(result_list, x->scope);
448  return afw_value_create_list(result_list, x->p, x->scope);
449 }
