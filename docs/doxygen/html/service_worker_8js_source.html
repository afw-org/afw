5 // This lets the app load faster on subsequent visits in production, and gives
6 // it offline capabilities. However, it also means that developers (and users)
7 // will only see deployed updates on subsequent visits to a page, after all the
8 // existing tabs open on the page have been closed, since previously cached
9 // resources are updated in the background.

14 const isLocalhost = Boolean(
15  window.location.hostname === "localhost" ||
16  // [::1] is the IPv6 localhost address.

18  // 127.0.0.1/8 is considered localhost for IPv4.

24 function registerValidSW(swUrl, config) {
25  navigator.serviceWorker
26  .register(swUrl)
27  .then(registration => {
28  registration.onupdatefound = () => {
29  const installingWorker = registration.installing;
30  installingWorker.onstatechange = () => {
31  if (installingWorker.state === "installed") {
32  if (navigator.serviceWorker.controller) {
33  // At this point, the updated precached content has been fetched,
34  // but the previous service worker will still serve the older
35  // content until all client tabs are closed.
36  console.log(
37  "New content is available and will be used when all " +
38  "tabs for this page are closed. See http://bit.ly/CRA-PWA."
39  );
40 
41  // Execute callback
42  if (config && config.onUpdate) {
43  config.onUpdate(registration);
44  }
45  } else {
46  // At this point, everything has been precached.
47  // It's the perfect time to display a
48  // "Content is cached for offline use." message.
49  console.log("Content is cached for offline use.");
50 
51  // Execute callback
52  if (config && config.onSuccess) {
53  config.onSuccess(registration);
54  }
55  }
56  }
57  };
58  };
59  })
60  .catch(error => {
61  console.error("Error during service worker registration:", error);
62  });
63 }

65 function checkValidServiceWorker(swUrl, config) {
66  // Check if the service worker can be found. If it can't reload the page.
67  fetch(swUrl)
68  .then(response => {
69  // Ensure service worker exists, and that we really are getting a JS file.
70  if (
71  response.status === 404 ||
72  response.headers.get("content-type").indexOf("javascript") === -1
73  ) {
74  // No service worker found. Probably a different app. Reload the page.
75  navigator.serviceWorker.ready.then(registration => {
76  registration.unregister().then(() => {
77  window.location.reload();
78  });
79  });
80  } else {
81  // Service worker found. Proceed as normal.
82  registerValidSW(swUrl, config);
83  }
84  })
85  .catch(() => {
86  console.log(
87  "No internet connection found. App is running in offline mode."
88  );
89  });
90 }

92 export function register(config) {
93  if (process.env.NODE_ENV === "production" && "serviceWorker" in navigator) {
94  // The URL constructor is available in all browsers that support SW.
95  const publicUrl = new URL(process.env.PUBLIC_URL, window.location);
96  if (publicUrl.origin !== window.location.origin) {
97  // Our service worker won't work if PUBLIC_URL is on a different origin
98  // from what our page is served on. This might happen if a CDN is used to
99  // serve assets; see https://github.com/facebook/create-react-app/issues/2374
100  return;
101  }
102 
103  window.addEventListener("load", () => {
104  const swUrl = `${process.env.PUBLIC_URL}/service-worker.js`;
105 
106  if (isLocalhost) {
107  // This is running on localhost. Let's check if a service worker still exists or not.
108  checkValidServiceWorker(swUrl, config);
109 
110  // Add some additional logging to localhost, pointing developers to the
111  // service worker/PWA documentation.
112  navigator.serviceWorker.ready.then(() => {
113  console.log(
114  "This web app is being served cache-first by a service " +
115  "worker. To learn more, visit http://bit.ly/CRA-PWA"
116  );
117  });
118  } else {
119  // Is not localhost. Just register service worker
120  registerValidSW(swUrl, config);
121  }
122  });
123  }
