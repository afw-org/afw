61 void
62 impl_afw_request_read_raw_request_body(
63  const afw_request_t * instance,
64  afw_size_t buffer_size,
65  void * buffer,
66  afw_size_t * size,
67  afw_boolean_t * more_to_read,
68  afw_xctx_t *xctx)
69 {
70  afw_command_local_request_self_t *self =
71  (afw_command_local_request_self_t *)instance;
72 
73  /* Make sure in correct state then set it. */
74  if (self->state > afw_request_state_content_read) {
75  AFW_THROW_ERROR_Z(general,
76  "read_request_body() called out of order", xctx);
77  }
78  self->state = afw_request_state_content_read;
79 
80  *size = buffer_size;
81  if (*size > self->remaining_body) {
82  *size = self->remaining_body;
83  }
84 
85  if (*size > 0) {
86  memcpy(buffer,
87  self->body->ptr + self->body->size - self->remaining_body,
88  *size);
89  self->remaining_body -= *size;
90  }
91 
92  *more_to_read = self->remaining_body > 0;
93 }
