212 /* Get a property in an object's lineage. */
213 AFW_DEFINE(const afw_value_t *)
214 afw_object_get_property_extended(
215  const afw_object_t *instance,
216  const afw_utf8_t *property_name_extended,
217  afw_scope_t *scope)
218 {
219  const afw_object_t *obj;
220  const afw_utf8_t *meta_value;
221  afw_utf8_t after_dot;
222  const afw_value_t *result;
223  afw_utf8_t meta_pn;
224  afw_utf8_t pn;
225 
227  if (!instance ||
228  !property_name_extended ||
229  property_name_extended->len == 0)
230  {
231  return NULL;
232  }
233 
234  /* Return property from object passed or an ancestor. */
235  obj = instance;
236  result = NULL;
237 
238  /* If the property name starts with _meta_, process accordingly. */
239  if (afw_utf8_starts_with_z(property_name_extended,
240  AFW_OBJECT_Q_PN_META "."))
241  {
242  meta_value = NULL;
243  afw_utf8_substring_byte(&meta_pn, property_name_extended,
244  strlen(AFW_OBJECT_Q_PN_META "."), property_name_extended->len);
245  if (afw_utf8_equal_utf8_z(&meta_pn, "objectId")) {
246  meta_value = afw_object_meta_get_object_id(instance, scope);
247  }
248  else if (afw_utf8_equal_utf8_z(&meta_pn, "objectType")) {
249  meta_value = afw_object_meta_get_object_type_id(instance, scope);
250  }
251  else if (afw_utf8_equal_utf8_z(&meta_pn, "path")) {
252  meta_value = afw_object_meta_get_path(instance, scope);
253  }
260  if (meta_value) {
261  result = afw_value_create_string(meta_value, scope->p, scope);
262  }
263  return result;
264  }
265 
266  /* Check for dotted property name. */
267  pn.s = property_name_extended->s;
268  pn.len = property_name_extended->len;
269 
270  for (after_dot.s = property_name_extended->s,
271  after_dot.len = property_name_extended->len;
272  after_dot.len > 0;
273  (after_dot.s)++, (after_dot.len)--)
274  {
276  if (*after_dot.s == '.') {
277  pn.len -= after_dot.len;
278  (after_dot.s)++;
279  (after_dot.len)--;
280  break;
281  }
282  }
283 
284 
285  /* Attempt to get property and return if error or found. */
286  result = afw_object_get_property(obj, &pn, scope);
287  if (result) {
288 
289  /* If dotted name, process rest of name if it's object. */
290  if (after_dot.len > 0) {
291  if (afw_value_is_object(result)) {
292  result = afw_object_get_property_extended(
293  ((afw_value_object_t *)result)->internal,
294  &after_dot, scope);
295  }
296  }
297  }
298 
299  /* Return result. */
300  return result;
301 }

375 AFW_DEFINE(afw_boolean_t)
376 afw_object_old_get_property_as_boolean_deprecated(
377  const afw_object_t *instance,
378  const afw_utf8_t *property_name,
379  afw_scope_t *scope)
380 {

384 
385  result = AFW_FALSE;
386  value = afw_object_get_property(instance, property_name, scope);
387  if (value) {
388  value = afw_value_evaluate(value, scope->p, scope);
389  if (afw_value_is_boolean(value)) {
390  result = ((const afw_value_boolean_t *)value)->internal;
391  }
392  else {
393  string = afw_value_as_utf8(value, scope->p, scope);

397  (string->len == 1 &&
398  (

400  *string->s == 'T' ||
401  *string->s == '1')
402  )
403  )
404  {
405  result = AFW_TRUE;
406  }
407 
408  else if (
409  afw_utf8_compare_ignore_case(string, &afw_s_false, scope)== 0 ||
410  (string->len == 1 &&
411  (
412  *string->s == 'f' ||
413  *string->s == 'F' ||
414  *string->s == '0')
415  )
416  )
417  {
418  result = AFW_FALSE;
419  }
420 
421  else
422  {
423  AFW_THROW_ERROR_Z(general, "Boolean must be true or false",
424  scope);
425  }
426  }
427  }
428

514 /* Merge properties from one object into another. */
515 AFW_DEFINE(void)
516 afw_object_merge(
517  const afw_object_t *instance,
518  const afw_object_t *from,
519  afw_boolean_t replace,
520  afw_scope_t *scope)
521 {
522  const afw_iterator_t *iterator;
523  const afw_value_t *value;
524  const afw_value_t *from_value;
525  const afw_utf8_t *property_name;
526  const afw_object_t *embedded_object;
527 
528  iterator = NULL;
529  while ( (from_value = afw_object_get_next_property(
530  from, &iterator, &property_name, scope)) )
531  {
532  value = afw_object_get_property(instance, property_name, scope);
533  if (!value)
534  {
535  if (afw_value_is_object(from_value)) {
536  embedded_object = afw_object_create_embedded(instance,
537  property_name, scope);
538  afw_object_merge(embedded_object,
539  ((const afw_value_object_t *)from_value)->internal,
540  replace, scope);
541  } else {
542  afw_object_set_property(instance, property_name, from_value, scope);
543  }
544  }
545 
546  else if (afw_value_is_object(from_value) &&
547  afw_value_is_object(value))
548  {
549  afw_object_merge(
550  ((const afw_value_object_t *)value)->internal,
551  ((const afw_value_object_t *)from_value)->internal,
552  replace, scope);
553  }
554  else if (replace)
555  {
556  afw_object_set_property(instance, property_name, from_value, scope);
557  }
558  }
559 }
