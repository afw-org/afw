27 /**
28  * AdaptorDropdown presents a Dropdown component with the options
29  * that represent adaptors that are currently active.
30  *
31  * @param {string} label
32  * @param {string} description
33  * @param {string} selectedAdaptorId
34  * @param {boolean} disabled
35  * @param {boolean} required
36  * @param {boolean} isModelLocation
37  * @param {boolean} isPolicyLocation
38  * @param {boolean} isDevelopmentInput
39  * @param {string} type
40  *
41  */
42 export const AdaptorDropdown = (props) => {
43  const {adaptors} = useAdaptors(props.adaptorType);
44 
45  const {
46  id,
47  "data-testid": dataTestid,
48  "aria-label": ariaLabel,
49  label,
50  description,
51  value,
52  disabled,
53  required,
54  isModelLocation,
55  isPolicyLocation,
56  isDevelopmentInput,
57  onChanged,
58  allowNone,
59  } = props;
60 
61  const adaptorOptions = useMemo(() => {
62  if (!adaptors)
63  return;
64 
65  /* use filter, sort and map to transform the adaptors into dropdown items */
66  const adaptorOptions = adaptors.filter(adaptor => {
67 
68  if (isModelLocation)
69  return adaptor.properties.isModelLocation;
70 
71  if (isPolicyLocation)
72  return adaptor.properties.isPolicyLocation;
73 
74  if (isDevelopmentInput)
75  return adaptor.properties.isDevelopmentInput;
76 
77  /* passed the filters */
78  return true;
79  }).sort((A, B) => {
80  /* first sort by type */
81  if (A?.properties?.adaptorType.localeCompare(B?.properties?.adaptorType) !== 0)
82  return (A.properties.adaptorType.localeCompare(B.properties.adaptorType));
83 
84  /* sort by adaptorId */
85  const a = A.adaptorId.toLowerCase();
86  const b = B.adaptorId.toLowerCase();
87 
88  return a.localeCompare(b);
89  }).map(adaptor => {
90  /* map each adaptor to an Dropdown item, including icons */
91  const adaptorId = adaptor.adaptorId;
92  const adaptorType = adaptor?.properties?.adaptorType;
93  const icon = AdaptorIcons[adaptorType];
94 
95  return ({
96  adaptor,
97  key: adaptorId,
98  text: adaptorId,
99  icon,
100  });
101  });
102 
103  for (let i = 0; i < adaptorOptions.length; i++) {
104  const currType = adaptorOptions[i].adaptor.properties.adaptorType;
105  const prevType = (i > 0) ? adaptorOptions[i-1].adaptor.properties.adaptorType : currType;
106 
107  if (currType !== prevType) {
108  adaptorOptions.splice(i, 0, { key: currType + "-divider", type: "divider" });
109  adaptorOptions.splice(i+1, 0, { key: currType + "-header", type: "header", text: currType + " adaptors" });
110  i+=2;
111  }
112  }
113 
114  if (allowNone) {
115  adaptorOptions.unshift({
116  key: "none",
117  text: "(None)",
118  value: "",
119  });
120  }
121 
122  return adaptorOptions;
123  }, [adaptors, isModelLocation, isPolicyLocation, isDevelopmentInput, allowNone]);
124 
125 
126  const handleOnChanged = useCallback((selectedAdaptorId) => {
127  if (onChanged) {
128  onChanged(selectedAdaptorId.key, selectedAdaptorId.adaptor);
129  }
130  }, [onChanged]);
131 
132  return (
133  <Dropdown
134  id={id}
135  data-testid={dataTestid}
136  data-component-type="AdaptorDropdown"
137  aria-label={ariaLabel}
138  label={label ? label : "Adaptor"}
139  description={description ? description : "Select an AdaptorId"}
140  options={adaptorOptions}
141  selectedKey={adaptorOptions ? value : adaptorOptions}
142  onChanged={handleOnChanged}
143  disabled={disabled}
144  required={required}
145  />
146  );
147 };
