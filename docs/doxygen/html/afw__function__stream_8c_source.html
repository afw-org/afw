403  *
404  * afw_function_execute_open_file
405  *
406  * See afw_function_bindings.h for more information.
407  *
408  * This will open a file stream.
409  *
410  * This function is not pure, so it may return a different result
411  * given exactly the same parameters and has side effects.
412  *
413  * Declaration:
414  *
415  * function open_file(
416  * streamId: string,
417  * path: string,
418  * mode: string,
419  * autoFlush?: boolean
420  * ): integer;
421  *
422  * Parameters:
423  *
424  * streamId - (string) This is the streamId that will be associated with this
425  * open file stream.
426  *
427  * path - (string) This is the path to the file to open. The rootDirectory of
428  * the path is defined in the application object.
429  *
430  * mode - (string) This is the access mode string. Values can be:
431  * r - Open an existing file text file for read.
432  * w - Open a text file for writing. If the file does not exist, it
433  * will be created.
434  * a - Open a text file for writing additional data to the end. If the
435  * file does not exist, it will be created.
436  * r+ - Open a text file for both reading and writing.
437  * w+ - Open a text file for both reading and writing. If the file
438  * exists, it will be overwritten. If the file does not exist, it will be
439  * created.
440  * a+ - Open a text file for both reading and writing. Reading will
441  * begin at the start of the file while writing will be appended to the
442  * end.
443  *
444  * All of these modes expect data type string. If you are using data type
445  * base64Binary or hexBinary you can use corresponding binary modes,
446  * "rb", "wb", "ab", "rb+", "r+b", "wb+", "w+b", "ab+", and "a+b".
447  *
448  * autoFlush - (optional boolean) If specified and true, this will
449  * automatically flush the stream's buffers after every write.
450  *
451  * Returns:
452  *
453  * (integer) The streamNumber for the streamId or -1 if there was an error.
454  * Use get_stream_error() for error information.
455  */
456 const afw_value_t *
457 afw_function_execute_open_file(
458  afw_function_execute_t *x)
459 {
460  const afw_value_string_t *streamId;
461  const afw_value_string_t *path;
462  const afw_value_string_t *mode;
463  const afw_value_boolean_t *autoFlush;
464  const afw_stream_t *stream;
465  afw_size_t number;
466 
467  AFW_FUNCTION_EVALUATE_REQUIRED_DATA_TYPE_PARAMETER(streamId,
468  1, string);
469  AFW_FUNCTION_EVALUATE_REQUIRED_DATA_TYPE_PARAMETER(path,
470  2, string);
471  AFW_FUNCTION_EVALUATE_REQUIRED_DATA_TYPE_PARAMETER(mode,
472  3, string);
473  AFW_FUNCTION_EVALUATE_DATA_TYPE_PARAMETER(autoFlush,
474  4, boolean);
475 
476  number = afw_stream_get_streamNumber_for_streamId(
477  &streamId->internal, x->scope);
478  if (number == -1) {
479  AFW_THROW_ERROR_FZ(general, x->scope,
480  "streamId %" AFW_UTF8_FMT " is already open",
481  AFW_UTF8_FMT_ARG(&streamId->internal));
482  }
483

486  autoFlush && autoFlush->internal, x->p, x->scope);
487 
488  number = afw_stream_set(stream, x->scope);
489  if (number == -1) {
490  AFW_THROW_ERROR_FZ(general, x->scope,

492  AFW_UTF8_FMT_ARG(&streamId->internal));

494

496 }
