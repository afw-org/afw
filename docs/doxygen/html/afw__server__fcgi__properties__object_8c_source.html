159 const afw_value_t *
160 impl_afw_object_get_next_property(

164  afw_scope_t * scope)

169  const afw_utf8_octet_t *s;
170  const afw_utf8_octet_t *c;
171  const afw_utf8_t *string;
172 
173  if (!*iterator) {
174  *iterator = (afw_iterator_t *)afw_scope_calloc_type(
175  impl_request_properties_iterator_t, scope);
176 
177  }
178  it = (impl_request_properties_iterator_t *)*iterator;
179 
180  if (!it->envp) {
181  value = afw_object_get_next_property(self->properties,
182  &it->iterator, property_name, scope);
183  if (value) {
184  return value;
185  }
186  it->envp = self->request->fcgx_request->envp;
187  }
188 
189  if (!it->envp || !*it->envp) {
190  *iterator = NULL;
191  return NULL;
192  }
193 
194  for (s = c = *it->envp; *c && *c != '='; c++);
195  if (property_name) {
196  *property_name = afw_utf8_create(s, c - s, scope->p, scope);
197  }
198 
199  if (!*c) {
200  string = &afw_s_a_empty_string;
201  }
202  else {
203  string = afw_utf8_create(c + 1, AFW_UTF8_Z_LEN, scope->p, scope);
204  }
205 
206  value = afw_value_create_string(string, scope->p, scope);
207  (it->envp)++;
208  return value;
209 }
