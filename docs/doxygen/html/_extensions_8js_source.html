300  const {extensions, isLoading: isLoadingExtensions} = useExtensions();
301  const {manifests, onRefresh: reloadManifests, isLoading: isLoadingManifests} = useExtensionManifests();
302  const {notification, client} = useApplication();
303  const {application, reloadApp} = useAppCore();
304  const model = useModel();
305  const theme = useTheme();
306  const isMounted = useIsMounted();
307 
308  const breadcrumbItems = useBreadcrumbs(breadcrumbsRoot);
309 
310  const extensionsMap = useMemo(() => {
311  if (extensions) {
312  return Object.assign( ...extensions.map(i => ({[i.extensionId]: i})) );
313  } else return [];
314  }, [extensions]);
315 
316  /* sort the manifest objects from those that are loaded, followed by those that are not. Then, sort alphabetically */
317  const sortedManifests = useMemo(() => {
318  if (manifests && extensionsMap) {
319  /* sort loaded above not loaded */
320  let sortedManifests = manifests.sort((a, b) => {
321  let A = a.getPropertyValue("extensionId");
322  let B = b.getPropertyValue("extensionId");
323 
324  if (extensionsMap[A] && !extensionsMap[B])
325  return -1;
326  else if (extensionsMap[B] && !extensionsMap[A])
327  return 1;
328  else
329  return A.toLowerCase().localeCompare(B.toLowerCase());
330  });
331 
332  return sortedManifests;
333  } else return [];
334  }, [manifests, extensionsMap]);
335 
336  /* once we have our application object, loaded extensions, and manifests, go ahead and fetch our _AdaptiveConf_application object */
337  useEffect(() => {
338  let controller;
339 
340  /**
341  * This routine examines both the extensions and extension manifests objects to create a table
342  * of extensions, some loaded, some that haven't been loaded. The manifests shows all extensions, but the
343  * extensions shows only ones that have been loaded. The loaded are necessary to pull out version information.
344  */
345  const loadConfig = async () => {
346  dispatch({ type: "LOADING_CONFIGURATION" });
347 
348  try {
349  const {applicationId, confAdaptorId} = application.getPropertyValues();
350  const response = model.getObject({
351  objectTypeId: "_AdaptiveConf_application",
352  objectId: applicationId,
353  adaptorId: confAdaptorId
354  });
355  controller = response.controller;
356 
357  const applicationConf = await response.object;
358  if (!controller.signal.aborted)
359  dispatch({ type: "APPLICATION_CONF_LOADED", applicationConf });
360 
361 
362  } catch (error) {
363  if (!controller.signal.aborted) {
364  dispatch({ type: "ERROR", error });
365  notification({ message: error, type: "error" });
366  }
367  }
368  };
369 
370  if (application && extensions && manifests) {
371  loadConfig();
372  }
373 
374  return () => {
375  if (controller)
376  controller.abort();
377  };
378 
379  }, [application, extensions, manifests, model, notification]);

388  if (!isMounted()) return;
389

391  await reloadApp();
392  if (!isMounted()) return;

394  await reloadManifests();
395 
396  dispatch({ type: "EXTENSION_ADDED" });
397  /* let user know it was loaded and refreshed */
398  notification({ message: "Extension " + extensionId + " was loaded successfully.", type: "success" });
399  } catch (error) {
400  dispatch({ type: "ERROR", error });
401  notification({ message: error, type: "error" });
402  }
403  };

408

411  await afwExtensionLoad(client, manifest.getPropertyValue("extensionId")).result();
412  if (!isMounted()) return;
413  }
414 
415  if (onAppStartup) {
416  let extensions = applicationConf.getPropertyValue("extensions");
417  if (!extensions)
418  extensions = applicationConf.getProperty("extensions").setValue([]);
419 
420  selectedManifests.forEach(manifest => {
421  if (!extensions.includes(manifest.getPropertyValue("extensionId"))) {
422  let extension = extensions.newValue();
423  extension.setValue(manifest.getPropertyValue("extensionId"));
424  }
425  });
426 
427  /* get a fresh copy */
428  await applicationConf.refresh({ reconcilable: true });
429  if (!isMounted()) return;
430  applicationConf.setPropertyValue("extensions", extensions);
431 
432  const response = applicationConf.reconcile();
433  await response.result();
434  if (!isMounted()) return;
435 
436  /* reload application and functions that may have been updated by this extension load */
437  await reloadApp();
438  if (!isMounted()) return;
439  await reloadManifests();
440  if (!isMounted()) return;
441  } else {
442  /* reload application and functions that may have been updated by this extension load */
443  await reloadApp();
444  if (!isMounted()) return;
445  await reloadManifests();
446  if (!isMounted()) return;
447  }
448
