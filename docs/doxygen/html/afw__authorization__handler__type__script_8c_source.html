55  const afw_pool_t *p, afw_scope_t *scope)
56 {
57  afw_authorization_handler_script_self_t *self;
58  const afw_authorization_handler_t *authorization_handler;
59  const afw_utf8_t *detail_source_location;
60  const afw_utf8_t *context_type_id;
61  const afw_utf8_t *conf_object_type_id;
62  const afw_object_t *context_type_object;
63  const afw_object_t *qualifier_definitions;
64  const afw_object_t *variable_definitions;
65 
66  authorization_handler = afw_authorization_handler_impl_create_cede_p(
67  &impl_afw_authorization_handler_inf,
68  sizeof(afw_authorization_handler_script_self_t),
69  properties, p, scope);
70 
71  self = (afw_authorization_handler_script_self_t *)authorization_handler;
72 
73  /* Get and prepare qualifiers. */
74  detail_source_location = afw_utf8_printf(
75  self->pub.properties->p, scope,
76  "%" AFW_UTF8_FMT "/authorizationCheck",
77  AFW_UTF8_FMT_ARG(self->pub.source_location));
78  self->authorization_check =
79  afw_object_old_get_property_as_compiled_hybrid(
80  self->pub.properties,
81  &afw_s_authorizationCheck,
82  detail_source_location, NULL, p, scope);
83 
84  /* Make context. */
85  context_type_id = afw_utf8_printf(scope->env->p, scope,
86  "authorizationHandler-%" AFW_UTF8_FMT,
87  AFW_UTF8_FMT_ARG(&self->pub.authorization_handler_id));
88  conf_object_type_id = &afw_s__AdaptiveConf_authorizationHandler_script;
89  context_type_object = afw_context_type_create(
90  context_type_id, scope->env->p, scope);
91  qualifier_definitions =
92  afw_context_type_insure_qualifier_definitions_object_exists(
93  context_type_object, scope);
94  afw_object_meta_add_parent_path(qualifier_definitions,
95  &afw_s_a_context_type_application_qualifier_definitions_path,
96  scope);
97 
98  /* Configuration variables. */
99  variable_definitions =
100  afw_context_type_insure_variable_definitions_object_exists(
101  context_type_object, &afw_s_authorizationHandler, scope);
102  afw_context_variable_definitions_add_based_on_object_type_id(
103  variable_definitions, conf_object_type_id,
104  false, scope);
105 
106  /* Current variables. */
107  variable_definitions =
108  afw_context_type_insure_variable_definitions_object_exists(
109  context_type_object, &afw_s_current, scope);
110  afw_context_variable_definition_add_z(variable_definitions,
111  &afw_s_message, &afw_s_internal,
112  &afw_value_evaluated_string_inf,
113  "Message",
114  "Unformatted message that is being logged.",
115  NULL, NULL,
116  scope);
117 
118  /* qualifiedVariables definitions. */
119  self->qualified_variables = afw_object_old_get_property_as_object(
120  self->pub.properties, &afw_s_qualifiedVariables, scope);
121  if (self->qualified_variables) {
122  detail_source_location = afw_utf8_printf(
123  self->qualified_variables->p, scope,
124  "%" AFW_UTF8_FMT "/%" AFW_UTF8_FMT,
125  AFW_UTF8_FMT_ARG(self->pub.source_location),
126  AFW_UTF8_FMT_ARG(&afw_s_qualifiedVariables));
127  self->qualified_variables = afw_object_create_clone(
128  self->qualified_variables, p, scope);
129  afw_context_variable_definitions_compile_and_add_based_on_qualifiers_object(
130  context_type_object, self->qualified_variables,
131  detail_source_location, scope);
132  }
133 
134  /* Register context type. */
135  afw_environment_register_context_type(context_type_id,
136  context_type_object, scope);
137 
138  /* Return new instance. */
139  return authorization_handler;
140 }

163  * Implementation of method check for interface afw_authorization_handler.
164  */
165 const afw_value_t *
166 impl_afw_authorization_handler_check(
167  const afw_authorization_handler_t * instance,
168  const afw_value_t *resource_id,
169  const afw_value_t *object,
170  const afw_value_t *action_id,
171  const afw_pool_t * p,
172  afw_scope_t * scope)
173 {
174  afw_authorization_handler_script_self_t *self =
175  (afw_authorization_handler_script_self_t *)instance;
176  const afw_value_t *result;
177  int top;
178 
179  top = afw_scope_get_qualifier_stack_top(scope);
180  AFW_TRY{
181  if (self->qualified_variables) {
182  afw_scope_push_qualifiers_object(self->qualified_variables,
183  true, scope->p, scope);
184  }
185  result = afw_value_evaluate(self->authorization_check, p, scope);
186  if (!result || afw_value_is_null(result)) {
187  result = scope->env->authorization_control->not_applicable_result;
188  }
189  else if (!afw_value_is_object(result)) {
190  AFW_THROW_ERROR_Z(general,
191  "authorizationCheck must return an "
192  "_AdaptiveAuthorizationResult_ object",
193  scope);
194  }
195  }
196  AFW_CATCH_UNHANDLED{
197  AFW_MARK_UNHANDLED;
198  }
199  AFW_FINALLY{
200  afw_scope_set_qualifier_stack_top(top, scope);
201  }
202  AFW_ENDTRY;
203 
204  return result;
205 }
