48 static const afw_value_t *
49 impl_current_get_variable_cb(
50  const afw_scope_qualifier_stack_entry_t *entry,
51  const afw_utf8_t *name,
52  afw_scope_t *scope)
53 {
54  const afw_value_t *result;
55  afw_integer_t pid;
56 
57  result = NULL;
58  if (afw_utf8_equal(name, &afw_s_mode)) {
59  result = scope->mode;
60  }
61  else if (afw_utf8_equal(name, &afw_s_pid)) {
62  pid = afw_os_get_pid();
63  result = afw_value_create_integer(pid, scope->p, scope);
64  }
65  else if (afw_utf8_equal(name, &afw_s_scopeUUID)) {
66  result = afw_value_create_string(scope->uuid, scope->p, scope);
67  }
68  else if (afw_utf8_equal(name, &afw_s_programName)) {
69  result = afw_value_create_string(
70  &scope->env->program_name, scope->p, scope);
71  }
72 
73  return result;
74 }

161 afw_application_internal_application_conf_type_create_cede_p(
162  const afw_utf8_t *type,
163  const afw_object_t *entry,
164  const afw_utf8_t *source_location,
165  const afw_pool_t *p, afw_scope_t *scope)
166 {
167  afw_environment_t *env = (afw_environment_t *)scope->env;
168  const afw_utf8_t *application_id;
169  const afw_utf8_t *conf_adaptor_id;
170  const afw_adaptor_session_t *session;
171  const afw_iterator_t *iterator;
172  const afw_value_t *value;
173  const afw_value_t *entry_value;
174  const afw_utf8_t *property_name;
175  const afw_object_t *properties;
176  const afw_adaptor_t *layout_adaptor;
177  const afw_object_t *context_type_object;
178  const afw_object_t *variable_definitions_object;
179  const afw_list_t *default_flags;
180  const afw_utf8_t * const *extension_id;
181  const afw_utf8_t * const *module_path;
182  const afw_utf8_t *detail_source_location;
183  const afw_object_t *object;
184  const afw_utf8_t *s;
185  afw_boolean_t error;
186 
187  /* Only one application conf allowed. */
188  if (env->application_object) {
189  AFW_THROW_ERROR_FZ(general, scope,
190  "%" AFW_UTF8_CONTEXTUAL_LABEL_FMT
191  "type=application conf entry already specified.",
192  AFW_UTF8_FMT_ARG(source_location));
193  }
194 
195  /* Log application starting. */
196  AFW_LOG_Z(info, "Application starting.", scope);
197 
198  /* Get optional confAdaptorId. */
199  conf_adaptor_id = afw_object_old_get_property_as_utf8(
200  entry, &afw_s_confAdaptorId, p, scope);
201 
202  /* Get conf adaptor. It will not ever be released. */
203  if (conf_adaptor_id) {
204  AFW_LOG_FZ(debug, scope,
205  "Application specified confAdaptorId %" AFW_UTF8_FMT
206  ".",
207  AFW_UTF8_FMT_ARG(conf_adaptor_id));
208  env->conf_adaptor = afw_adaptor_get_reference(conf_adaptor_id, scope);
209  afw_adaptor_impl_set_supported_core_object_type(env->conf_adaptor,
210  &afw_s__AdaptiveServiceConf_, true, true, scope);
211  afw_adaptor_impl_set_supported_core_object_type(env->conf_adaptor,
212  &afw_s__AdaptiveConf_application, true, true, scope);
213  afw_adaptor_impl_set_supported_core_object_type(env->conf_adaptor,
214  &afw_s__AdaptiveHybridProperties_, false, true, scope);
215  }
216 
217  /* Get optional applicationId and default to "application". */
218  application_id = afw_object_old_get_property_as_utf8(entry,
219  &afw_s_applicationId, p, scope);
220  if (!application_id) {
221  application_id = &afw_s_Adaptive;
222  afw_object_set_property_as_string(entry,
223  &afw_s_applicationId, application_id, scope);
224  }
225  ((afw_environment_t *)env)->application_id.len = application_id->len;
226  ((afw_environment_t *)env)->application_id.s = application_id->s;
227 
228  /*
229  * If conf adaptor, merge in the _AdaptiveConf_application application_id
230  * object to entry.
231  */
232  if (conf_adaptor_id) {
233  object = NULL;
234  error = false;
235  session = afw_adaptor_session_create(conf_adaptor_id, scope);
236  AFW_TRY {
237  afw_adaptor_session_get_object(
238  session, NULL,
239  &afw_s__AdaptiveConf_application, application_id,
240  (void *)&object, impl_conf_object_cb, NULL, p, scope);
241  }
242 
243  AFW_CATCH(not_found) {
244  /* Okay, object is NULL. */
245  }
246 
247  AFW_CATCH_UNHANDLED{
248  error = true;
249  AFW_LOG_FZ(err, scope,
250  "%" AFW_UTF8_CONTEXTUAL_LABEL_FMT
251  "%s",
252  AFW_UTF8_FMT_ARG(source_location),
253  AFW_ERROR_THROWN->message_z);
254  }
255 
256  AFW_FINALLY{
257  afw_adaptor_session_release(session, scope);
258  }
259 
260  AFW_ENDTRY;
261 
262  /* If there was an unexpected error, throw error. */
263  if (error) {
264  AFW_THROW_ERROR_FZ(general, scope,
265  "%" AFW_UTF8_CONTEXTUAL_LABEL_FMT
266  "error logged _AdaptiveService_",
267  AFW_UTF8_FMT_ARG(source_location));
268  }
269 
270  if (object) {
271  for (iterator = NULL;;) {
272  value = afw_object_get_next_property(object, &iterator,
273  &property_name, scope);
274  if (!value) break;
275  entry_value = afw_object_get_property(entry,
276  property_name, scope);
277  if (entry_value ||
278  afw_utf8_equal(property_name, &afw_s_confAdaptorId) ||
279  afw_utf8_equal(property_name, &afw_s_applicationId))
280  {
281  if (!afw_value_equal(value, entry_value, scope)) {
282  AFW_LOG_FZ(warning, scope,
283  "%" AFW_UTF8_CONTEXTUAL_LABEL_FMT
284  "configuration type \"application\" ignored "
285  "/%" AFW_UTF8_FMT
286  "/_AdaptiveConf_application/%" AFW_UTF8_FMT
287  ".%" AFW_UTF8_FMT
288  " because it is allowWrite=false or specified "
289  " in the conf file.",
290  AFW_UTF8_FMT_ARG(source_location),
291  AFW_UTF8_FMT_ARG(conf_adaptor_id),
292  AFW_UTF8_FMT_ARG(application_id),
293  AFW_UTF8_FMT_ARG(property_name));
294  }
295  }
296  else {
297  afw_object_set_property(entry, property_name, value, scope);
298  }
299  }
300  }
301  }
302 
303  /* Prepare/validate application conf. */
304  properties = afw_environment_prepare_conf_type_properties(entry, scope);
305  env->application_object = properties;
306  afw_object_meta_set_ids(env->application_object, &afw_s_afw,
307  &afw_s__AdaptiveApplication_, &afw_s_current, scope);
308  afw_runtime_env_set_object(env->application_object, false, scope);
309 
310  /* If extensions specified, load them. */
311  value = afw_object_get_property(env->application_object,
312  &afw_s_extensions, scope);
313  if (value) {
314  for (extension_id = afw_value_as_array_of_utf8(value, p, scope);
315  *extension_id;
316  extension_id++)
317  {
318  afw_environment_load_extension(*extension_id, NULL, NULL, scope);
319  }
320  }
321 
322  /* If extensionModulePaths specified, load them. */
323  value = afw_object_get_property(env->application_object,
324  &afw_s_extensionModulePaths, scope);
325  if (value) {
326  for (module_path = afw_value_as_array_of_utf8(value, p, scope);
327  *module_path;
328  module_path++)
329  {
330  afw_environment_load_extension(NULL, *module_path, NULL, scope);
331  }
332  }
333 
334  /* Make application context object. */
335  context_type_object = afw_environment_get_context_type(&afw_s_application,
336  scope);
337 
338  /* application:: variable definitions. */
339  variable_definitions_object =
340  afw_context_type_insure_variable_definitions_object_exists(
341  context_type_object, &afw_s_application, scope);
342  afw_context_variable_definitions_add_based_on_object(
343  variable_definitions_object, env->application_object, scope);
344 
345  /* qualifiedVariables definitions. */
346  env->application_qualified_variables = afw_object_old_get_property_as_object(
347  env->application_object, &afw_s_qualifiedVariables, scope);
348  if (env->application_qualified_variables) {
349  detail_source_location = afw_utf8_printf(
350  env->application_qualified_variables->p, scope,
351  "%" AFW_UTF8_FMT "/%" AFW_UTF8_FMT,
352  AFW_UTF8_FMT_ARG(source_location),
353  AFW_UTF8_FMT_ARG(&afw_s_qualifiedVariables));
354  env->application_qualified_variables = afw_object_create_clone(
355  env->application_qualified_variables, p, scope);
356  afw_context_variable_definitions_compile_and_add_based_on_qualifiers_object(
357  context_type_object, env->application_qualified_variables,
358  detail_source_location, scope);
359  }
360 
361  /* rootFilePaths */
362  env->root_file_paths = afw_object_old_get_property_as_object(
363  env->application_object, &afw_s_rootFilePaths, scope);
364 
365  /* defaultFlags */
366  default_flags = afw_object_old_get_property_as_list(env->application_object,
367  &afw_s_defaultFlags, scope);
368  if (default_flags) {
369  afw_flag_set_default_flag_ids(default_flags, scope);
370  }
371 
372  /* Get optional layoutAdaptorId. */
373  env->layout_adaptor_id = afw_object_old_get_property_as_utf8(
374  properties, &afw_s_layoutsAdaptorId, p, scope);
375 
376  /* Set supported core object type in adaptor. */
377  if (env->layout_adaptor_id) {
378  AFW_LOG_FZ(debug, scope,
379  "Application specified layoutAdaptorId %" AFW_UTF8_FMT
380  ".",
381  AFW_UTF8_FMT_ARG(env->layout_adaptor_id));
382  layout_adaptor = afw_adaptor_get_reference(env->layout_adaptor_id, scope);
383  afw_adaptor_impl_set_supported_core_object_type(layout_adaptor,
384  &afw_s__AdaptiveLayoutComponentType_, true, true, scope);
385  afw_adaptor_impl_set_supported_core_object_type(layout_adaptor,
386  &afw_s__AdaptiveLayoutComponent_, false, true, scope);
387  afw_adaptor_release(layout_adaptor, scope);
388  }
389 
390  /* Process authorizationControl*/
391  object = afw_object_old_get_property_as_object(properties,
392  &afw_s_authorizationControl, scope);
393  afw_authorization_internal_set_control(object, scope);
394 
395  /* If conf adaptor, start any services that are ready. */
396  if (conf_adaptor_id) {
397  afw_service_internal_start_initial_services(p, scope);
398  }
399 
400  /* Push qualifiers on env scope. */
401  afw_application_internal_push_qualifiers(scope);
402 
403  /*
404  * If onApplicationStartupComplete specified, run its script and terminate
405  * if it returns a non-nullish value of other than integer 0.
406  */
407  value = afw_object_get_property(properties,
408  &afw_s_onApplicationStartupComplete, scope);
409  if (value) {
410  AFW_LOG_Z(info,
411  "Application onApplicationStartupComplete script being called.",
412  scope);
413  value = afw_value_compile_and_evaluate(value,
414  &impl_s_a_onApplicationStartupComplete_source_location,
415  p, scope);
416  if (!afw_value_is_nullish(value) &&
417  (!afw_value_is_integer(value) ||
418  ((afw_value_integer_t *)value)->internal != 0))
419  {
420  s = afw_value_as_casted_utf8(value, p, scope);
421  AFW_THROW_ERROR_FZ(general, scope,
422  "Application onApplicationStartupComplete script returned value "
423  "other than 0 - %" AFW_UTF8_FMT,
424  AFW_UTF8_FMT_ARG(s));
425  }
426  else {
427  AFW_LOG_Z(info,
428  "Application onApplicationStartupComplete script successfully completed.",
429  scope);
430 
431  }
432  }
433 
434  /* Log application startup complete. */
435  AFW_LOG_FZ(info, scope,
436  "%" AFW_UTF8_FMT " application startup complete.",
437  AFW_UTF8_FMT_ARG(application_id));
438
