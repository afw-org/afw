33  break;
34  }

36  }

40  x->cursor++;

42  return;

52  x->cursor++;
53  return;
54  }
55

57  x->token_type = afw_command_local_parse_token_type_colon;
58  x->cursor++;
59  return;
60  }

63  x->cursor++;
64  start = x->cursor;
65

67  if (x->cursor >= x->end) {
68  x->token_type = afw_command_local_parse_token_type_invalid;

78  x->cursor++;

80  }
81  }

87  return;

217  x.line_number);
