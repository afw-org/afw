36  * ']'

39  * '['

50 afw_compile_parse_List(
51  afw_compile_parser_t *parser,
52  afw_boolean_t allow_expression,
53  afw_boolean_t allow_enhanced_literals)
54 {
55  const afw_list_t *list;
56  const afw_value_t *v;
57  const afw_value_t *result;
58  afw_compile_args_t *args;
59  const afw_value_t **argv;
60  afw_size_t argc;
61  afw_size_t start_offset;
62  afw_boolean_t is_list_expression;
63 
64  /* Return NULL if next token is not '['. */
65  afw_compile_save_cursor(start_offset);
66  afw_compile_get_token();
67  if (!afw_compile_token_is(open_bracket)) {
68  afw_compile_reuse_token();
69  return NULL;
70  }
71 
72  args = NULL;
73  is_list_expression = false;
74 
75  /* Create result list. */
76  list = NULL;
77  args = NULL;
78 
79  /* Process array values until ']'. */
80  afw_compile_get_token();
81  if (!afw_compile_token_is(close_bracket)) {
82  afw_compile_reuse_token();
83  for (;;) {
84 
85  /* Spread operator. */
86  afw_compile_get_token();
87  if (allow_expression && afw_compile_token_is(ellipsis)) {
88  if ((parser)->compile_type == afw_compile_type_json ||
89  (parser)->compile_type == afw_compile_type_relaxed_json)
90  {
91  AFW_COMPILE_THROW_ERROR_Z(
92  "'...' is not allowed as json list entry");
93  }
94 
95  /*
96  * If this is first spread, start making args for
97  * add_entries().
98  */
99  if (!args) {
100  args = afw_compile_args_create(parser);
101  afw_compile_args_add_value(args,
102  (const afw_value_t *)&afw_function_definition_add_entries);
103  if (!list) {
104  list = afw_list_create_generic(parser->p, parser->scope);
105  v = afw_value_create_list(list, parser->p, parser->scope);
106  afw_compile_args_add_value(args, v);
107  }
108  if (is_list_expression) {
109  result = afw_value_create_list_expression(
110  afw_compile_create_contextual_to_cursor(start_offset),
111  list,
112  parser->p, parser->scope);
113  }
114  else {
115  result = afw_value_create_list(list, parser->p, parser->scope);
116  }
117  afw_compile_args_add_value(args, result);
118  }
119 
120  /* Add spread list expression to args. */
121  v = afw_compile_parse_Expression(parser);
122  afw_compile_args_add_value(args, v);
123 
124  /* A new list will be started if needed. */
125  list = NULL;
126  is_list_expression = false;
127  }
128 
129  /* Not spread operator. */
130  else {
131  afw_compile_reuse_token();
132 
133  /* If a spread has occurred and no list yet, make one. */
134  if (args && !list) {
135  list = afw_list_create_generic(
136  parser->p, parser->scope);
137  v = afw_value_create_list(list, parser->p, parser->scope);
138  afw_compile_args_add_value(args, v);
139  }
140 
141  /* If no list yet, make one. */
142  if (!list) {
143  list = afw_list_create_generic(parser->p, parser->scope);
144  }
145 
146  /* Next should be a value. */
147  if (allow_expression) {
148  v = afw_compile_parse_Expression(parser);
149  }
150  else if (allow_enhanced_literals) {
151  v = afw_compile_parse_Literal(parser,
152  NULL, true, false);
153  }
154  else {
155  v = afw_compile_parse_Json(parser);
156  }
157 
158  /* If not a defined evaluated value, make list expression. */
159  if (!afw_value_is_defined_and_evaluated(v)) {
160  is_list_expression = true;
161  }
162 
163  /* Add value to list. */
164  afw_list_add_value(list, v, parser->scope);
165  }
166 
167  /*
168  * If next token is ',' continue if strict mode or next tokens is
169  * not ']'.
170  */
171  afw_compile_get_token();
172  if (afw_compile_token_is(comma)) {
173  if (!parser->strict) {
174  afw_compile_get_token();
175  if (afw_compile_token_is(close_bracket)) {
176  break;
177  }
178  afw_compile_reuse_token();
179  }
180  continue;
181  }
182 
183  /* Or if it is a ']', it's end of array. */
184  if (afw_compile_token_is(close_bracket)) {
185  break;
186  }
187 
188  /* Anything else is an error. */
189  AFW_COMPILE_THROW_ERROR_Z(
190  "Array values must be separated with a comma");
191  }
192  }
193 
194  /* If there is a spread, a result is call to add_entries(). */
195  if (args) {
196  afw_compile_args_finalize(args, &argc, &argv);
197  result = afw_value_call_built_in_function_create(
198  afw_compile_create_contextual_to_cursor(start_offset),
199  argc - 1, argv, parser->p, parser->scope);
200  }
201 
202  /* Else if no entries yet, result is empty list. */
203  else if (!list) {
204  list = afw_list_create_generic(parser->p, parser->scope);
205  result = afw_value_create_list(list, parser->p, parser->scope);
206  }
207 
208  /* Else if list expression, result is list expression value. */
209  else if (is_list_expression) {
210  result = afw_value_create_list_expression(
211  afw_compile_create_contextual_to_cursor(start_offset),
212  list, parser->p, parser->scope);
213  }
214 
215  /* Else result is list value. */
216  else {
217  result = afw_value_create_list(list, parser->p, parser->scope);
218  }
219 
220  /* Return list value or call to add_entries. */
221  return result;
222 }

285  const afw_object_t *embedding_object;
286  const afw_utf8_t *property_name;
287  const afw_object_t *_meta_;
288  afw_compile_args_t *args;
289  const afw_value_t **argv;
290  const afw_value_t *result;
291  afw_size_t argc;
292  afw_size_t start_offset;
293  afw_boolean_t save_doing_object_spread;
294  afw_boolean_t is_object_expression;
295 
296  /* Return NULL if next token is not '{'. */
297  afw_compile_save_cursor(start_offset);
298  afw_compile_get_token();
299  if (!afw_compile_token_is(open_brace)) {
300  afw_compile_reuse_token();
301  return NULL;
302  }
303 
304  /* Create new memory object.*/
305  AFW_OBJECT_CREATE_ENTITY_OR_EMBEDDED(
306  obj,
307  parser->embedding_object,
308  parser->property_name,
309  parser->doing_object_spread,
310  parser->cede_p, parser->p, parser->scope);
311  _meta_ = NULL;
312  args = NULL;
313  result = NULL;
314  is_object_expression = false;
315 
316  /*
317  * Save parser->embedding_object and set to new object.
318  *
319  * Save parser->property_name and set to NULL.
320  */
321  embedding_object = parser->embedding_object;
322  parser->embedding_object = obj;
323  property_name = parser->property_name;
324  parser->property_name = NULL;
325 
326  /* Process name/value pairs until '}'. */
327  afw_compile_get_token();
328  if (!afw_compile_token_is(close_brace)) {
329  afw_compile_reuse_token();
330  for (;;) {
331 
332  afw_compile_get_token();
333  parser->property_name = NULL;
334 
335  if (parser->strict) {
336  if (afw_compile_token_is(utf8_string) &&
337  parser->token->string_quote_character == '"')
338  {
339  parser->property_name = parser->token->string;
340  }
341  }
342 
343  else if (afw_compile_token_is(utf8_string)) {
344  parser->property_name = parser->token->string;
345  }
346 
347  else if (afw_compile_token_is(identifier) &&
348  !parser->token->identifier_qualifier)
349  {
350  parser->property_name = parser->token->identifier_name;
351  }
352 
353  /* If spread operator. */
354  if (allow_expression && afw_compile_token_is(ellipsis)) {
355 
356  /*
357  * If this is first spread, start making args for add_entries().
358  */
359  if (!args) {
360  args = afw_compile_args_create(parser);
361  afw_compile_args_add_value(args,
362  (const afw_value_t *)&afw_function_definition_add_properties);
363  if (is_object_expression) {
364  result = afw_value_create_object_expression(
365  afw_compile_create_contextual_to_cursor(start_offset),
366  obj, parser->p, parser->scope);
367  }
368  else {
369  result = afw_value_create_object(obj, parser->p, parser->scope);
370  }
371  afw_compile_args_add_value(args, result);
372  is_object_expression = false;
373  }
374 
375  /* Add spread object expression to args. */
376  save_doing_object_spread = parser->doing_object_spread;
377  parser->doing_object_spread = true;
378  v = afw_compile_parse_Expression(parser);
379  parser->doing_object_spread = save_doing_object_spread;
380  afw_compile_args_add_value(args, v);
381 
382  /* A new object will be started if needed. */
383  obj = NULL;
384  }
385 
386  /* If not spread */
387  else {
388  if (!parser->property_name) {
389  AFW_COMPILE_THROW_ERROR_Z("Invalid property name");
390  }
391 
392  /* Next should be a colon. */
393  afw_compile_get_token();
394  if (!afw_compile_token_is(colon)) {
395  AFW_COMPILE_THROW_ERROR_Z(
396  "Name of an object value must be followed by a colon");
397  }
398 
399  /* Next should be a value. */
400  if (allow_expression) {
401  v = afw_compile_parse_Expression(parser);
402  }
403  else if (allow_enhanced_literals) {
404  v = afw_compile_parse_Literal(parser,
405  NULL, true, false);
406  }
407  else {
408  v = afw_compile_parse_Json(parser);
409  }
410 
411  /* If this is _meta_ property, set meta in object. */
412  if (afw_utf8_equal(parser->property_name, &afw_s__meta_)) {
413  if (!afw_value_is_object(v)) {
414  AFW_COMPILE_THROW_ERROR_Z(
415  "_meta_ property must be an object");
416  }
417  _meta_ = ((const afw_value_object_t *)v)->internal;
418  }
419 
420  /*
421  * If not _meta_ property, set property in new object to this
422  * value if it's not an object. Object will already have been
423  * added by afw_object_create_embedded().
424  */
425  else if (!afw_value_is_object(v)) {
426  if (!obj) {
427  obj = afw_object_create(
428  parser->p, parser->scope);
429  afw_compile_args_add_value(args,
430  afw_value_create_object(obj,
431  parser->p, parser->scope));
432  }
433  afw_object_set_property(obj, parser->property_name, v,
434  parser->scope);
435  if (!afw_value_is_defined_and_evaluated(v)) {
436  is_object_expression = true;
437  }
438  }
439 
440  }
441 
442  /*
443  * If next token is ',' continue if strict mode or next tokens is
444  * not '}'.
445  */
446  afw_compile_get_token();
447  if (afw_compile_token_is(comma)) {
448  if (!parser->strict) {
449  afw_compile_get_token();
450  if (afw_compile_token_is(close_brace)) {
451  break;
452  }
453  afw_compile_reuse_token();
454  }
455  continue;
456  }
457 
458  /* Or if it is '}', it's end of object. */
459  if (afw_compile_token_is(close_brace)) break;
460 
461  /* Anything else is an error. */
462  AFW_COMPILE_THROW_ERROR_Z(
463  "Name/Value pairs must be separated with a comma");
464  }
465  }
466 
467  /* If there were spread operators, result is call to add_properties(). */
468  if (args) {
469  afw_compile_args_finalize(args, &argc, &argv);
470  result = afw_value_call_built_in_function_create(
471  afw_compile_create_contextual_to_cursor(start_offset),
472  argc - 1, argv, parser->p, parser->scope);
473  }
474 
475  /* Else if object expression, result is object expression value. */
476  else if (is_object_expression) {
477  result = afw_value_create_object_expression(
478  afw_compile_create_contextual_to_cursor(start_offset),
479  obj, parser->p, parser->scope);
480  }
481 
482  /* Else result is object value. */
483  else {
484  afw_object_meta_set_meta_object(obj, _meta_, parser->scope);
485  result = afw_value_create_object(obj, parser->p, parser->scope);
486  }
487 
488  /* Restore saved embedding object and property name. */
489  parser->embedding_object = embedding_object;
490  parser->property_name = property_name;
491 
492  /* Return object value. */
493  return result;
494 }

499  *
500  * Json ::= Double | Integer | List | Object | String | 'true' | 'false' | 'null'

502  * RelaxedJson ::= Double | Integer | RelaxedList | RelaxedObject | String | 'true' | 'false' | 'null'

505 AFW_DEFINE_INTERNAL(const afw_value_t *)
506 afw_compile_parse_Json(afw_compile_parser_t *parser)
507 {
508  /* parser->strict false causes RelaxedJson to be used. */
509 
510  const afw_value_t *value = NULL;
511 
512  afw_compile_get_token();
513 
514  switch (parser->token->type) {
515 
516  case afw_compile_token_type_end:
517  AFW_COMPILE_THROW_ERROR_Z("Expecting Value");
518 
519  case afw_compile_token_type_open_brace:
520  afw_compile_reuse_token();
521  value = afw_compile_parse_Object(parser, false, false);
522  break;
523 
524  case afw_compile_token_type_open_bracket:
525  afw_compile_reuse_token();
526  value = afw_compile_parse_List(parser, false, false);
527  break;
528 
529  case afw_compile_token_type_boolean:
530  value = (parser->token->boolean)
531  ? afw_value_true
532  : afw_value_false;
533  break;
534 
535  case afw_compile_token_type_integer:
536  value = afw_value_create_integer(
537  parser->token->integer,
538  parser->p, parser->scope);
539  break;
540 
541  case afw_compile_token_type_number:
542  value = afw_value_create_double(
543  parser->token->number,
544  parser->p, parser->scope);
545  break;
546 
547  case afw_compile_token_type_utf8_string:
548  value = afw_value_create_string(parser->token->string,
549  parser->p, parser->scope);
550  break;
551 
552  case afw_compile_token_type_null:
553  value = afw_value_null;
554  break;
555 
556  case afw_compile_token_type_invalid:
557  default:
558  AFW_COMPILE_THROW_ERROR_Z("Invalid token type");
559  }
560 
561  /* If value was not assigned, reuse token. */
562  if (!value) {
563  afw_compile_reuse_token();
564  }
565 
566  /* Return value. */
567  return value;
568 }

571 /*ebnf>>>
572  *
573  * ScalarLiteral ::= Double | Integer | String | 'true' | 'false' | 'null'
574  *
575  *# This may want to include Datatype(Literal) at some point too.
576  * Literal ::= ListLiteral | ObjectLiteral | ScalarLiteral | 'undefined'
577  *
578  *<<<ebnf*/
579 AFW_DEFINE_INTERNAL(const afw_value_t *)
580 afw_compile_parse_Literal(
581  afw_compile_parser_t *parser,
582  afw_boolean_t *is_Literal,
583  afw_boolean_t must_be_literal,
584  afw_boolean_t scalar_only)
585 {
586  const afw_value_t *value;
587  afw_boolean_t matches_production;
588 
589  matches_production = false;
590  value = NULL;
591 
592  afw_compile_get_token();
593 
594  switch (parser->token->type) {
595 
596  case afw_compile_token_type_end:
597  AFW_COMPILE_THROW_ERROR_Z("Expecting Value");
598 
599  case afw_compile_token_type_open_brace:
600  afw_compile_reuse_token();
601  if (scalar_only) {
602  break;
603  }
604  matches_production = true;
605  value = afw_compile_parse_Object(parser, false, true);
606  break;
607 
608  case afw_compile_token_type_open_bracket:
609  afw_compile_reuse_token();
610  if (scalar_only) {
611  break;
612  }
613  matches_production = true;
614  value = afw_compile_parse_List(parser, false, true);
615  break;
616 
617  case afw_compile_token_type_boolean:
618  matches_production = true;
619  value = (parser->token->boolean)
620  ? afw_value_true
621  : afw_value_false;
622  break;
623 
624  case afw_compile_token_type_integer:
625  matches_production = true;
626  value = afw_value_create_integer(
627  parser->token->integer,
628  parser->p, parser->scope);
629  break;
630 
631  case afw_compile_token_type_number:
632  matches_production = true;
633  value = afw_value_create_double(
634  parser->token->number,
635  parser->p, parser->scope);
636  break;
637 
638  case afw_compile_token_type_utf8_string:
639  matches_production = true;
640  value = afw_value_create_string(parser->token->string,
641  parser->p, parser->scope);
642  break;
643 
644  case afw_compile_token_type_null:
645  matches_production = true;
646  value = afw_value_null;
647  break;
648 
649  case afw_compile_token_type_undefined:
650  matches_production = true;
651  value = NULL;
652  break;
653 
654  default:
655  break;
656  }
657 
658  /* If requested, return is_literal. */
659  if (is_Literal) {
660  *is_Literal = matches_production;
661  }
662 
663  /* If must be Literal, throw error if it's not. */
664  if (must_be_literal && !matches_production) {
665  AFW_COMPILE_THROW_ERROR_Z("Expecting a literal");
666  }
667 
668  /* If doesn't match production, reuse token. */
669  if (!matches_production) {
670  afw_compile_reuse_token();
671  }
672 
673  /* Return value. */
674  return value;
675 }

679 /*ebnf>>>
680  *
681  *#
682  *# Evaluation must be before ParenthesizedExpression so Lambda has first shot
683  *# at parenthesis.
684  *#
685  *
686  * Value ::= ListValue |

692  *

694 AFW_DEFINE_INTERNAL(const afw_value_t *)
695 afw_compile_parse_Value(afw_compile_parser_t *parser)
696 {
697  const afw_value_t *value = NULL;
698  afw_boolean_t is_Literal;
699 
700  /* If ListValue, return its value. */
701  value = afw_compile_parse_List(parser, true, true);
702  if (value) {
703  return value;
704  }
705 
706  /* If ObjectValue, return its value. */
707  value = afw_compile_parse_Object(parser, true, true);
708  if (value) {
709  return value;
710  }
711 
712  /* If ScalarLiteral, return its value. */
713  value = afw_compile_parse_Literal(parser, &is_Literal, false, true);
714  if (is_Literal) {
715  return value;
716  }
717 
718  /* If evaluation, return its value. */
719  value = afw_compile_parse_Evaluation(parser);
720  if (value) {
721  return value;
722  }
723 
724  /* If next is '(', parse ParenthesizedExpression. */
725  if (afw_compile_peek_next_token_is(open_parenthesis)) {
726  value = afw_compile_parse_ParenthesizedExpression(parser);
727  return value;
728  }
729 
730  /* If next is '`', parse TemplateString. */
731  if (afw_compile_peek_next_token_is(grave)) {
732  value = afw_compile_parse_TemplateString(parser);
733  return value;
734  }
735 
736  /* Invalid. */
737  AFW_COMPILE_THROW_ERROR_Z("Expecting Value");
738 }
