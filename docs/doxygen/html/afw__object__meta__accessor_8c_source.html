35 AFW_DEFINE(const afw_object_t *)
36 afw_object_meta_create_accessor_with_options(
37  const afw_object_t *instance,
38  const afw_object_options_t *options,
39  const afw_pool_t *p,
40  afw_scope_t *scope)
41 {
42  afw_object_internal_meta_accessor_self_t *self;
43  const afw_iterator_t *iterator;
44  const afw_utf8_t *property_name;
45 
46  /* If instance of this implementation with no meta, return NULL. */
47  if (!instance->meta.meta_object && instance->inf == &impl_afw_object_inf)
48  {
49  return NULL;
50  }
51 
52  /* Create self. */
53  self = afw_pool_calloc_type(p, afw_object_internal_meta_accessor_self_t, scope);
54  self->pub.inf = &impl_afw_object_inf;
55  self->pub.p = p;
56  self->options = options;
57  self->instance = instance;
58  self->embedded = (instance->meta.embedding_object != NULL);
59  self->limited_meta = !afw_object_meta_object(self->instance) ||
60  AFW_OBJECT_OPTION_IS(self->options, metaLimited);
61 
62  /* If there are no properties to return, return NULL. */
63  iterator = NULL;
64  if (!impl_afw_object_get_next_property((afw_object_t *)self,
65  &iterator, &property_name, scope))
66  {
67  return NULL;
68  }
69  self->iterator = NULL;
70 
71  /* Return new instance. */
72  return (afw_object_t *)self;
73 }

118 /*
119  * Implementation of method get_next_property of interface afw_object.
120  */
121 const afw_value_t *
122 impl_afw_object_get_next_property(
123  const afw_object_t *instance,
124  const afw_iterator_t **iterator,
125  const afw_utf8_t **property_name,
126  afw_scope_t *scope)
127 {
128  afw_object_internal_meta_accessor_self_t *self =
129  (afw_object_internal_meta_accessor_self_t *)instance;
130  const afw_pool_t *p = self->pub.p;
131  const afw_value_t *result = NULL;
132  const afw_utf8_t *s;
133  int state;
134 
135  /*
136  * Note: The options are fairly simple atm. If there is an option added
137  * to merge in associated _AdaptiveObjectType_ properties, multiple
138  * internal iterators will be needed.
139  */
140 
141  /* Initial state is 0 and after that iterator holds state. */
142  if (!*iterator) {
143  *iterator = afw_pool_calloc(self->pub.p, sizeof(int), scope);
144  }
145  state = *(int *)* iterator;
146 
147  /* Process state. */
148  for (;;) {
149  switch (state) {
150 
151  /* Initial state. */
152  case 0:
153 
154  /* Next state is 1 or 2 depending on limited_meta. */
155  state = (!self->limited_meta) ? 1 : 2;
156 
157  /* If limited_data, always provide parentPaths. */
158  if (self->limited_meta && afw_object_meta_object(self->instance)) {
159  result = afw_object_get_property(
160  afw_object_meta_object(self->instance), &afw_s_parentPaths,
161  scope);
162  if (result) {
163  if (property_name) {
164  *property_name = &afw_s_parentPaths;
165  }
166  goto return_result;
167  }
168  }
169 
170  /* Next state. */
171  break;
172 
173  /* Not limited meta. Just return all properties in meta and end. */
174  case 1:
175 
176  result = afw_object_get_next_property(
177  afw_object_meta_object(self->instance),
178  &self->iterator, property_name, scope);
179  if (result) {
180  goto return_result;
181  }
182  state = 5;
183 
184  /* options = objectId. */
185  case 2:
186 
187  /* Next state is 3. */
188  state = 3;
189 
190  /* If not already returned by fullMetaObject, return value. */
191  if (!self->embedded &&
192  AFW_OBJECT_OPTION_IS(self->options, objectId))
193  {
194  if (self->limited_meta ||
195  !afw_object_has_property(
196  afw_object_meta_object(self->instance),
197  &afw_s_objectId, scope))
198  {
199  s = afw_object_meta_get_object_id(self->instance, scope);
200  if (s) {
201  if (property_name) {
202  *property_name = &afw_s_objectId;
203  }
204  result = afw_value_create_string(s, p, scope);
205  goto return_result;
206  }
207  }
208  }
209 
210 
211  /* options = objectType. */
212  case 3:
213 
214  /* Next state is 4. */
215  state = 4;
216 
217  /*
218  * If option objectType or metaLimited and this is an embedded
219  * object, set meta objectType, return value.
220  */
221  if (AFW_OBJECT_OPTION_IS(self->options, objectType) ||
222  (self->limited_meta && self->instance->meta.embedding_object))
223  {
224  if (self->limited_meta ||
225  !afw_object_has_property(
226  afw_object_meta_object(self->instance),
227  &afw_s_objectType, scope))
228  {
229  s = afw_object_meta_get_object_type_id(self->instance,
230  scope);
231  if (s) {
232  if (property_name) {
233  *property_name = &afw_s_objectType;
234  }
235  result = afw_value_create_string(s, p, scope);
236  goto return_result;
237  }
238  }
239  }
240 
241 
242  /* options = path and pathEmbedded. */
243  case 4:
244 
245  /* Next state is 5. */
246  state = 5;
247 
248  /* If not already returned by fullMetaObject, return value. */
249  if ((!self->embedded &&
250  AFW_OBJECT_OPTION_IS(self->options, path))
251  ||
252  (self->embedded &&
253  AFW_OBJECT_OPTION_IS(self->options, pathEmbedded))
254  )
255  {
256  if (self->limited_meta ||
257  !afw_object_has_property(
258  afw_object_meta_object(self->instance), &afw_s_path, scope))
259  {
260  s = afw_object_meta_get_path(self->instance, scope);
261  if (s) {
262  if (property_name) {
263  *property_name = &afw_s_path;
264  }
265  result = afw_value_create_string(s, p, scope);
266  goto return_result;
267  }
268  }
269  }
270 
271 
272  /* Nothing left. */
273  default:
274  result = NULL;
275  goto return_result;
276  }
277  }
278 
279 
280  /* Return result. */
281 return_result:
282  if (!result) {
283  *iterator = NULL;
284  }
285  else {
286  *(int *)*iterator = state;
287  }
288  return result;
289 }
