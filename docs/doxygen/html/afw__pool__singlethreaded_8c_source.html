96 static void
97 impl_remove_child(afw_pool_internal_singlethreaded_self_t *parent,
98  afw_pool_internal_singlethreaded_self_t *child, afw_scope_t *scope)
99 {
100  afw_pool_internal_singlethreaded_self_t *prev;
101  afw_pool_internal_singlethreaded_self_t *sibling;
102 
103  for (prev = NULL, sibling = parent->first_child;
104  sibling;
105  prev = sibling, sibling = sibling->next_sibling)
106  {
107  if (sibling == child) {
108  if (!prev) {
109  parent->first_child = sibling->next_sibling;
110  }
111  else {
112  prev->next_sibling = sibling->next_sibling;
113  }
114  break;
115  }
116  }
117 
118  if (!sibling) {
119  AFW_THROW_ERROR_Z(general, "Not a child of parent", scope);
120  }
121 
122  afw_pool_release(&parent->pub, scope);
123 }

126 /* Create skeleton pool struct. */
127 static afw_pool_internal_singlethreaded_self_t *
128 impl_pool_create(afw_pool_internal_singlethreaded_self_t *parent, afw_scope_t *scope)
129 {
130  apr_pool_t *apr_p;
131  afw_pool_internal_singlethreaded_self_t *self;
132 
133  apr_pool_create(&apr_p, (parent) ? parent->apr_p : NULL);
134  if (!apr_p) {
135  AFW_THROW_ERROR_Z(memory, "Unable to allocate pool", scope);
136  }
137  self = apr_pcalloc(apr_p, sizeof(afw_pool_internal_singlethreaded_self_t));
138  if (!self) {
139  AFW_THROW_ERROR_Z(memory, "Unable to allocate pool", scope);
140  }
141  self->pub.inf = &impl_afw_pool_inf;
142  self->apr_p = apr_p;
143  self->reference_count = 1;
144  self->parent = parent;
145  self->pool_number = (afw_integer_t)afw_atomic_integer_increment(
146  &((afw_environment_t *)scope->env)->pool_number);
147 
148  /* If parent, add this new child. */
149  if (parent) {
150  impl_add_child(parent, self, scope);
151  }
152 
153  /* Return new pool. */
154  return self;
155 }

280 /*
281  * Implementation of method destroy for interface afw_pool.
282  */
283 void
284 impl_afw_pool_destroy(
285  const afw_pool_t * instance,
286  afw_scope_t * scope)
287 {
288  afw_pool_internal_singlethreaded_self_t *self =
289  (afw_pool_internal_singlethreaded_self_t *)instance;
290  afw_pool_internal_singlethreaded_self_t *child;
291  afw_pool_cleanup_t *e;
292 
293  /* If instance is NULL, just return. */
294  if (!instance) return;
295 
296  /*
297  * Call all of the cleanup routines for this pool before destroying
298  * children.
299  */
300  for (e = self->first_cleanup; e; e = e->next_cleanup) {
301  e->cleanup(e->data, e->data2, instance, scope);
302  }
303 
304  /*
305  * Destroy children.
306  *
307  * Release of child sets self->first_child to its next sibling.
308  */
309  for (child = self->first_child;
310  child;
311  child = self->first_child)
312  {
313  afw_pool_destroy((const afw_pool_t *)child, scope);
314  }
315 
316  /* If parent, removed self as child. */
317  if (self->parent) {
318  impl_remove_child(self->parent, self, scope);
319  }
320 
321  /* Destroy apr pool. */
322  apr_pool_destroy(self->apr_p);
323 }
