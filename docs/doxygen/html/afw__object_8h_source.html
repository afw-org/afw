425  const afw_utf8_t *path)
426 {
427  const afw_utf8_octet_t *s;
428 
429  entity_path->s = path->s;
430  entity_path->len = path->len;
431  for (s = path->s + path->len - 1;
432  s > path->s && *s != '/';
433  s--)
434  {
435  if (*s == '.') {
436  entity_path->len = s - path->s;
437  }
438  }
439 }
