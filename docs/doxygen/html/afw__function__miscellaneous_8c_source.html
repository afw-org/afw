530  * Adaptive function: trace
531  *
532  * afw_function_execute_trace
533  *
534  * See afw_function_bindings.h for more information.
535  *
536  * Write a value to a trace log.
537  *
538  * This function is not pure, so it may return a different result
539  * given exactly the same parameters and has side effects.
540  *
541  * Declaration:
542  *
543  * function trace(
544  * value: any,
545  * filter?: boolean,
546  * number?: integer
547  * ): null;
548  *
549  * Parameters:
550  *
551  * value - (any dataType) This is the value that will be converted to its
552  * string representation and written the trace log. An undefined value is
553  * represented by '<undefined>'.
554  *
555  * filter - (optional boolean) If this optional filter is false, nothing will
556  * be written to the trace log. The default is true.
557  *
558  * number - (optional integer) This is an optional number between 1 and 8
559  * that is appended to "trace" to identify the trace log. The default is
560  * 1.
561  *
562  * Returns:
563  *
564  * (null)
565  */
566 const afw_value_t *
567 afw_function_execute_trace(
568  afw_function_execute_t *x)
569 {
570  const afw_value_t *value;
571  const afw_value_boolean_t *filter;
572  const afw_value_integer_t *number;
573  const afw_utf8_t *s;
574  afw_log_priority_t priority;
575 
576  /* If parameter 2 is present and false, skip writing value. */
577  if (AFW_FUNCTION_PARAMETER_IS_PRESENT(2)) {
578  AFW_FUNCTION_EVALUATE_REQUIRED_DATA_TYPE_PARAMETER(filter, 2, boolean);
579  if (!filter->internal) {
580  return afw_value_null;
581  }
582  }
583 
584  /* Determine log priority. */
585  priority = afw_log_priority_trace1;
586  if (AFW_FUNCTION_PARAMETER_IS_PRESENT(3)) {
587  AFW_FUNCTION_EVALUATE_REQUIRED_DATA_TYPE_PARAMETER(number, 3, integer);
588  if (number->internal < 1 || number->internal > 8) {
589  AFW_THROW_ERROR_Z(general,
590  "Parameter 3 must be an integer between 1 and 8.", x->scope);
591  }
592  priority = (afw_log_priority_t)
593  (number->internal - 1 + afw_log_priority_trace1);
594  }
595 
596  /* Get value as a string. */
597  value = afw_value_evaluate(x->argv[1], x->p, x->scope);
598  if (!afw_value_is_string(value)) {
599  value = afw_value_convert_to_string(value, true, x->p, x->scope);
600  }
601  s = AFW_VALUE_INTERNAL(value);
602 
603  /* Write string to trace log. */
604  afw_trace_write(priority, NULL, AFW__FILE_LINE__, s, x->scope);
605 
606  /* Return null. */
607  return afw_value_null;
608 }
