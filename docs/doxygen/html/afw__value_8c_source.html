154 /* Determine if value and all of it contained values are evaluated. */
155 AFW_DEFINE(afw_boolean_t)
156 afw_value_is_fully_evaluated(
157  const afw_value_t *value,
158  afw_xctx_t *xctx)
159 {
160  afw_boolean_t result;
161  const afw_iterator_t *iterator;
162  const afw_utf8_t *property_name;
163  const afw_value_t *v;
164 
165  result = true;
166 
167  if (!afw_value_is_defined_and_evaluated(value)) {
168  result = false;
169  }
170 
171  else if (afw_value_is_object(value))
172  {
173  for (iterator = NULL;;) {
174  v = afw_object_get_next_property(
175  ((const afw_value_object_t *)value)->internal,
176  &iterator, &property_name, xctx);
177  if (!v) {
178  break;
179  }
180  if (!afw_value_is_fully_evaluated(v, xctx)) {
181  result = false;
182  break;
183  }
184  }
185  }
186 
187  else if (afw_value_is_list(value))
188  {
189  for (iterator = NULL;;) {
190  v = afw_list_get_next_value(
191  ((const afw_value_list_t *)value)->internal,
192  &iterator, xctx->p, xctx);
193  if (!v) {
194  break;
195  }
196  if (!afw_value_is_fully_evaluated(v, xctx)) {
197  result = false;
198  break;
199  }
200  }
201  }
202 
203  return result;
204 }

270 /* Convert value to casted utf8 in specified pool. */
271 AFW_DEFINE(const afw_utf8_t *)
272 afw_value_as_casted_utf8(
273  const afw_value_t *value, const afw_pool_t *p, afw_xctx_t *xctx)
274 {
275  const afw_utf8_t *result;
276  const afw_data_type_t *data_type;
277  const afw_utf8_t * const *s;
278  afw_size_t n;
279  afw_size_t len;
280  afw_utf8_octet_t *c;
281 
282  value = afw_value_evaluate(value, p, xctx);
283  if (!value) {
284  return NULL;
285  }
286  data_type = afw_value_get_data_type(value, xctx);
287 
289  if (afw_value_is_list(value)) {
290  s = afw_value_as_array_of_utf8(value, p, xctx);
291 
292  if (!s[0]) {
293  len = 5; /* "bag()" */
294  }
295  else {
296  for (n = 0, len = 3 /* "bag" */; s[n]; n++) {
297  /*
298  * '<datatype>("<string value>") plus surrounding '()' for bag
299  * or ', '
300  */
301  len += s[n]->len + 6 + data_type->data_type_id.len;
302  }
303  }
304 
305  c = afw_pool_calloc(p, len, xctx);
306  result = afw_utf8_create(c, len, p, xctx);
307 
308  memcpy(c, "bag(", 4);
309  c += 4;
310 
311  for (n = 0; s[n]; n++) {
312  if (n != 0) {
313  *c++ = ',';
314  *c++ = ' ';
315  }
316  memcpy(c, data_type->data_type_id.s,
317  data_type->data_type_id.len);
318  c += data_type->data_type_id.len;
319  *c++ = '(';
320  *c++ = '\"';
321  memcpy(c, s[n]->s, s[n]->len);
322  c += s[n]->len;
323  *c++ = '\"';
324  *c++ = ')';
325  }
326 
327  *c++ = ')';
328  }
329 
330  else {
331  result = afw_utf8_concat(p, xctx,
332  &data_type->data_type_id,
333  &afw_s_a_open_parenthesis, &impl_s_a_quote,
334  afw_value_as_utf8(value, p, xctx),
335  &impl_s_a_quote, &afw_s_a_close_parenthesis,
336  NULL);
337  }
338 
339  return result;
340 }

368  /* Else result is just value. */
369  else {
370  result = value;
371  }

520 /* Convert a value to a value/data type. */
521 AFW_DEFINE(const afw_value_t *)
522 afw_value_convert(
523  const afw_value_t *value,
524  const afw_data_type_t *to_data_type,
525  afw_boolean_t required,
526  const afw_pool_t *p, afw_xctx_t *xctx)
527 {
528  const afw_value_t *result;
529  const afw_data_type_t *v_data_type;
530  afw_value_evaluated_t *single;
531  const afw_list_t *list;
532  const afw_iterator_t *iterator;
533  const void *internal;
534  const afw_data_type_t *data_type;
535  afw_size_t evaluate_count;
536 
537  /* Evaluate value. */
538  result = value;
539  AFW_VALUE_UNDECORATE(result);
540  for (evaluate_count = 0;
541  result && !afw_value_is_defined_and_evaluated(result);
542  evaluate_count++)
543  {
544  if (evaluate_count >= 20) {
545  AFW_THROW_ERROR_FZ(general, xctx,
546  "afw_value_convert() value required > %d evaluations",
547  20);
548  }
549  result = afw_value_evaluate(result, p, xctx);
550  AFW_VALUE_UNDECORATE(result);
551  }
552 
553  if (!result) {
554  if (!required) {
555  return NULL;
556  }
557  AFW_THROW_ERROR_Z(undefined, "Result is undefined", xctx);
558  }
559 
560  /* If to_data_type is any, return result now. */
561  if (!to_data_type || to_data_type == afw_data_type_any) {
562  return result;
563  }
564 
565  v_data_type = afw_value_get_data_type(result, xctx);
566 
567  if (v_data_type != to_data_type) {
568 
569  /* Upconvert to one entry list. */
570  if (to_data_type == afw_data_type_list) {
571  list = afw_list_create_wrapper_for_array(
572  &((afw_value_evaluated_t *)value)->internal, false,
573  v_data_type, 1, p, xctx);
574  result = afw_value_create_list(list, p, xctx);
575  }
576 
577  /* Down convert from a single entry list. */
578  else if (v_data_type == afw_data_type_list &&
579  !afw_data_type_is_string(to_data_type))
580  {
581  list = ((const afw_value_list_t *)value)->internal;
582  if (afw_list_get_count(list, xctx) != 1) {
583  AFW_THROW_ERROR_Z(general,
584  "Can't down convert a list with more than one entry",
585  xctx);
586  }
587  iterator = NULL;
588  afw_list_get_next_internal(list,
589  &iterator, &data_type, &internal, xctx);
590  single = afw_value_evaluated_allocate(to_data_type, p, xctx);
591  if (data_type == to_data_type) {
592  memcpy(AFW_VALUE_INTERNAL(single), internal, data_type->c_type_size);
593  }
594  else {
595  afw_data_type_convert_internal(
596  data_type,
597  &single->internal,
598  internal,
599  to_data_type,
600  p, xctx);
601  }
602  result = (const afw_value_t *)single;
603  }
604 
605  /* Not list. */
606  else {
607  single = afw_value_evaluated_allocate(to_data_type, p, xctx);
608  afw_data_type_convert_internal(
609  v_data_type,
610  &single->internal,
611  &((const afw_value_evaluated_t *)value)->internal,
612  to_data_type,
613  p, xctx);
614  result = (const afw_value_t *)single;
615  }
616  }
617 
618  /* Return either original or converted value. */
619  return result;
620 }

699 impl_value_as_array_of_values(
700  const afw_value_t *value, const afw_pool_t *p, afw_xctx_t *xctx)
701 {
702  afw_size_t count;
703  const afw_value_t **e;
704  const afw_value_t **result;
705  const afw_iterator_t *iterator;
706 
707  /* If value is NULL, n is 0. */
708  if (!value) {
709  count = 0;
710  }
711 
712  /* If value is a list, count is number of entries in list. */
713  else if (afw_value_is_list(value)) {
714  count = afw_list_get_count(
715  ((const afw_value_list_t *)value)->internal, xctx);
716  }
717 
718  /* For other types of values, count is 1. */
719  else {
720  count = 1;
721  }
722 
723  /* Allocate storage for result array with n + 1 entries. */
724  e = result = afw_pool_malloc(p, sizeof(afw_value_t *)*(count + 1), xctx);
725  e[count] = NULL;
726 
727  /* Return now if count is 0. */
728  if (count == 0) {
729  return result;
730  }
731 
732  /* If value is a list, added list values. */
733  else if (afw_value_is_list(value)) {
734  for (iterator = NULL; ; e++) {
735  *e = afw_list_get_next_value(
736  ((const afw_value_list_t *)value)->internal,
737  &iterator, p, xctx);
738  if (!*e) break;
739  }
740  }
741 
742  /* For other types of values, just add them. */
743  else {
744  *e = value;
745  }
746 
747  /* Return result. */
748  return result;
749 }

891  if (!substring_data_type ||

896  xctx);

904  result = true;
