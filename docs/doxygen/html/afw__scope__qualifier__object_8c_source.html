65 afw_scope_qualifiers_object_create(
66  afw_boolean_t for_testing,
67  const afw_pool_t *p,
68  afw_scope_t *scope)
69 {
70  const afw_object_t *qualifiers;
71  const afw_object_t *qualifier;
72  const afw_scope_qualifier_stack_entry_t *c;
73 
74  qualifiers = afw_object_create(p, scope);
75  for (c = scope->qualifier_stack->top;
76  c >= scope->qualifier_stack->first;
77  c--)
78  {
79  if (!afw_object_has_property(qualifiers, &c->qualifier, scope)) {
80  qualifier = afw_scope_qualifier_object_create(
81  &c->qualifier, for_testing,
82  qualifiers->p, scope);
83  afw_object_set_property_as_object(qualifiers,
84  &c->qualifier, qualifier, scope);
85  }
86  }
87 
88  return qualifiers;
89 }
