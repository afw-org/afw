82 /**
83  * The ListResponsive is a component that wraps ListEditor to provide options
84  * derived from an AfwProperty of dataType list. It handles the modals to
85  * prompt for new values and manages the options when new list entries are added
86  * and removed.
87  */
88 export const ListResponsive = (props) => {
89 
90  const [showModal, setShowModal] = useState(false);
91  const [modalTitle, setModalTitle] = useState("");
92  const [addItem, setAddItem] = useState(false);
93  const [items, setItems] = useState([]);
94  const [savable, setSavable] = useState();
95  const [editValue, setEditValue] = useState();
96  const theme = useTheme();
97 
98  const property = useAfwProperty(props);
99  const {editable} = useOperational(props);
100 
101  /* track when the property gets values added */
102  const onChanged = useEventId({ property, eventId: "onChanged" });
103 
104  const {id, "aria-label": ariaLabel} = props;
105  const label = property?.getLabel() || property?.getName();
106  const description = property?.getBrief() || property?.getDescription();
107 
108  useEffect(() => {
109  const items = [];
110 
111  if (property && property.getValue() !== undefined) {
112  const label = property.getLabel() ? property.getLabel() : property.getName();
113  property.getValue().forEach((value, index) => {
114  let text;
115 
116  // \fixme the heuristic here may be better off requesting the dataType metadata handlers
117  // since they know how to handle the specific dataType. For example base64Binary
118  // or hexBinary.
119  if (value.getValue() !== undefined) {
120  if (value.getDataType() === "object") {
121  const descriptionPropertyName = value.getValue().getDescriptionPropertyName();
122  if (descriptionPropertyName)
123  text = value.getValue().getPropertyValue(descriptionPropertyName);
124  } else if (value.getValue() !== undefined) {
125  // \fixme look at CSS text-overflow to make sure they're not too long.
126  text = String(value.getValue());
127  }
128 
129  if (text === undefined)
130  text = label + " [" + String(index) + "]";
131 
132  items.push({ text, value, index });
133  }
134  });
135  }
136 
137  setItems(items);
138  }, [property, onChanged]);
139 
140  const onAddItem = () => {
141  if (property) {
142  /* if this is the first value for this property, then we need to allocate a new AfwList */
143  if (!property.getValue()) {
144  property.setValue([]);
145  }
146 
147  /* create a new value on this list */
148  const editValue = property.getValue().newValue();
149  editValue.addEventListener("onSavable", setSavable);
150 
151  setEditValue(editValue);
152  setModalTitle("Add New Value");
153  }
154 
155  setShowModal(true);
156  setAddItem(true);
157  };
158 
159  const onEditItem = (item) => {
160  const editValue = item.value;
161 
162  editValue.detach();
163  editValue.addEventListener("onSavable", setSavable);
164 
165  setEditValue(editValue);
166  setModalTitle("Edit Value");
167  setShowModal(true);
168  setAddItem(false);
169  };
170 
171  const onDeleteItems = (deleteItems) => {
172 
173  let newItems = items.filter(item => {
174  for (let i = 0; i < deleteItems.length; i++) {
175  let deleteItem = deleteItems[i];
176  if (item.index === deleteItem.index)
177  return false;
178  }
179 
180  return true;
181  });
182 
183  /* actually remove the values from the property, too */
184  deleteItems.forEach(item => item.value.remove());
185 
186  setItems(newItems);
187  };
188 
189  const onSave = () => {
190  editValue.attach();
191 
192  const items = [];
193 
194  if (property && property.getValue()) {
195  const label = property.getLabel() ? property.getLabel() : property.getName();
196  property.getValue().forEach((value, index) => {
197  let text;
198 
199  if (value.getDataType() === "object") {
200  const descriptionPropertyName = value.getValue().getDescriptionPropertyName();
201  if (descriptionPropertyName)
202  text = value.getValue().getPropertyValue(descriptionPropertyName);
203  } else if (value.getValue() !== undefined)
204  text = String(value.getValue());
205 
206  if (!text)
207  text = label + " [" + String(index) + "]";
208 
209  items.push({ text, value, index });
210  });
211  }
212 
213  setItems(items);
214  setShowModal(false);
215  };
216 
217  const onCancel = () => {
218  if (addItem)
219  editValue.remove();
220  else
221  editValue.discardChanges();
222 
223  setShowModal(false);
224  };
225 
226  if (editable !== true) {
227  return (
228  <ListResponsiveReadonly
229  {...props}
230  items={items}
231  />
232  );
233  }
234 
235  return (
236  <>
237  <ListEditor
238  aria-label={ariaLabel}
239  label={label}
240  description={description}
241  items={items}
242  editable={editable}
243  onAddItem={onAddItem}
244  onEditItem={onEditItem}
245  onDeleteItems={onDeleteItems}
246  />
247  <Dialog
248  open={showModal}
249  title={modalTitle}
250  isBlocking={false}
251  showClose={true}
252  maxWidth={(editValue && editValue.getDataType() === "object") ? "md" : "sm"}
253  onDismiss={() => setShowModal(false)}
254  contains={
255  <div>
256  {
257  editValue &&
258  <AdaptiveLayoutMergeProviders layoutParameters={{ nestingDepth: 0, headerOptions: "all" }}>
259  <ValueLayout
260  // eslint-disable-next-line jsx-a11y/no-autofocus
261  autoFocus={true}
262  id={id + ".ListResponsive.ValueLayout"}
263  value={editValue}
264  valueMeta={{ ...property.getMeta(), ...editValue.getMeta() }}
265  />
266  </AdaptiveLayoutMergeProviders>
267  }
268  </div>
269  }
270  footer={
271  <div style={{ display: "flex", padding: theme.spacing(2), textAlign: "right" }}>
272  <Button
273  color="primary"
274  variant="contained"
275  style={{ marginRight: theme.spacing(1) }}
276  label="Ok"
277  disabled={!savable}
278  onClick={onSave}
279  />
280  <Button
281  label="Cancel"
282  onClick={onCancel}
283  />
284  </div>
285  }
286  />
287  </>
288  );
289 };
