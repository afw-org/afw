36  */

39  property.setValue( property.getDefaultValue() );

42  else if (property.getDataType() === "object")
43  property.setValue({});
44

53  /**

55  *
56  * @param {object} otherPropertyType
57  */
58  const onSelectOtherProperty = (otherPropertyType) => {
59  // \fixme useReducer?
60  setSelectedPropertyType(otherPropertyType);

64  setObjectResponsivePropertyTypeOpen(true);
65  };

69

71  let onlyOtherProperties;
72 
73  if (object) {

75  const properties = object.getProperties().sort((a, b) => {
76  const A = a.getLabel() ? a.getLabel() : a.getName();
77  const B = b.getLabel() ? b.getLabel() : b.getName();
78 
79  return A.toLowerCase().localeCompare(B.toLowerCase());
80  });
81 
82  /* build the menu items. \fixme there may be more to exclude by ObjectResponsive (if layout displays more than just "valued" for example) */
83  properties.forEach(property => {

92  }
93  });
94 
95  /* check for "otherProperties" */
96  const otherPropertyType = object.getPropertyType();
97  if (otherPropertyType !== undefined) {
98  if (propertyMenuItems.length > 0)
99  propertyMenuItems.push({ key: "divider", type: "divider" });
100  else

108  }
109  }
110

112  }, [object]);

162  tooltip="Add New Property"
163  color="primary"
164  disabled={propertyMenuItems.length === 0}
165  onClick={event => {

167  onSelectOtherProperty(object.getPropertyType());

169  setAddMenuTarget(event.target);
170  }}

172  <Button

174  type="icon"
