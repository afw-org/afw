186 AFW_DEFINE_STATIC_INLINE(afw_size_t)
187 afw_endian_safe_big_uint64_to_native_size_t(
188  afw_endian_big_uint64_t big, afw_xctx_t *xctx)
189 {
190  afw_uint64_t result = big.i;
191 
192  /* Convert to native and throw error if it didn't fit afw_size_t. */
193  AFW_ENDIAN_BIG_TO_NATIVE_64(&result);
194  if (result != (afw_size_t)result) {
195  AFW_THROW_ERROR_Z(general,
196  "Value in afw_endian_big_uint64_t exceeds the value that will "
197  "fit in afw_size_t",
198  xctx);
199  }
200 
201  return (afw_size_t)result;
202 }
