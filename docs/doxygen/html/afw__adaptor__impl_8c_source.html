212 afw_adaptor_impl_create_cede_p(
213  const afw_adaptor_inf_t *inf,
214  afw_size_t instance_size,
215  const afw_object_t *properties,
216  const afw_pool_t *p,

224  afw_boolean_t b;
225

227  adaptor = (afw_adaptor_t *)afw_pool_calloc(p,
228  (instance_size != 0) ? instance_size : sizeof(afw_adaptor_t),
229  scope);
230  adaptor->inf = &impl_afw_adaptor_inf;
231  adaptor->p = p;
232  adaptor->adaptor_type_id = afw_object_old_get_property_as_string(
233  properties, &afw_s_adaptorType, scope);
234  impl = afw_pool_calloc_type(p, afw_adaptor_impl_t, scope);
235  adaptor->impl = impl;
236  impl->adaptor = adaptor;
237  impl->wrapped_inf = inf;
238 
239  /* Prepare properties. */
240  adaptor->properties = afw_environment_prepare_conf_type_properties(
241  properties, scope);
242 
243  /* Get adaptorType from properties. */
244  adaptor->adaptor_type_id = afw_object_old_get_property_as_string(
245  properties, &afw_s_adaptorType, scope);
246 
247  /* Get source location. Default it to adaptor. */
248  impl->source_location = afw_object_old_get_property_as_string(
249  properties, &afw_s_sourceLocation, scope);
250  if (!impl->source_location) {
251  impl->source_location = &afw_s_adaptor;
252  }
253 
254  /* Get adaptor_id from parameters. */
255  s = afw_object_old_get_property_as_utf8(properties,
256  &afw_s_adaptorId, p, scope);
257 
258  if (!s) {
259  AFW_THROW_ERROR_FZ(general, scope,
260  "%" AFW_UTF8_FMT " requires id property.",
261  AFW_UTF8_FMT_ARG(impl->source_location));
262  }
263  afw_memory_copy(&adaptor->adaptor_id, s);
264  impl->adaptor_id = &adaptor->adaptor_id;
265 
266  /* Create lock. */
267  s = afw_utf8_printf(p, scope,
268  "adaptor_id:%" AFW_UTF8_FMT,
269  AFW_UTF8_FMT_ARG(impl->adaptor_id));
270  if (!afw_environment_get_lock(s, scope)) {
271  impl->adaptor_lock_rw = afw_lock_create_rw_and_register(
272  afw_utf8_clone(s, scope->env->p, scope),
273  afw_utf8_printf(scope->env->p, scope,
274  "Adaptor id %" AFW_UTF8_FMT " read/write lock",
275  AFW_UTF8_FMT_ARG(impl->adaptor_id)),
276  afw_utf8_printf(scope->env->p, scope,
277  "Read/write lock used internally by adaptor id %" AFW_UTF8_FMT " implementation.",
278  AFW_UTF8_FMT_ARG(impl->adaptor_id)),
279  scope);
280  }
281 
282  /* Service id. */
283  adaptor->service_id = afw_utf8_printf(p, scope,
284  "adaptor-%" AFW_UTF8_FMT,
285  AFW_UTF8_FMT_ARG(impl->adaptor_id));
286 
287  /* Set trace fields */
288  impl_set_trace_flag_fields(adaptor, scope);
289 
290  /* Trace create */
291  afw_trace_fz(1, adaptor->trace_flag_index, adaptor, scope,
292  "adaptorId %" AFW_UTF8_FMT " is being created",
293  AFW_UTF8_FMT_ARG(&adaptor->adaptor_id));
294 
295  /* As default, allow read _AdaptiveObjectType_/_AdaptiveObjectType_. */
296  impl->supported_core_object_types = apr_hash_make(
297  afw_pool_get_apr_pool(p));
298  afw_adaptor_impl_set_supported_core_object_type(adaptor,
299  &afw_s__AdaptiveObjectType_, true, false, scope);
300 
301  /* Get optional authorizationHandlerId from parameters. */
302  impl->authorization_handler_id =
303  afw_object_old_get_property_as_string(adaptor->properties,
304  &afw_s_authorizationHandlerId, scope);
305  if (impl->authorization_handler_id)
306  {
307  authorization_handler = NULL;
308  AFW_TRY{
309  authorization_handler = afw_authorization_handler_get_reference(
310  impl->authorization_handler_id, scope);
311  }
312  AFW_FINALLY {
313  if (authorization_handler) {
314  afw_authorization_handler_destroy(authorization_handler, scope);
315  }
316  }
317  AFW_ENDTRY;
318  if (!authorization_handler) {
319  AFW_THROW_ERROR_FZ(general, scope,
320  "%" AFW_UTF8_FMT
321  " authorizationHandlerId %" AFW_UTF8_FMT
322  " is not startable",
323  AFW_UTF8_FMT_ARG(impl->source_location),
324  AFW_UTF8_FMT_ARG(impl->authorization_handler_id));
325  }
326  }
327 
328  /* Get optional journalAdaptorId from parameters. */
329  impl->journal_adaptor_id = afw_object_old_get_property_as_string(
330  adaptor->properties, &afw_s_journalAdaptorId, scope);
331  if (impl->journal_adaptor_id) {
332  AFW_LOG_FZ(debug, scope,
333  "Adaptor %" AFW_UTF8_FMT
334  " specified journalAdaptorId %" AFW_UTF8_FMT
335  ".",
336  AFW_UTF8_FMT_ARG(&adaptor->adaptor_id),
337  AFW_UTF8_FMT_ARG(impl->journal_adaptor_id));
338  }
339 
340  if (impl->journal_adaptor_id &&
341  !afw_utf8_equal(s, &adaptor->adaptor_id))
342  {
343  temp_session = NULL;
344  AFW_TRY{
345  temp_session = afw_adaptor_session_create(
346  impl->journal_adaptor_id, scope);
347  if (!temp_session ||
348  !afw_adaptor_session_get_journal_interface(temp_session, scope))
349  {
350  AFW_THROW_ERROR_FZ(general, scope,
351  "%" AFW_UTF8_FMT
352  " Adaptor id %" AFW_UTF8_FMT
353  " specified in journalAdaptorId property is not startable"
354  " or does not support journaling.",
355  AFW_UTF8_FMT_ARG(impl->source_location),
356  AFW_UTF8_FMT_ARG(impl->journal_adaptor_id));
357  }
358  }
359  AFW_FINALLY{
360  if (temp_session) {
361  afw_adaptor_session_release(temp_session, scope);
362  }
363  }
364  AFW_ENDTRY;
365  }
366 
367  /* Authorization */
368  impl->authorization = afw_object_get_property(
369  properties, &afw_s_authorization, scope);
370 
372  /* Create runtime metrics object and set in properties. */
373  impl->metrics_object = afw_runtime_object_create_indirect(
374  &afw_s__AdaptiveAdaptorMetrics_,
375  &adaptor->adaptor_id, impl, scope->env->p, scope);
376 
377  /* If this is layout adaptor id, allow layout object type. */
378  if (scope->env->layout_adaptor_id &&
379  afw_utf8_equal(scope->env->layout_adaptor_id, &adaptor->adaptor_id))
380  {
381  afw_adaptor_impl_set_supported_core_object_type(adaptor,
382  &afw_s__AdaptiveLayoutComponentType_, true, true, scope);
383  afw_adaptor_impl_set_supported_core_object_type(adaptor,
384  &afw_s__AdaptiveLayoutComponent_, false, true, scope);
385  }
386 
387  /* If isModelLocation is true, provide appropriate object types. */
388  b = afw_object_old_get_property_as_boolean_deprecated(properties,
389  &afw_s_isModelLocation, scope);
390  if (b) {
391  impl->model_location = afw_model_location_create(adaptor, p, scope);
392  afw_adaptor_impl_set_supported_core_object_type(adaptor,
393  &afw_s__AdaptiveHybridProperties_, false, true, scope);
394  afw_adaptor_impl_set_supported_core_object_type(adaptor,
395  &afw_s__AdaptiveModel_, true, true, scope);
396  afw_adaptor_impl_set_supported_core_object_type(adaptor,
397  &afw_s__AdaptiveModelObjectTypes_, false, true, scope);
398  afw_adaptor_impl_set_supported_core_object_type(adaptor,
399  &afw_s__AdaptiveModelObjectType_, false, true, scope);
400  afw_adaptor_impl_set_supported_core_object_type(adaptor,
401  &afw_s__AdaptiveModelPropertyTypes_, false, true, scope);
402  afw_adaptor_impl_set_supported_core_object_type(adaptor,
403  &afw_s__AdaptiveModelPropertyType_, false, true, scope);
404  }
405 
439  /* Return new adaptor. */
440  return adaptor;
441 }

524 AFW_DEFINE(void)
525 afw_adaptor_impl_set_supported_core_object_type(
526  const afw_adaptor_t *adaptor,
527  const afw_utf8_t *object_type_id,
528  afw_boolean_t allow_entity,
529  afw_boolean_t allow_write,
530  afw_scope_t *scope)
531 {
532  const afw_pool_t *p = adaptor->p;
533  apr_hash_t *ht;
534  afw_adaptor_impl_core_object_type_t *e;
535  const afw_utf8_t *path;
536  afw_value_list_t *parent_paths;
537 
538  /* Ignore call for afw adaptor. Will cause parentPaths loops. */
539  if (afw_utf8_equal(&adaptor->adaptor_id, &afw_s_afw)) {
540  return;
541  }
542 
543  ht = adaptor->impl->supported_core_object_types;
544 
545  e = apr_hash_get(ht, object_type_id->s, object_type_id->len);
546  if (e) {
547  e->allow_entity = allow_entity;
548  e->allow_write = allow_write;
549  impl_update_allow(e->object, allow_entity, allow_write, scope);
550  }
551  else {
552  e = afw_pool_calloc_type(p,
553  afw_adaptor_impl_core_object_type_t, scope);
554  e->object_type_id = object_type_id;
555  e->object = afw_runtime_get_object(&afw_s__AdaptiveObjectType_,
556  object_type_id, scope);
557  if (!e->object) {
558  AFW_THROW_ERROR_Z(general, "Invalid object type id", scope);
559  }
561  e->object = afw_object_create_clone(e->object, p, scope);
562  path = afw_object_meta_get_path(e->object, scope);
563  afw_object_meta_set_ids(e->object, &adaptor->adaptor_id,
564  &afw_s__AdaptiveObjectType_, object_type_id, scope);
565  parent_paths = afw_value_allocate_list(p, scope);
566  parent_paths->internal = afw_list_create_wrapper_for_array(
567  (const void *)path, false, afw_data_type_anyURI, 1, p, scope);
568  afw_object_meta_set_parent_paths(e->object, parent_paths, scope);
569 
570  impl_update_allow(e->object, allow_entity, allow_write, scope);
571 
572  apr_hash_set(ht, e->object_type_id->s, e->object_type_id->len, e);
573  }
574 }
