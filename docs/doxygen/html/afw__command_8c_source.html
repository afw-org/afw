133

150 /* Get input. */
151 static afw_utf8_t *
152 impl_get_input(afw_command_self_t *self, afw_scope_t *scope)
153 {
154  int c;
155  char prev_c;
156 
157  /* If eof, just return NULL. */
158  if (self->eof) return NULL;
159 
160  /* Allocate array for input. */
161  if (!self->input_buffer) {
162  self->input_buffer = apr_array_make(
163  afw_pool_get_apr_pool(self->scope->p),
164  2000, 1);
165  }
166 
167  /* Read input up to a \n that is not preceded by a \ */
168  apr_array_clear(self->input_buffer);
169  for (prev_c = 0; ; prev_c = c) {
170  c = fgetc(self->fd_input);
171  if (c == EOF) {
172  self->eof = true;
173  break;
174  }
175  if (c == '\n' && prev_c == '\\') {
176  apr_array_pop(self->input_buffer);
177  continue;
178  }
179  APR_ARRAY_PUSH(self->input_buffer, unsigned char) = c;
180  if (c == '\n') break;
181  }
182 
183  /* Return resulting string making sure it is NFC utf8. */
184  return (self->input_buffer->nelts == 0)
185  ? NULL
186  : (afw_utf8_t *)afw_utf8_create(self->input_buffer->elts,
187  self->input_buffer->nelts, scope->p, scope);
188 }

191 /*
192  * If exit_code is not NULL, the exit_code is returned instead of printing
193  * output.
194  */
195 static afw_boolean_t
196 impl_evaluate(
197  afw_command_self_t *self,
198  const afw_utf8_t *input,
199  int *exit_code)
200 {
201  const afw_value_t *value;
202  const afw_value_t *evaluated_value;
203  const afw_utf8_t *string;
204  afw_scope_t *scope;
205  afw_boolean_t error_occurred;
206  afw_boolean_t keep_going;
207  afw_utf8_t *line;
208 
209 
210  scope = afw_scope_create(&afw_command_s_afw_command, 0, self->scope);
211  error_occurred = false;
212  keep_going = true;
213  if (exit_code) {
214  *exit_code = 0;
215  }
216  AFW_TRY {
217 
218  /* If input is not NULL, insure it is NFC normalized utf-8. */
219  if (input) {
220  input = afw_utf8_create(input->s, input->len, scope->p, scope);
221  }
222 
223  /* If input is NULL, get a non-empty input or break if eof. */
224  else if (!self->read_full) {
225  for (;;) {
226  line = impl_get_input(self, scope);
227  if (!line) {
228  keep_going = false;
229  break;
230  }
231  if (line->len != 0 && !afw_utf8_equal_utf8_z(line, "\n")) {
232  /* If this is a template, remove trailing \n. */
233  if (self->compile_option == afw_compile_type_template ||
234  (self->compile_option == afw_compile_type_hybrid &&
235  *line->s != '['))
236  {
237  if (line->s[line->len - 1] == '\n') {
238  line->len--;
239  }
240  }
241  input = line;
242  break;
243  }
244  impl_print_end(self);
245  }
246  }
247  if (!keep_going) break;
248 
249  /* If input is "exit", just set keep_going to false and break. */
250  if (input) {
251  if (afw_utf8_equal_utf8_z(input, "exit\n") ||
252  afw_utf8_equal_utf8_z(input, "exit"))
253  {
254  keep_going = false;
255  break;
256  }
257  }
258 
259  /* Compile eval and evaluate if not check mode. */
260  value = afw_compile_to_value_with_callback(input,
261  self->callback, self->callback_data,
262  self->source_location, self->compile_option,
263  self->residual_check,
264  NULL, NULL, scope->p, scope);
265 
266  /*
267  * If option is parenthesized_expression, NULL can be returned if
268  * everything remaining is whitespace/comments. In this case, set
269  * keep_going to false and break.
270  */
271  if (!value) {
272  keep_going = false;
273  break;
274  }
275 
276  /* If not check mode, evaluate expression and print or return result. */
277  if (!self->check_mode) {
278  evaluated_value = afw_value_evaluate(value, scope->p, scope);
279 
280  /* If requested, determine exit code. */
281  if (exit_code) {
282  if (!afw_value_is_null(evaluated_value)) {
283  if (!afw_value_is_integer(evaluated_value) ||
284  ((const afw_value_integer_t *)evaluated_value)
285  ->internal < 0 ||
286  ((const afw_value_integer_t *)evaluated_value)
287  ->internal > 255)
288  {
289  *exit_code = EXIT_FAILURE;
290  fprintf(scope->env->stderr_fd,
291  "Adaptive script run from shell must return "
292  "null or an integer between 0 and 255 but returned "
293  "this instead:\n");
294  string = afw_value_as_casted_utf8(evaluated_value,
295  scope->p, scope);
296  if (string) {
297  fprintf(scope->env->stderr_fd,
298  "%" AFW_UTF8_FMT "\n", AFW_UTF8_FMT_ARG(string));
299  }
300  }
301  else {
302  *exit_code = (int)
303  ((const afw_value_integer_t *)evaluated_value)
304  ->internal;
305  }
306  }
307  }
308 
309  /* If not requesting exit code, print result. */
310  else {
311 
312  /* JSON does not support undefined values. */
313  if (afw_value_is_undefined(evaluated_value)) {
314  impl_print_result(self,
315  "%" AFW_UTF8_FMT,
316  AFW_UTF8_FMT_ARG(&afw_s_a_undefined_as_string));
317  }
318 
319  /* If value is defined and evaluated, represent as JSON. */
320  else if (afw_value_is_defined_and_evaluated(evaluated_value)) {
321  string = afw_json_from_value(evaluated_value,
322  &afw_object_options_whitespace,
323  scope->p, scope);
324  impl_print_result(self, "%" AFW_UTF8_FMT, AFW_UTF8_FMT_ARG(string));
325  }
326 
327  /* If defined but not evaluated, return decompiled value. */
328  else {
329  string = afw_value_decompile_to_string(evaluated_value, NULL,
330  scope->p, scope);
331  impl_print_result(self,
332  "<unevaluated> %" AFW_UTF8_FMT,
333  AFW_UTF8_FMT_ARG(string));
334  }
335  }
336  }
337  }
338 
339  AFW_CATCH_UNHANDLED{
340  impl_print_error(self, AFW_ERROR_THROWN, scope);
341  error_occurred = true;
342  }
343 
344  AFW_FINALLY{
345  afw_adaptor_session_commit_and_release_cache(error_occurred, scope);
346  afw_scope_release(scope, scope);
347  /* Special case: scope is gone, so return before AFW_ENDTRY. */
348  if (keep_going) impl_print_end(self);
349  return keep_going && !self->terminal_error;
350  }
351 
352  AFW_ENDTRY;
353 
354  /* Fix warning. Should not get to here. See Special case comment above. */
355  return false;
356 }

393 static int
394 process_args_getopt(afw_command_self_t *self, int argc, const char * const *argv,
395  afw_scope_t *scope)
396 {
397  apr_getopt_t *os;
398  int option_ch;
399  const char * option_arg;
400  int rv;
401 
402  /* Parse parameters. */
403  if ((apr_getopt_init(&os, afw_pool_get_apr_pool(scope->p), argc, argv))
404  != APR_SUCCESS)
405  {
406  fprintf(scope->env->stderr_fd, "apr_getopt_init() error.\n");
407  return EXIT_FAILURE;
408  }
409 
410  self->compile_option = afw_compile_type_error;
411  self->residual_check = afw_compile_residual_check_to_full;
412  self->index_first_non_option = 1;
413  while ((rv = apr_getopt_long(os, opts, &option_ch, &option_arg))
414  == APR_SUCCESS)
415  {
416  self->index_first_non_option = os->ind;
417  switch (option_ch) {
418 
419  case 'f':
420  self->conf_z = option_arg;
421  self->conf.len = strlen(self->conf_z);
422  break;
423 
424  case 'e':
425  self->extension_z = option_arg;
426  self->extension.len = strlen(self->extension_z);
427  break;
428 
429  case 'k':
430  self->check_mode = true;
431  break;
432 
433  case 'l':
434  self->local_mode_z = option_arg;
435  self->local_mode_out.len = strlen(self->local_mode_z);
436  break;
437 
438  case 's':
439 
440  if (afw_utf8_z_equal(option_arg, "expression_tuple"))
441  {
442  self->compile_option = afw_compile_type_expression_tuple;
443  self->can_span_lines = false;
444  }
445 
446  else if (afw_utf8_z_equal(option_arg, "expression"))
447  {
448  self->compile_option = afw_compile_type_expression;
449  self->can_span_lines = false;
450  }
451 
452  else if (afw_utf8_z_equal(option_arg, "hybrid"))
453  {
454  self->compile_option = afw_compile_type_hybrid;
455  self->can_span_lines = true;
456  }
457 
458  else if (afw_utf8_z_equal(option_arg, "parenthesized"))
459  {
460  self->compile_option = afw_compile_type_parenthesized_expression;
461  self->can_span_lines = true;
462  }
463 
464  else if (afw_utf8_z_equal(option_arg, "script"))
465  {
466  self->compile_option = afw_compile_type_script;
467  self->can_span_lines = false;
468  }
469 
470  else if (afw_utf8_z_equal(option_arg, "template"))
471  {
472  self->compile_option = afw_compile_type_template;
473  self->can_span_lines = false;
474  }
475 
476  else if (afw_utf8_z_equal(option_arg, "test_script"))
477  {
478  self->compile_option = afw_compile_type_test_script;
479  self->can_span_lines = false;
480  }
481  else {
482  rv = fprintf(stderr, "Invalid option --syntax %s\n", option_arg);
483  if (rv < 0) exit(EXIT_FAILURE);
484  print_usage();
485  return EXIT_FAILURE;
486  }
487 
488  break;
489 
490  case 't':
491  self->type_in_z = option_arg;
492  self->type_in.len = strlen(self->type_in_z);
493  break;
494 
495  case 'v':
496  rv = fprintf(stderr, "%s\n", AFW_COMMAND_VERSION_COMMAND_STRING);
497  if (rv < 0) exit(EXIT_FAILURE);
498  self->help_option = true;
499  return EXIT_SUCCESS;
500 
501  case 'x':
502  self->expression_z = option_arg;
503  self->expression.len = strlen(self->expression_z);
504  break;
505 
506  case 'h':
507  self->help_option = true;
508  print_usage();
509  return EXIT_SUCCESS;
510 
511  default:
512  rv = fprintf(stderr, "Error: Invalid command line parameter.\n");
513  if (rv < 0) exit(EXIT_FAILURE);
514  print_usage();
515  return EXIT_FAILURE;
516  }
517  }
518 
519  /* Error if apr_getopt_long() returns other than success. */
520  if (rv != APR_EOF) {
521  fprintf(scope->env->stderr_fd, "Try --help.\n");
522  return EXIT_FAILURE;
523  }
524 
525  /* Normal return. */
526  return EXIT_SUCCESS;
527 }

530 static int
531 process_args(afw_command_self_t *self, int argc, const char * const *argv,
532  afw_scope_t *scope)
533 {
534  int rv;
535 
536  /* Process args with the ones from command line. */
537  rv = process_args_getopt(self, argc, argv, scope);
538  if (rv != EXIT_SUCCESS) {
539  return rv;
540  }
541 
542  /* Default type to a relaxed json syntax. */
543  if (!self->type_in_z) {
544  self->type_in.s = "json";
545  }
546  self->type_in.len = strlen(self->type_in.s);
547 
548  /* Can have no more than 1 positional. */
549  if (argc > self->index_first_non_option + 1) {
550  rv = fprintf(stderr, "Error: Can not specify more than one positional parameter.\n");
551  if (rv < 0) exit(EXIT_FAILURE);
552  print_usage();
553  goto error;
554  }
555 
556  /* Set input if [IN] specified. */
557  if (argc == self->index_first_non_option + 1) {
558  self->in_z = argv[self->index_first_non_option];
559  self->in.len = strlen(self->in_z);
560  }
561 
562  /* Can not specify both -l and [IN] */
563  if (self->in_z && self->local_mode_z) {
564  rv = fprintf(stderr, "Error: Can not specify [IN] if -l specified.\n");
565  if (rv < 0) exit(EXIT_FAILURE);
566  print_usage();
567  goto error;
568  }
569 
570  /* Default -s is script if [IN] specified and e if not. */
571  if (self->compile_option == afw_compile_type_error)
572  {
573  self->compile_option = (self->in_z)
574  ? afw_compile_type_script
575  : afw_compile_type_expression;
576  }
577 
578  /* Interactive mode if not -x or [IN]. */
579  if (!self->local_mode_z && !self->expression_z && !self->in_z) {
580  self->interactive_mode = true;
581  }
582 
583  /* Normal return. */
584  return EXIT_SUCCESS;
585 
586  /* Error return. */
587 error:
588  return EXIT_FAILURE;
589 }

635 int
636 main(int argc, const char * const *argv) {
637  afw_command_self_t *self;
638  const afw_error_t *create_error ;
639  afw_scope_t *scope;
640  int rv;
641  const afw_memory_t *conf_file;
642  const afw_value_t *conf;
643  const char *s;
644 
645  /* Create Adaptive Framework environment for command. */
646  AFW_ENVIRONMENT_CREATE(scope, argc, argv, &create_error);
647  if (!scope) {
648  afw_error_print(scope->env->stderr_fd, create_error);
649  return EXIT_FAILURE;
650  }
651 
652  /* stdout except for result goes to stderr.*/
653  afw_environment_set_stdout_fd(stderr, scope);
654 
655  /* Generated register. */
656  afw_command_generated_register(scope);
657 
658  /* Make sure environment is cleaned up. */
659  self = NULL;
660  AFW_TRY{
661 
662  /* Allocate and initialize self. */
663  self = afw_scope_calloc_type(afw_command_self_t, scope);
664  self->scope = scope;
665  self->fd_input = stdin;
666  self->fd_output = stdout;
667  self->source_location = &afw_command_s_afw_command;
668 
669  /* Process arguments. */
670  rv = process_args(self, argc, argv, scope);
671  if (self->help_option || rv != EXIT_SUCCESS) {
672  break;
673  }
674

685  self->content_type_in = afw_environment_get_content_type(

694  conf_file, &self->conf, scope->p, scope);
695  if (!afw_value_is_list(conf)) {

698  }

702  }

705  if (self->expression_z) {
706  self->source_location = &afw_command_s_afw_command_dash_x;
707  impl_evaluate(self, &self->expression, NULL);
708  self->source_location = &afw_command_s_afw_command;
709  }
710

712  if (self->in_z) {
713  self->source_location = &self->in;
714  self->fd_input = fopen(self->in_z, "r");
715  if (!self->fd_input) {
716  fprintf(scope->env->stderr_fd,
717  "Error opening input %s errno %d - %s\n",
718  self->in_z, errno, strerror(errno));
719  return -1;
720  }
721  }
722

724  if (self->local_mode_z) {
725  self->source_location = &afw_command_s_afw_command_local_mode;
726  for (s = self->local_mode_z; *s && *s >= '0' && *s <= '9'; s++);
727  if (*s == 0)
728  {
729  rv = atoi(self->local_mode_z);
730  if (rv > 0) {
731  self->fd_output = fdopen(rv, "w");
732  }
733  }
734  else {
735  self->fd_output = fopen(self->local_mode_z, "w");
736  }
737  if (!self->fd_output) {
738  fprintf(scope->env->stderr_fd,
739  "Error opening -l path %s errno %d - %s\n",
740  self->local_mode_z, errno, strerror(errno));
741  return -1;
742  }
743  }
744 
745  if (self->interactive_mode) {
746  self->source_location = &afw_command_s_afw_command_interactive;
747  }
748 
749  /* If script, only evaluate once. */
750  if (self->compile_option == afw_compile_type_script ||
751  self->compile_option == afw_compile_type_test_script)
752  {
753  self->read_full = true;
754  self->callback = impl_octet_get_cb;
755  self->callback_data = self;
756  self->residual_check = afw_compile_residual_check_to_full;
757  if (self->compile_option == afw_compile_type_script) {
758  impl_evaluate(self, NULL, &rv);
759  }
760  else {
761  impl_evaluate(self, NULL, NULL);
762  rv = 0;
763  }
764  }
765 
766  /* If local mode, start local server and run. */
767  else if (self->local_mode_z) {
768  self->local_server = afw_command_local_server_create(self);
769  afw_server_run(self->local_server, NULL, scope);
770  }
771 
772  /*
773  * If local or interactive mode, loop evaluating expressions until
774  * exit.
775  */
776  else if (self->in_z || self->interactive_mode) {
777  if (self->can_span_lines) {
778  self->callback = impl_octet_get_cb;
779  self->callback_data = self;
780  self->residual_check = afw_compile_residual_check_to_newline;
781  }
782  if (self->interactive_mode) {
783  impl_print_result(self,
784  "afw " AFW_VERSION_STRING "\n\n"
785  "Input will be evaluated as entered. "
786  "Type exit to end.\n");
787  }
788  impl_print_end(self);
789 
790  while (impl_evaluate(self, NULL, NULL));
791  }
792  }
793 
794  /* Print any unhandled errors. */
795  AFW_CATCH_UNHANDLED{
796  impl_print_error(self, AFW_ERROR_THROWN, scope);
797  impl_print_end(self);
798  rv = EXIT_FAILURE;
799  }
800

802  AFW_FINALLY{
803

805  if (self && self->fd_input && self->in_z) {
806  fclose(self->fd_input);
807  }
808 
809  /* Release enviornment. */
810  afw_environment_release(scope);
811 
812  /* Special case: scope is gone, so return before AFW_ENDTRY. */
813  return rv;
814  }
815 
816  AFW_ENDTRY;
