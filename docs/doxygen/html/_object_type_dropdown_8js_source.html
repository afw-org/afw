16 /**
17  * The ObjectTypeDropdown component read Object Types from an adaptor and
18  * presents them in Dropdown component, so one can be selected.
19  *
20  * @param {string} adaptorId
21  * @param {string} value
22  * @param {boolean} requireEntity
23  * @param {function} onChanged
24  * @param {string} label
25  * @param {string} description
26  * @param {boolean} disabled
27  *
28  */
29 export const ObjectTypeDropdown = (props) => {
30 
31  const [objectTypeOptions, setObjectTypeOptions] = useState([]);
32  const [selectedObjectTypeOption, setSelectedObjectTypeOption] = useState();
33 
34  const {
35  id,
36  "data-testid": dataTestid,
37  adaptorId,
38  value,
39  requireEntity,
40  onChanged,
41  label,
42  description,
43  disabled,
44  ignoreAdaptive,
45  defaultFirstNonAdaptive,
46  autoSelect,
47  onError
48  } = props;
49 
50  const {objects, isLoading, error} = useRetrieveObjects({
51  adaptorId,
52  objectTypeId,
53  objectOptions
54  });
55 
56  useEffect(() => {
57  let objectTypeOptions = [];
58 
59  if (objects && objects.length > 0) {
60  let objectTypeObjects = objects;
61 
62  /* only allow objectTypeObjects that allowEntities to be in the dropdown */
63  if (requireEntity === true)
64  objectTypeObjects = objectTypeObjects.filter(objectTypeObject =>
65  objectTypeObject.getPropertyValue("allowEntity") === true);
66 
67  /* create dropdown options for each object type object */
68  objectTypeObjects.forEach(objectTypeObject => {
69  let objectId = objectTypeObject.getObjectId();
70  if (ignoreAdaptive && objectId && objectId.startsWith("_Adaptive"))
71  return;
72 
73  if (objectId) {
74  const option = {
75  key: objectId,
76  text: objectId,
77  description: objectTypeObject.getPropertyValue("description"),
78  objectTypeObject,
79  };
80 
81  objectTypeOptions.push(option);
82  }
83  });
84 
85  /* sort the object types alphabetically */
86  objectTypeOptions.sort((A, B) => {
87  const a = A.key.toLowerCase();
88  const b = B.key.toLowerCase();
89 
90  return a.localeCompare(b);
91  });
92  }
93 
94  setSelectedObjectTypeOption();
95  setObjectTypeOptions(objectTypeOptions);
96  }, [ignoreAdaptive, objects, requireEntity]);
97 
98  useEffect(() => {
99  let selectedObjectTypeOption;
100  if (autoSelect === false)
101  return;
102 
103  if (value) {
104  for (let i = 0; i < objectTypeOptions.length; i++) {
105  if (value === objectTypeOptions[i].key) {
106  selectedObjectTypeOption = objectTypeOptions[i];
107  break;
108  }
109  }
110  }
111 
112  if (!selectedObjectTypeOption) {
113  if (defaultFirstNonAdaptive) {
114  for (let i = 0; i < objectTypeOptions.length; i++) {
115  if (!objectTypeOptions[i].key.startsWith("_Adaptive")) {
116  selectedObjectTypeOption = objectTypeOptions[i];
117  break;
118  }
119  }
120 
121  /* can't find one, so choose first one */
122  if (!selectedObjectTypeOption)
123  selectedObjectTypeOption = objectTypeOptions[0];
124  } else
125  selectedObjectTypeOption = objectTypeOptions[0];
126  }
127 
128  if (selectedObjectTypeOption) {
129  if (onChanged)
130  onChanged(selectedObjectTypeOption.key, selectedObjectTypeOption.objectTypeObject);
131  setSelectedObjectTypeOption(selectedObjectTypeOption);
132  }
133  }, [autoSelect, defaultFirstNonAdaptive, objectTypeOptions, onChanged, value]);
134 
135  useEffect(() => {
136  for (let option of objectTypeOptions) {
137  if (option.key === value)
138  setSelectedObjectTypeOption(option);
139  }
140  }, [value, objectTypeOptions]);
141 
142  useEffect(() => {
143  if (error && onError) {
144  onError(error);
145  }
146 
147  // we don't want to call onError every time it onError changes, because
148  // the caller may have provided an anonymous function that changes on
149  // every render.
150  // eslint-disable-next-line react-hooks/exhaustive-deps
151  }, [error]);
152 
153  const onCreateObjectTypeOption = useCallback((value) => {
154  if (!value)
155  return;
156 
157  let options = [...objectTypeOptions];
158  const selectedObjectTypeOption = {
159  key: value,
160  text: value,
161  description: value + " Object Type"
162  };
163 
164  options.push(selectedObjectTypeOption);
165 
166  setObjectTypeOptions(options);
167  setSelectedObjectTypeOption(selectedObjectTypeOption);
168 
169  if (onChanged)
170  onChanged(selectedObjectTypeOption.key);
171  }, [objectTypeOptions, setObjectTypeOptions, setSelectedObjectTypeOption, onChanged]);
172 
173  const onSelectObjectType = useCallback((selectedObjectTypeOption) => {
174  if (selectedObjectTypeOption && onChanged)
175  onChanged(selectedObjectTypeOption.key, selectedObjectTypeOption.objectTypeObject);
176  }, [onChanged]);
177 
178  return (
179  <Autocomplete
180  id={id}
181  data-testid={dataTestid}
182  data-component-type="ObjectTypeDropdown"
183  label={label}
184  placeholder="Object Type"
185  description={description}
186  options={objectTypeOptions}
187  isLoading={isLoading}
188  value={selectedObjectTypeOption}
189  onChanged={onSelectObjectType}
190  onCreateOption={onCreateObjectTypeOption}
191  disabled={disabled || isLoading}
192  error={(error && error.message) ? error.message : undefined}
193  noOptionsText="No ObjectType"
194  />
195  );
196 };
