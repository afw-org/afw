15 /*
16  * The Call Adaptive Function expression tuple has the form:
17  *
18  * [ annotation, "callA", functionId, ...parameters ]
19  *
20  * The functionId parameter may be a string value, or an expression tuple
21  * that resolves to string dataType, which is the name of the adaptive
22  * function.
23  */
24 export const ExpressionTupleOperationCallAdaptive = (props) => {

30  const {functions} = useFunctions();
31  const {tuple} = props;
32  const {editable} = useOperational(props);
33 
34  useEffect(() => {
35  if (tuple && functions) {
36  const functionLabel = tuple[2];
37 
38  for (const f of functions) {
39  if (f.functionLabel === functionLabel || f.functionId === functionLabel) {
40  const params = f.parameters;
41  let isMulti = false;
42  const description = f.description;
43  if (params && params.length > 0) {
44  const minArgs = params[params.length - 1].minArgs;
45  if (minArgs > -1)
46  isMulti = true;
47  }
48

51  setDescription(description);

53  break;
54  }
55  }

64 
65  /*! \fixme if last parameter was removed, adjust the tuple to completely remove it. */
66  props.onChanged(tuple);
67  };
