38 /* Read a file into memory in a specifed pool. */
39 AFW_DECLARE(const afw_memory_t *)
40 afw_file_to_memory(
41  const afw_utf8_t * file_path,
42  apr_size_t file_size,
43  const afw_pool_t *p,
44  afw_scope_t * scope)
45 {
46  afw_memory_t *to_memory;
47  afw_byte_t *buff;
48  FILE *in;
49  apr_status_t rv;
50  apr_finfo_t finfo;
51  const afw_utf8_z_t *file_path_z;
52 
53  /* Allocate to_memory struct. */
54  to_memory = afw_pool_calloc_type(p, afw_memory_t, scope);
55 
56  /* Open file. */
57  file_path_z = afw_utf8_to_utf8_z(file_path, p, scope);
58  in = fopen(file_path_z, "r");
59  if (!in) {
60  AFW_THROW_ERROR_FZ(not_found, scope,
61  "Error opening %s errno %d", file_path_z, errno);
62  }
63 
64  AFW_TRY {
65 
66  /* Get size of file if not supplied. */
67  if (file_size == 0) {
68  rv = apr_stat(&finfo, file_path_z, APR_FINFO_SIZE,
69  afw_pool_get_apr_pool(p));
70  if (rv != APR_SUCCESS) {
71  AFW_THROW_ERROR_RV_Z(general, apr, rv,
72  "apr_stat() failed.", scope);
73  }
74  file_size = (size_t)finfo.size;
75  }
76  /* Create a buffer big enough for contents of file. */
77  buff = afw_pool_malloc(p, file_size, scope);
78 
79  /* Read file into buffer. */
80  to_memory->size = fread(buff, 1, file_size, in);
81  to_memory->ptr = buff;
82  if (to_memory->size < 1) {
83  AFW_THROW_ERROR_FZ(general, scope,
84  "Error reading %" AFW_UTF8_FMT ".",
85  AFW_UTF8_FMT_ARG(file_path));
86  }
87 
88  }
89 
90  AFW_FINALLY{
91  /* Close input file. */
92  fclose(in);
93  }
94 
95  AFW_ENDTRY;
96 
97  /* Return result. */
98  return to_memory;
99 
100 }

103 /* Write a file from memory. */
104 AFW_DECLARE(void) afw_file_from_memory(
105  const afw_utf8_t * file_path,
106  const afw_memory_t * from_memory,
107  afw_file_mode_t mode,
108  afw_scope_t * scope)
109 {
110  apr_pool_t *apr_p = afw_pool_get_apr_pool(scope->p);
111  apr_file_t *f;
112  afw_utf8_z_t *file_path_z;
113  apr_status_t rv;
114  const afw_utf8_z_t *step_z;
115  const afw_utf8_octet_t *i;
116  afw_utf8_octet_t *o;
117  afw_utf8_octet_t c;
118  afw_utf8_octet_t *last_slash;
119  afw_size_t count;
120 
121  /*
122  * Make afw_u8_z copy of name with '\' changed to '/'. Remember location
123  * of last slash.
124  */
125  file_path_z = afw_scope_calloc(file_path->len + 1, scope);
126  last_slash = NULL;
127  for (i = file_path->s, o = file_path_z, count = file_path->len;
128  count > 0;
129  i++, o++, count--)
130  {
131  *o = *i;
132  if (*o == '\\') *o = '/';
133  if (*o == '/') last_slash = o;
134  }
135  *o = 0;
136 
137 
138  /* Process based on file mode. */
139  switch (mode) {
140 
141  /* Mode write ok. */
142  case afw_file_mode_write:
143  break;
144 
145  /*
146  * For mode write_new, check that file does not already exist and that
147  * the directory stucture exists.
148  */
149  case afw_file_mode_write_new:
150  rv = apr_file_open(&f, file_path_z, APR_FOPEN_READ,
151  APR_FPROT_OS_DEFAULT, apr_p);
152  if (rv == APR_SUCCESS) {
153  apr_file_close(f);
154  AFW_THROW_ERROR_FZ(general, scope,
155  "File %s already exists.", file_path_z);
156  }
157  /* If there is a '/' in path, make sure all of the directories exist */
158  if (last_slash) {
159  o = last_slash + 1;
160  c = *o;
161  *o = 0;
162  step_z = "apr_dir_make_recursive()";
163  rv = apr_dir_make_recursive(file_path_z,
164  APR_FPROT_OS_DEFAULT, apr_p);
165  if (rv != APR_SUCCESS) goto error_apr;
166  *o = c;
167  }
168  break;
169 
170  /* Mode write_existing ok. */
171  case afw_file_mode_write_existing:
172  break;
173 
174  /* Other modes are invalid. */
175  default:
176  AFW_THROW_ERROR_FZ(general, scope, "Invalid mode %d.", mode);
177  };
178 
179  /* Open file for write. */
180  step_z = "apr_file_open()";
181  rv = apr_file_open(&f, file_path_z, APR_FOPEN_WRITE + APR_FOPEN_CREATE,
182  APR_FPROT_OS_DEFAULT, apr_p);
183  if (rv != APR_SUCCESS) goto error_apr;
184 
185  /* Write memory to file. */
186  step_z = "apr_file_write()";
187  count = from_memory->size;
188  rv = apr_file_write(f, from_memory->ptr, &count);
189  if (rv != APR_SUCCESS) goto error_apr;
190 
191  /* Truncate file to length written. */
192  step_z = "apr_file_trunc()";
193  rv = apr_file_trunc(f, from_memory->size);
194  if (rv != APR_SUCCESS) goto error_apr;
195 
196  /* Close file. */
197  step_z = "apr_file_close()";
198  rv = apr_file_close(f);
199  if (rv != APR_SUCCESS) goto error_apr;
200 
201  /* Return. */
202  return;
203 
204 error_apr:
205  AFW_THROW_ERROR_RV_FZ(general, apr, rv, scope,
206  "Error writing file %s - %s",
207  file_path_z, step_z);
208 }

252 AFW_DEFINE(const afw_adaptor_t *)
253 afw_file_adaptor_create_cede_p(
254  const afw_object_t *properties,
255  const afw_pool_t *p, afw_scope_t *scope)
256 {
257  afw_file_internal_adaptor_t *self;
258  afw_adaptor_t *adaptor;
259  const afw_utf8_t *content_type;
260  const char *c;
261  const char *last;
262  afw_utf8_octet_t *n;
263  const afw_utf8_octet_t *new_root;
264  afw_size_t len;
265  afw_boolean_t b;
266 
267  /* Create adaptor and process common properties. */
268  adaptor = afw_adaptor_impl_create_cede_p(
269  &impl_afw_adaptor_inf,
270  sizeof(afw_file_internal_adaptor_t),
271  properties, p, scope);
272  self = (afw_file_internal_adaptor_t *)adaptor;
273  p = self->pub.p;
274 
275  /* Get content_type parameters. */
276  content_type = afw_object_old_get_property_as_utf8(properties,
277  &afw_s_contentType, p, scope);
278  self->content_type = afw_environment_get_content_type(content_type,
279  scope);
280  if (!self->content_type)
281  {
282  afw_adaptor_impl_throw_property_invalid(adaptor,
283  &afw_s_contentType, scope);
284  }
285 
286  /* Get optional filename extension */
287  self->filename_suffix = afw_object_old_get_property_as_utf8(
288  properties, &afw_s_filenameSuffix, p, scope);
289  if (!self->filename_suffix) {
290  self->filename_suffix = &afw_s_a_empty_string;
291  }
292 
293  /* Get root from parameters. */
294  self->root = afw_object_old_get_property_as_utf8(properties,
295  &afw_s_root, p, scope);
296  if (!self->root) {
297  AFW_THROW_ERROR_Z(general,
298  "root parameter required for adaptor_type file", scope);
299  }
300 
301  /* If root is not "", all '\'s change to '/' with a trailing '/' */
302  if (self->root->len != 0) {
303  len = self->root->len;
304  n = afw_pool_calloc(p, len + 1, scope);
305  new_root = n;
306  for (c = self->root->s, last = c + len - 1;; c++, n++) {
307  *n = *c;
308  if (*n == '\\') *n = '/';
309  if (c == last) {
310  if (*n != '/') {
311  n++;
312  len++;
313  *n = '/';
314  }
315  break;
316  }
317  }
318  self->root = afw_utf8_create(new_root, len, p, scope);
319  }
320 
321  /* Make path for journal directory. */
322  self->journal_dir_path_z = afw_utf8_z_printf(p, scope,
323  "%" AFW_UTF8_FMT AFW_OBJECT_Q_OBJECT_TYPE_ID_JOURNAL_ENTRY "/",
324  AFW_UTF8_FMT_ARG(self->root));
325 
326 
327  /* Make path to journal lock file. */
328  self->journal_lock_file_path_z = afw_utf8_z_printf(p, scope,
329  "%" AFW_UTF8_FMT AFW_OBJECT_Q_OBJECT_TYPE_ID_JOURNAL_ENTRY "/journal_lock",
330  AFW_UTF8_FMT_ARG(self->root));
331 
332 
333  /* If isDevelopmentInput is true, provide appropriate object types. */
334  b = afw_object_old_get_property_as_boolean_deprecated(properties,
335  &afw_s_isDevelopmentInput, scope);
336  if (b) {
337  afw_adaptor_impl_set_supported_core_object_type(adaptor,
338  &afw_s__AdaptiveCollection_, true, true, scope);
339  afw_adaptor_impl_set_supported_core_object_type(adaptor,
340  &afw_s__AdaptiveDataTypeGenerate_, true, true, scope);
341  afw_adaptor_impl_set_supported_core_object_type(adaptor,
342  &afw_s__AdaptiveManifest_, true, true, scope);
343  afw_adaptor_impl_set_supported_core_object_type(adaptor,
344  &afw_s__AdaptiveFunctionGenerate_, true, true, scope);
345  afw_adaptor_impl_set_supported_core_object_type(adaptor,
346  &afw_s__AdaptiveObjectType_, true, true, scope);
347  afw_adaptor_impl_set_supported_core_object_type(adaptor,
348  &afw_s__AdaptiveValueMeta_, true, true, scope);
349  }
350 
351  /* Return adaptor. */
352  return adaptor;
353 }

439 /*
440  * Implementation of method retrieve_objects for interface
441  * afw_adaptor_session.
442  */
443 void
444 impl_afw_adaptor_session_retrieve_objects(
445  const afw_adaptor_session_t *instance,
446  const afw_adaptor_impl_request_t *impl_request,
447  const afw_utf8_t *object_type_id,
448  const afw_query_criteria_t *criteria,
449  void *context,
450  afw_object_cb_t callback,
451  const afw_object_t *adaptor_type_specific,
452  const afw_pool_t *p,
453  afw_scope_t *scope)
454 {
455  afw_file_internal_adaptor_session_t * self = (afw_file_internal_adaptor_session_t *)instance;
456  afw_file_internal_adaptor_t *adaptor = (afw_file_internal_adaptor_t *)self->adaptor;
457  const char *dirname_z;
458  const afw_utf8_t *full_path;
459  apr_dir_t *dir;
460  apr_status_t rv;
461  apr_finfo_t finfo;
462  char err[100];
463  const afw_pool_t *obj_p;
464  const afw_object_t *obj;
465  const afw_memory_t *raw;
466  const afw_utf8_t *object_id;
467  afw_size_t len;
468 
469  /* Open ObjectType's directory. */
470  dirname_z = apr_psprintf(afw_pool_get_apr_pool(p),
471  "%" AFW_UTF8_FMT "%" AFW_UTF8_FMT "/",
472  AFW_UTF8_FMT_ARG(adaptor->root),
473  AFW_UTF8_FMT_ARG(object_type_id));
474  rv = apr_dir_open(&dir, dirname_z, afw_pool_get_apr_pool(p));
475 
476  /* If not found, return no objects. */
477  if (APR_STATUS_IS_ENOENT(rv)) {
478  callback(NULL, context, scope);
479  return;
480  }
481 
482  /* If there is another problem, throw error. */
483  if (rv != APR_SUCCESS) {
484  AFW_THROW_ERROR_RV_Z(general, apr, rv, "apr_dir_open() failed.",
485  scope);
486  }
487 
488  /* Process each JSON object in directory. */
489  while (1) {
490 
491  /* Read next directory entry until there are no more.*/
492  rv = apr_dir_read(&finfo, APR_FINFO_SIZE + APR_FINFO_NAME, dir);
493  if (rv == APR_ENOENT || rv == 720018) break;
495  if (rv != APR_SUCCESS) {
496  apr_strerror(rv, &err[0], 100);
497  AFW_THROW_ERROR_RV_Z(general, apr, rv, "apr_dir_open() failed.",
498  scope);
499  }
500 
501  /* Skip ., .., and hidden files. */
502  if (*(finfo.name) == '.') {
503  continue;
504  }
505 
506  /* Create object from corresponding file. */
507  len = strlen(finfo.name);
508  if (adaptor->filename_suffix) {
509  if (len <= adaptor->filename_suffix->len ||
510  memcmp(finfo.name + (len - adaptor->filename_suffix->len),
511  adaptor->filename_suffix->s,
512  adaptor->filename_suffix->len) != 0)
513  {
514  continue;
515  }
516  len -= adaptor->filename_suffix->len;
517  }
518 
519 
520  /* Create pool for object and related memory. */
521  obj_p = afw_pool_create(p, scope);
522 
523  /* Determine object_id and full_path. */
524  object_id = afw_utf8_create(finfo.name, len, obj_p, scope);
525  full_path = afw_utf8_printf(obj_p, scope,
526  "%" AFW_UTF8_FMT "%" AFW_UTF8_FMT "/%" AFW_UTF8_FMT "%" AFW_UTF8_FMT,
527  AFW_UTF8_FMT_ARG(adaptor->root),
528  AFW_UTF8_FMT_ARG(object_type_id),
529  AFW_UTF8_FMT_ARG(object_id),
530  AFW_UTF8_FMT_ARG(adaptor->filename_suffix));
531 
532  /*
533  * Load file to memory and convert to object instance. Ceed control
534  * of obj_p to object.
535  */
536  raw = afw_file_to_memory(full_path, (apr_size_t)finfo.size,
537  obj_p, scope);
538  obj = afw_content_type_raw_to_object(
539  adaptor->content_type, raw, full_path, &adaptor->pub.adaptor_id,
540  object_type_id, object_id, true, obj_p, scope);
541 
542  /*
543  * If query criteria met, callback with object. Callback will release
544  * object. If callback returns true, prematurely stop retrieving.
545  */
546  if (afw_query_criteria_test_object(obj, criteria, p, scope)) {
547  if (callback(obj, context, scope)) {
548  break;
549  }
550  }
551 
552  /* If query criteria not met, release object. */
553  else {
554  afw_object_release(obj, scope);
555  }
556  }
557 
558  /* Close ObjectType's directory. */
559  rv = apr_dir_close(dir);
560  if (rv != APR_SUCCESS) {
561  AFW_THROW_ERROR_RV_Z(general, apr, rv, "apr_dir_close() failed.",
562  scope);
563  }
564 
565  /* Call callback one more time with NULL object pointer. */
566  callback(NULL, context, scope);
567 }
