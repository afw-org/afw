44  const [editMode, setEditMode] = useState();

50 
51  const {object, isLoading, error, savable} = useGetObject({ adaptorId, objectTypeId, objectId, objectOptions });
52  const {notification} = useApplication();
53  const match = useRouteMatch();
54  const location = useLocation();
55  const theme = useTheme();
56  const isMounted = useIsMounted();
57  const layoutRef = useRef();
58

66 
67  if (!match.isExact) {
68  /* we must have requested an embeddedObject */
69  const embeddedObject = location.pathname.split("/" + adaptorId + "/" + objectTypeId + "/" + objectId + "/")[1];
70  setEmbeddedObject(embeddedObject.split("/"));
71  } else {
72  setEmbeddedObject();
73  }
74  }
75  }, [match, location]);
76 
77  useEffect(() => onSelectObject(object), [onSelectObject, object]);

79  /**
80  * lightDismiss()
81  *
82  * Dismisses edit mode without discarding anything.
83  */
84  const lightDismiss = useCallback( () => {
85

87  setShowDismissWarning(false);
88  setEditMode(false);
89  }, [setSourceChanged, setShowDismissWarning]);
90 
91  /*
92  * confirmDismiss()
93  *
94  * Once the user confirms that they no wish to discard changes, the object changes
95  * are actually discarded, the parent is notified, and the warning is dismissed.
96  */
97  const confirmDismiss = () => {
98  setSourceChanged(false);
99  setShowDismissWarning(false);
100 
101  object.discardChanges();
102 
103  setEditMode(false);
104  };

106  /*
107  * onSave()
108  *
109  * When the user clicks on the "Save" mode, this routine will actually save the object back out to the store
110  * and notify the parent.
111  */
112  const onSave = useCallback( async () => {
113 
114  /* first let the layout do any last-second preparation for the save, such as parsing */
115  layoutRef.current.onFinalizeObject();
116 
117  try {
118  setIsProcessing(true);
119  const response = object.reconcile();
120  await response.result();
121 
122  if (isMounted()) {
123  setSourceChanged(false);
124  setShowObjectDifferencesModal(false);
125  setEditMode(false);
126 
127  notification({ message: "Object " + object.getObjectId() + " saved.", type: "success", response });
128  }
129  } catch (error) {
130  setShowObjectDifferencesModal(false);
131  notification({ message: error, type: "error" });
132  }
133 
134  if (isMounted())
135  setIsProcessing(false);
136  }, [layoutRef, object, setIsProcessing, setSourceChanged, setShowObjectDifferencesModal, setEditMode, notification, isMounted]);
137 
138  /*
139  * onReplace()
140  *
141  * The replace() operation allows the user to ignore the object state in the current adaptor and go ahead
142  * and replace its contents with the contents of the current object.
143  */
144  const onReplace = useCallback( async () => {
145  /* first let the layout do any last-second preparation for the save, such as parsing */
146  layoutRef.current.onFinalizeObject();
147 
148  try {
149  setIsProcessing(true);
150  const response = object.replace();
151  await response.result();
152 
153  if (isMounted()) {
154  setSourceChanged(false);
155  setEditMode(false);
156 
157  notification({ message: "Object " + object.getObjectId() + " replaced.", type: "success", response });
158  }
159  } catch (error) {
160  notification({ message: error, type: "error" });
161  }
162 
163  if (isMounted())
164  setIsProcessing(false);
165  }, [layoutRef, object, setIsProcessing, setSourceChanged, setEditMode, notification, isMounted]);
166 
167  /*
168  * onReconcile()
169  *
170  * The reconcile() operation allows the user to leverage the reconcilable metadata and save changes back out
171  * to the adaptor. If the object has been updated since it was last retrieved, it undergoes a reconcile process
172  * to attempt to merge our changes back in with the others.
173  */
174  const onReconcile = useCallback( async () => {
175  /* first let the layout do any last-second preparation for the save, such as parsing */
176  layoutRef.current.onFinalizeObject();
177 
178  try {
179  setIsProcessing(true);
180  const response = object.reconcile(false);
181  const result = await response.result();
182 
183  if (isMounted()) {
184  setSourceChanged(false);
185  setEditMode(false);
186 
187  notification({ message: "Object reconciled.", type: "success", response: result });
188  }
189  } catch (error) {
190  notification({ message: error, type: "error" });
191  }
192 
193  if (isMounted())
194  setIsProcessing(false);
195  }, [layoutRef, object, setIsProcessing, setSourceChanged, setEditMode, notification, isMounted]);
196 
197  /*
198  * onReview()
199  *
200  * The user may click to "Review" changes of an object to see the "diff" details and confirm that they
201  * definitely want to save the changes in question. This routine displays the Object Differences
202  * Modal for this purpose.

204  const onReview = useCallback( async () => {
205  /* first let the layout do any last-second preparation for the save, such as parsing */
206  layoutRef.current.onFinalizeObject();
207 
208  try {
209  setIsProcessing(true);
210  const response = object.reconcile(true);
211  const objectDifferences = await response.result();
212 
213  if (isMounted()) {
214  setObjectDifferences(objectDifferences);
215  setShowObjectDifferencesModal(true);
216  }
217  } catch (error) {
218  notification({ message: error, type: "error" });
219  }
220 
221  if (isMounted())
222  setIsProcessing(false);
223  }, [layoutRef, object, setIsProcessing, setObjectDifferences, setShowObjectDifferencesModal, notification, isMounted]);
224

226  * onDiscard()
227  *
228  * The user has intentionally selected to discard changes on the object by clicking on the
229  * Discard Changes button option.
230  */
231  const onDiscard = useCallback( () => {
232 
233  if (sourceChanged || savable)
234  setShowDismissWarning(true);
235  else
236  lightDismiss();
237  }, [sourceChanged, setShowDismissWarning, lightDismiss, savable]);

239  /**
240  * onSourceChanged()
241  *
242  * The object may not have changed, but the source has, if the user has used the CodeEditor.

244  const onSourceChanged = useCallback(() => {

250  *
251  * The user has requested new object options and wishes to refresh the object.

255  object.refresh(objectOptions);

257  notification({ message: error, type: "error" });

261  if (isLoading)
262  /* spin until the object is loaded */
263  return <Spinner size="large" label="Loading Object..." fullScreen={true} />;
264

269  <div style={{ textAlign: "center" }}>

278  return null;

281  let obj = object;

284  obj = object.getProperty(embeddedObject).getEmbeddingObject();

313  isBlocking={false}

346  <Spinner size="large" label="Processing..." fullScreen={true} />
347  }
348  <Prompt

352  "This Object has unsaved changes. Are you sure you want to leave?" }
353  />
354  </>
