36  * ): boolean;
37  *

67 /*
68  * Adaptive function: n_of
69  *
70  * afw_function_execute_n_of
71  *
72  * See afw_function_bindings.h for more information.
73  *
74  * integer n specifies the number of boolean conditions that follow that must
75  * evaluate to true for boolean true to be returned. If n is 0, true is
76  * returned. Once n conditions evaluate to true, true is returned and the
77  * remaining conditions remain unevaluated.
78  *
79  * This function is pure, so it will always return the same result
80  * given exactly the same parameters and has no side effects.
81  *
82  * Declaration:
83  *
84  * function n_of(
85  * n: integer,
86  * ...conditions: (list of boolean)
87  * ): boolean;
88  *
89  * Parameters:
90  *
91  * n - (integer)
92  *
93  * conditions - (0 or more boolean)
94  *
95  * Returns:
96  *
97  * (boolean)
98  *
99  * Errors thrown:
100  *
101  * arg_error - there are less than n conditions
102  */
103 const afw_value_t *
104 afw_function_execute_n_of(
105  afw_function_execute_t *x)
106 {
107  const afw_value_integer_t *n;
108  const afw_value_boolean_t *v;
109  afw_size_t i;
110  afw_integer_t trues;
111 
112  AFW_FUNCTION_EVALUATE_REQUIRED_DATA_TYPE_PARAMETER(n, 1, integer);
113 
114  /* Evaluated args until end or there are enough trues. */
115  for (i = 2, trues = 0; i <= x->argc &&
116  trues < n->internal; i++)
117  {
118  AFW_FUNCTION_EVALUATE_REQUIRED_DATA_TYPE_PARAMETER(v, i, boolean);
119  if (v->internal) {
120  trues++;
121  }
122  }
123 
124  /* Return true if there were enough trues. */
125  return (trues >= n->internal)
126  ? afw_value_true
127  : afw_value_false;
128 }
