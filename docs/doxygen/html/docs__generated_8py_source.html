498 # simple little function that takes paragraphs that likely came from XML and
499 # splits them on \n\n, into logical paragraphs. Then, on each paragraph, it
500 # splits on whitespace and rejoins with a single space. This is to ensure that
501 # the paragraphs are formatted nicely in the HTML output.
502 def get_paras(text):
503  if not text:
504  return None
505 
506  paras = re.split(r"\n *\n", text)
507  return "\n\n".join(map(lambda x: " ".join(x.split()), paras))
