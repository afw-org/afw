152 /**
153  *
154  * @param {*} object The object to subscribe to.
155  * @return [savable, changed]
156  *
157  * This hook adds event listeners to the object in order to detect state
158  * changes for when the object has changed and/or is savable.
159  */
160 export const useEventId = ({ object, property, eventId }) => {
161 
162  const [event, setEvent] = useState();
163  const isMounted = useIsMounted();
164 
165  useDebugValue(eventId + "(" + (object ? object.getObjectId() : property ? property.getName() : "[object]") + ")");
166 
167  useEffect(() => {
168  if (object && eventId) {
169  const onEvent = object.addEventListener(eventId,
170  (event) => {
171  if (isMounted()) {
172  if (eventId === "onSavable") {
173  setEvent(evt => {
174  if (!evt || evt.value !== event.value)
175  return event;
176  else
177  return evt;
178  });
179  } else
180  setEvent(event);
181  }
182  }
183  );
184 
185  return () => {
186  object.removeEventListener(eventId, onEvent);
187  };
188  } else if (property && eventId) {
189  const onEvent = property.addEventListener(eventId,
190  (event) => {
191  if (isMounted())
192  setEvent(event);
193  }
194  );
195 
196  return () => {
197  property.removeEventListener(eventId, onEvent);
198  };
199  }
200 
201  }, [object, property, eventId, isMounted]);
202 
203  return event;
204 };

261 
262  case "got":

264  ...state,
265  object: action.object,
266  isLoading: false,

268  refresh: undefined,
269  });
270

272  return ({

276  error: undefined,
277  refresh: undefined,
278  });
279 
280  case "error":
281  return ({
282  ...state,

288  case "savable":

291  return state;
292 
293  return ({
294  ...state,
295  savable: action.savable,
296  });
297 
298  case "discard":
299  return ({
300  ...state,
301  savable: false,

303 
304  case "refresh":
305  return ({
306  ...state,
307  isLoading: true,
308  object: undefined,
309  savable: false,

316  default:

322 /**
323  * useGetObject()
324  *
325  * This hook gets an adaptive object, returning the object, whether it
326  * is loading, or if an error occurred. Additionally, it will
327  * automatically trigger re-renders when the object has changed.
328  */
329 export const useGetObject = ({ objectUri, adaptorId, objectTypeId, objectId, objectOptions, modelOptions, cache = false }) => {
330 
331  const [state, dispatch] = useReducer(getObjectReducer, getObject_initialState);
332  const model = useModel();
333 
334  const key = (objectUri ? objectUri : ("/" + adaptorId + "/" + objectTypeId + "/" + objectId));
335  useDebugValue(key);
336 
337  const onRefresh = () => dispatch({ type: "refresh" });
338 
339  /* When the object has changed, register a new event listener */
340  useEffect(() => {
341  let onInvalidate, onSavable, onDiscardChanges;
342 
343  if (state.object) {
344  if (!(modelOptions && modelOptions.adaptiveObject === false)) {
345  onSavable = state.object.addEventListener("onSavable", (savable) => {
346  dispatch({ type: "savable", savable });
347  });
348  onDiscardChanges = state.object.addEventListener("onDiscardChanges", () => {
349  dispatch({ type: "discard" });
350  });
351  }
352 
353  onInvalidate = model.cacheOnInvalidate(key, (_key, _object) => {
354  if (_key && _object)
355  dispatch({ type: "invalidate", object: _object });
356  else
357  dispatch({ type: "refresh" });
358  });
359 
360  return () => {
361  if (onSavable)
362  state.object.removeEventListener("onSavable", onSavable);
363  if (onDiscardChanges)
364  state.object.removeEventListener("onDiscardChanges", onDiscardChanges);
365  if (onInvalidate)
366  model.cacheOffInvalidate(key, onInvalidate);
367  };
368  }
369 
370  }, [model, state.object, modelOptions, key]);
371 
372  /* Use the getObject() method to get the adaptive object, via model. */
373  useEffect(() => {
374  let response;
375 
376  const getObject = async () => {
377  dispatch({ type: "getting" });
378 
379  try {
380  let object;
381 
382  if (objectUri) {
383  response = model.getObjectWithUri({ uri: objectUri, objectOptions, modelOptions });
384  } else {
385  response = model.getObject({ objectTypeId, objectId, adaptorId, objectOptions, modelOptions });
386  }
387 
388  object = await response.object;
389 
390  if (cache)
391  model.cacheSet(key, object);
392 
393  if (!response.controller.signal.aborted)
394  dispatch({ type: "got", object });
395  } catch (error) {
396  if (!response || !response.controller.signal.aborted)
397  dispatch({ type: "error", error });
398  }
399  };
400 
401  if (model && key && cache && model.cacheGet(key)) {
402  dispatch({ type: "got", object: model.cacheGet(key) });
403  } else if (model && (objectUri || (adaptorId && objectTypeId && objectId))) {
404  getObject();
405  } else {
406  /* If we no longer have an adaptorId/objectTypeId/objectId, reset the state */
407  dispatch({ type: "reset" });
408  }
409 
410  return () => {
411  if (response) {
412  response.controller.abort();
413  }
414  };
415 
416 
417  }, [model, key, cache, objectUri, adaptorId, objectTypeId, objectId, objectOptions, modelOptions, state.refresh]);
418 
419  return { ...state, onRefresh };
420 };

487 /**
488  * useRetrieveObjects()
489  *
490  * This hook retrieves adaptive objects, returning the objects, whether
491  * they are loading, or if an error has occurred.
492  */
493 export const useRetrieveObjects = ({ adaptorId, objectTypeId, queryCriteria, objectOptions, modelOptions, cache = false }) => {
494 
495  const [state, dispatch] = useReducer(retrieveObjectsReducer, retrieveObjects_initialState);
496 
497  useDebugValue(state.isLoading ? "Loading" : state.error ? "Found Error" : state.objects ? (state.objects.length + " Objects") : "[Unknown]");
498 
499  const model = useModel();
500 
501  const onRefresh = () => dispatch({ type: "refresh" });
502 
503  const key = calcKey({ adaptorId, objectTypeId, queryCriteria, objectOptions, modelOptions });
504 
505  useEffect(() => {
506  let response;
507  let onInvalidate;
508 
509  const retrieveObjects = async () => {
510  dispatch({ type: "retrieving" });
511 
512  try {
513  response = model.retrieveObjects({
514  objectTypeId, adaptorId, queryCriteria, objectOptions, modelOptions
515  });
516 
517  const objects = await response.objects;
518 
519  if (cache && key)
520  model.cacheSet(key, objects);
521 
522  if (!response.controller.signal.aborted) {
523  dispatch({ type: "retrieved", objects });
524  }
525 
526  } catch (error) {
527  if (!response.controller.signal.aborted)
528  dispatch({ type: "error", error });
529  }
530  };
531 
532  if (cache && key && model.cacheGet(key) && !state.refresh) {
533  dispatch({ type: "retrieved_cache", objects: model.cacheGet(key) });
534  } else if (model && adaptorId && objectTypeId) {
535  retrieveObjects();
536  } else {
537  /* If we no longer have an adaptorId/objectTypeId, reset the state */
538  dispatch({ type: "reset" });
539  }
540 
541  return () => {
542  if (onInvalidate) {
543  model.cacheOffInvalidate(key, onInvalidate);
544  }
545  if (response) {
546  response.controller.abort();
547  }
548  };
549 
550  }, [model, cache, key, adaptorId, objectTypeId, queryCriteria, objectOptions, modelOptions, state.refresh]);
551 
552  useEffect(() => {
553  let onInvalidate;
554 
555  if (key && state.objects && cache) {
556  onInvalidate = model.cacheOnInvalidate(key, (_key, _objects) => {
557  if (_key && _objects)
558  dispatch({ type: "retrieved", objects: _objects });
559  else
560  dispatch({ type: "refresh" });
561  });
562  }
563 
564  return () => {
565  if (onInvalidate)
566  model.cacheOffInvalidate(onInvalidate);
567  };
568  }, [model, key, state.objects, cache]);
569 
570  return { ...state, onRefresh };
571 };
