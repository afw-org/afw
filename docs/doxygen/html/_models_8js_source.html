141 /**
142  * Models
143  *

148 export const Models = () => {
149 
150  const [state, dispatch] = useReducer(reducer, initialState);
151  const {pathname, hash} = useLocation();
152  const {application} = useAppCore();
153 
154  const {
155  objects: models = [],
156  isLoading: isLoadingModels,
157  onRefresh,
158  error: errorModels,
159  } = useRetrieveObjects({
160  adaptorId: state.adaptorId,
161  objectTypeId: "_AdaptiveModel_",
162  objectOptions: modelsObjectOptions,
163  });
164 
165  const {
166  object: model,
167  isLoading: isLoadingModel,
168  error: errorModel,
169  } = useGetObject({
170  adaptorId: state.adaptorId,
171  objectTypeId: "_AdaptiveModel_",
172  objectId: state.modelId,
173  objectOptions: modelObjectOptions,
174  });
175 
176  useEffect(() => {
177  if (application) {
178  let [, adaptorId, modelId,, objectType,, propertyType] = pathname.split("/").splice(2);
179 
180  if (!adaptorId)
181  adaptorId = application.getPropertyValue("defaultModelAdaptorId");
182 
183  dispatch({ type: "LOCATION", adaptorId, modelId, objectType, propertyType, hash });
184  }
185  }, [pathname, hash, application]);
186 
187  /* When multiple adaptors are used to store models, the user can select one */
188  const onSelectAdaptorId = (adaptorId) => dispatch({ type: "SELECT_ADAPTOR", adaptorId });
189 
190  return (
191  <div
192  id="admin-admin-models"
193  data-testid="admin-admin-models"

202  <Switch>
